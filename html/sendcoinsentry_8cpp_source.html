128 bool SendCoinsEntry::validate(interfaces::Node& node)
129 {
130  if (!model)
131  return false;
132 
133  // Check input validity
134  bool retval = true;
135 
136  // Skip checks for payment request
137  if (recipient.paymentRequest.IsInitialized())
138  return retval;
139 
140  if (!model->validateAddress(ui->payTo->text()))
141  {
142  ui->payTo->setValid(false);
143  retval = false;
144  }
145 
146  if (!ui->payAmount->validate())
147  {
148  retval = false;
149  }
150 
151  // Sending a zero amount is invalid
152  if (ui->payAmount->value(0) <= 0)
153  {
154  ui->payAmount->setValid(false);
155  retval = false;
156  }
157 
158  // Reject dust outputs:
159  if (retval && GUIUtil::isDust(node, ui->payTo->text(), ui->payAmount->value())) {
160  ui->payAmount->setValid(false);
161  retval = false;
162  }
163 
164  return retval;
165 }

195 void SendCoinsEntry::setValue(const SendCoinsRecipient &value)
196 {
197  recipient = value;
198 
199  if (recipient.paymentRequest.IsInitialized()) // payment request
200  {
201  if (recipient.authenticatedMerchant.isEmpty()) // unauthenticated
202  {
203  ui->payTo_is->setText(recipient.address);
204  ui->memoTextLabel_is->setText(recipient.message);
205  ui->payAmount_is->setValue(recipient.amount);
206  ui->payAmount_is->setReadOnly(true);
207  setCurrentWidget(ui->SendCoins_UnauthenticatedPaymentRequest);
208  }
209  else // authenticated
210  {
211  ui->payTo_s->setText(recipient.authenticatedMerchant);
212  ui->memoTextLabel_s->setText(recipient.message);
213  ui->payAmount_s->setValue(recipient.amount);
214  ui->payAmount_s->setReadOnly(true);
215  setCurrentWidget(ui->SendCoins_AuthenticatedPaymentRequest);
216  }
217  }
218  else // normal payment
219  {
220  // message
221  ui->messageTextLabel->setText(recipient.message);
222  ui->messageTextLabel->setVisible(!recipient.message.isEmpty());
223  ui->messageLabel->setVisible(!recipient.message.isEmpty());
224 
225  ui->addAsLabel->clear();
226  ui->payTo->setText(recipient.address); // this may set a label from addressbook
227  if (!recipient.label.isEmpty()) // if a label had been set from the addressbook, don't overwrite with an empty label
228  ui->addAsLabel->setText(recipient.label);
229  ui->payAmount->setValue(recipient.amount);
230  }
231 }
