204 class CompareTxMemPoolEntryByDescendantScore
205 {
206 public:
207  bool operator()(const CTxMemPoolEntry& a, const CTxMemPoolEntry& b) const
208  {
209  double a_mod_fee, a_size, b_mod_fee, b_size;
210 
211  GetModFeeAndSize(a, a_mod_fee, a_size);
212  GetModFeeAndSize(b, b_mod_fee, b_size);
213 
214  // Avoid division by rewriting (a/b > c/d) as (a*d > c*b).
215  double f1 = a_mod_fee * b_size;
216  double f2 = a_size * b_mod_fee;
217 
218  if (f1 == f2) {
219  return a.GetTime() >= b.GetTime();
220  }
221  return f1 < f2;
222  }

224  // Return the fee/size we're using for sorting this entry.
225  void GetModFeeAndSize(const CTxMemPoolEntry &a, double &mod_fee, double &size) const
226  {
227  // Compare feerate with descendants to feerate of the transaction, and

752  // It's almost certainly a logic bug if we don't clear out queuedTx before
753  // destruction, as we add to it while disconnecting blocks, and then we
754  // need to re-process remaining transactions to ensure mempool consistency.
755  // For now, assert() that we've emptied out this object on destruction.
756  // This assert() can always be removed if the reorg-processing code were
757  // to be refactored such that this assumption is no longer true (for
758  // instance if there was some other way we cleaned up the mempool after a
759  // reorg, besides draining this object).
760  ~DisconnectedBlockTransactions() { assert(queuedTx.empty()); }
761 
762  indexed_disconnected_transactions queuedTx;
763  uint64_t cachedInnerUsage = 0;
764 
765  // Estimate the overhead of queuedTx to be 6 pointers + an allocation, as
766  // no exact formula for boost::multi_index_contained is implemented.
767  size_t DynamicMemoryUsage() const {
768  return memusage::MallocUsage(sizeof(CTransactionRef) + 6 * sizeof(void*)) * queuedTx.size() + cachedInnerUsage;
769  }

771  void addTransaction(const CTransactionRef& tx)
772  {
773  queuedTx.insert(tx);
774  cachedInnerUsage += RecursiveDynamicUsage(tx);
775  }
776 
777  // Remove entries based on txid_index, and update memory usage.
778  void removeForBlock(const std::vector<CTransactionRef>& vtx)
779  {
780  // Short-circuit in the common case of a block being added to the tip
781  if (queuedTx.empty()) {
782  return;
783  }
784  for (auto const &tx : vtx) {
785  auto it = queuedTx.find(tx->GetHash());
786  if (it != queuedTx.end()) {
787  cachedInnerUsage -= RecursiveDynamicUsage(*it);
788  queuedTx.erase(it);
789  }
790  }
791  }
792 
793  // Remove an entry by insertion_order index, and update memory usage.
794  void removeEntry(indexed_disconnected_transactions::index<insertion_order>::type::iterator entry)
795  {
796  cachedInnerUsage -= RecursiveDynamicUsage(*entry);
797  queuedTx.get<insertion_order>().erase(entry);
798  }
