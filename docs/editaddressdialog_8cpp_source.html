94 void EditAddressDialog::accept()
95 {
96  if(!model)
97  return;
98 
99  if(!saveCurrentRow())
100  {
101  switch(model->getEditStatus())
102  {
103  case AddressTableModel::OK:
104  // Failed with unknown reason. Just reject.
105  break;
106  case AddressTableModel::NO_CHANGES:
107  // No changes were made during edit operation. Just reject.
108  break;
109  case AddressTableModel::INVALID_ADDRESS:
110  QMessageBox::warning(this, windowTitle(),
111  tr("The entered address \"%1\" is not a valid Bitcoin address.").arg(ui->addressEdit->text()),
112  QMessageBox::Ok, QMessageBox::Ok);
113  break;
114  case AddressTableModel::DUPLICATE_ADDRESS:
115  QMessageBox::warning(this, windowTitle(),
116  getDuplicateAddressWarning(),
117  QMessageBox::Ok, QMessageBox::Ok);
118  break;
119  case AddressTableModel::WALLET_UNLOCK_FAILURE:
120  QMessageBox::critical(this, windowTitle(),
121  tr("Could not unlock wallet."),
122  QMessageBox::Ok, QMessageBox::Ok);
123  break;
124  case AddressTableModel::KEY_GENERATION_FAILURE:
125  QMessageBox::critical(this, windowTitle(),
126  tr("New key generation failed."),
127  QMessageBox::Ok, QMessageBox::Ok);
128  break;
129 
130  }
131  return;
132  }
133  QDialog::accept();
134 }

144  return tr(
