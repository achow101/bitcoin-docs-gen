22 static feebumper::Result PreconditionChecks(interfaces::Chain::Lock& locked_chain, const CWallet* wallet, const CWalletTx& wtx, std::vector<std::string>& errors) EXCLUSIVE_LOCKS_REQUIRED(wallet->cs_wallet)
23 {
24  if (wallet->HasWalletSpend(wtx.GetHash())) {
25  errors.push_back("Transaction has descendants in the wallet");
26  return feebumper::Result::INVALID_PARAMETER;
27  }
28 
29  {
30  LOCK(mempool.cs);
31  auto it_mp = mempool.mapTx.find(wtx.GetHash());
32  if (it_mp != mempool.mapTx.end() && it_mp->GetCountWithDescendants() > 1) {
33  errors.push_back("Transaction has descendants in the mempool");
34  return feebumper::Result::INVALID_PARAMETER;
35  }
36  }
37 
38  if (wtx.GetDepthInMainChain(locked_chain) != 0) {
39  errors.push_back("Transaction has been mined, or is conflicted with a mined transaction");
40  return feebumper::Result::WALLET_ERROR;
41  }
42 
43  if (!SignalsOptInRBF(*wtx.tx)) {
44  errors.push_back("Transaction is not BIP 125 replaceable");
45  return feebumper::Result::WALLET_ERROR;
46  }
47 
48  if (wtx.mapValue.count("replaced_by_txid")) {
49  errors.push_back(strprintf("Cannot bump transaction %s which was already bumped by transaction %s", wtx.GetHash().ToString(), wtx.mapValue.at("replaced_by_txid")));
50  return feebumper::Result::WALLET_ERROR;
51  }
52 
53  // check that original tx consists entirely of our inputs
54  // if not, we can't bump the fee, because the wallet has no way of knowing the value of the other inputs (thus the fee)
55  if (!wallet->IsAllFromMe(*wtx.tx, ISMINE_SPENDABLE)) {
56  errors.push_back("Transaction contains inputs that don't belong to this wallet");
57  return feebumper::Result::WALLET_ERROR;
58  }
59 
60 
61  return feebumper::Result::OK;
62 }

78 Result CreateTransaction(const CWallet* wallet, const uint256& txid, const CCoinControl& coin_control, CAmount total_fee, std::vector<std::string>& errors,
79  CAmount& old_fee, CAmount& new_fee, CMutableTransaction& mtx)
80 {
81  auto locked_chain = wallet->chain().lock();
82  LOCK(wallet->cs_wallet);
83  errors.clear();
84  auto it = wallet->mapWallet.find(txid);
85  if (it == wallet->mapWallet.end()) {
86  errors.push_back("Invalid or non-wallet transaction id");
87  return Result::INVALID_ADDRESS_OR_KEY;
88  }
89  const CWalletTx& wtx = it->second;
90 
91  Result result = PreconditionChecks(*locked_chain, wallet, wtx, errors);
92  if (result != Result::OK) {
93  return result;
94  }
95 
96  // figure out which output was change
97  // if there was no change output or multiple change outputs, fail
98  int nOutput = -1;
99  for (size_t i = 0; i < wtx.tx->vout.size(); ++i) {
100  if (wallet->IsChange(wtx.tx->vout[i])) {
101  if (nOutput != -1) {
102  errors.push_back("Transaction has multiple change outputs");
103  return Result::WALLET_ERROR;
104  }
105  nOutput = i;
106  }
107  }
108  if (nOutput == -1) {
109  errors.push_back("Transaction does not have a change output");
110  return Result::WALLET_ERROR;
111  }
112 
113  // Calculate the expected size of the new transaction.
114  int64_t txSize = GetVirtualTransactionSize(*(wtx.tx));
115  const int64_t maxNewTxSize = CalculateMaximumSignedTxSize(*wtx.tx, wallet);
116  if (maxNewTxSize < 0) {
117  errors.push_back("Transaction contains inputs that cannot be signed");
118  return Result::INVALID_ADDRESS_OR_KEY;
119  }
120 
121  // calculate the old fee and fee-rate
122  old_fee = wtx.GetDebit(ISMINE_SPENDABLE) - wtx.tx->GetValueOut();
123  CFeeRate nOldFeeRate(old_fee, txSize);
124  CFeeRate nNewFeeRate;
125  // The wallet uses a conservative WALLET_INCREMENTAL_RELAY_FEE value to
126  // future proof against changes to network wide policy for incremental relay
127  // fee that our node may not be aware of.
128  CFeeRate walletIncrementalRelayFee = CFeeRate(WALLET_INCREMENTAL_RELAY_FEE);
129  if (::incrementalRelayFee > walletIncrementalRelayFee) {
130  walletIncrementalRelayFee = ::incrementalRelayFee;
131  }
132 
133  if (total_fee > 0) {
134  CAmount minTotalFee = nOldFeeRate.GetFee(maxNewTxSize) + ::incrementalRelayFee.GetFee(maxNewTxSize);
135  if (total_fee < minTotalFee) {
136  errors.push_back(strprintf("Insufficient totalFee, must be at least %s (oldFee %s + incrementalFee %s)",
137  FormatMoney(minTotalFee), FormatMoney(nOldFeeRate.GetFee(maxNewTxSize)), FormatMoney(::incrementalRelayFee.GetFee(maxNewTxSize))));
138  return Result::INVALID_PARAMETER;
139  }
140  CAmount requiredFee = GetRequiredFee(*wallet, maxNewTxSize);
141  if (total_fee < requiredFee) {
142  errors.push_back(strprintf("Insufficient totalFee (cannot be less than required fee %s)",
143  FormatMoney(requiredFee)));
144  return Result::INVALID_PARAMETER;
145  }
146  new_fee = total_fee;
147  nNewFeeRate = CFeeRate(total_fee, maxNewTxSize);
148  } else {
149  new_fee = GetMinimumFee(*wallet, maxNewTxSize, coin_control, mempool, ::feeEstimator, nullptr /* FeeCalculation */);
150  nNewFeeRate = CFeeRate(new_fee, maxNewTxSize);
151 
152  // New fee rate must be at least old rate + minimum incremental relay rate
153  // walletIncrementalRelayFee.GetFeePerK() should be exact, because it's initialized
154  // in that unit (fee per kb).
155  // However, nOldFeeRate is a calculated value from the tx fee/size, so
156  // add 1 satoshi to the result, because it may have been rounded down.
157  if (nNewFeeRate.GetFeePerK() < nOldFeeRate.GetFeePerK() + 1 + walletIncrementalRelayFee.GetFeePerK()) {
158  nNewFeeRate = CFeeRate(nOldFeeRate.GetFeePerK() + 1 + walletIncrementalRelayFee.GetFeePerK());
159  new_fee = nNewFeeRate.GetFee(maxNewTxSize);
160  }
161  }
162 
163  // Check that in all cases the new fee doesn't violate maxTxFee
164  if (new_fee > maxTxFee) {
165  errors.push_back(strprintf("Specified or calculated fee %s is too high (cannot be higher than maxTxFee %s)",
166  FormatMoney(new_fee), FormatMoney(maxTxFee)));
167  return Result::WALLET_ERROR;
168  }
169 
170  // check that fee rate is higher than mempool's minimum fee
171  // (no point in bumping fee if we know that the new tx won't be accepted to the mempool)
172  // This may occur if the user set TotalFee or paytxfee too low, if fallbackfee is too low, or, perhaps,
173  // in a rare situation where the mempool minimum fee increased significantly since the fee estimation just a
174  // moment earlier. In this case, we report an error to the user, who may use total_fee to make an adjustment.
175  CFeeRate minMempoolFeeRate = mempool.GetMinFee(gArgs.GetArg("-maxmempool", DEFAULT_MAX_MEMPOOL_SIZE) * 1000000);
176  if (nNewFeeRate.GetFeePerK() < minMempoolFeeRate.GetFeePerK()) {
177  errors.push_back(strprintf(
178  "New fee rate (%s) is lower than the minimum fee rate (%s) to get into the mempool -- "
179  "the totalFee value should be at least %s or the settxfee value should be at least %s to add transaction",
180  FormatMoney(nNewFeeRate.GetFeePerK()),
181  FormatMoney(minMempoolFeeRate.GetFeePerK()),
182  FormatMoney(minMempoolFeeRate.GetFee(maxNewTxSize)),
183  FormatMoney(minMempoolFeeRate.GetFeePerK())));
184  return Result::WALLET_ERROR;
185  }
186 
187  // Now modify the output to increase the fee.
188  // If the output is not large enough to pay the fee, fail.
189  CAmount nDelta = new_fee - old_fee;
190  assert(nDelta > 0);
191  mtx = CMutableTransaction{*wtx.tx};
192  CTxOut* poutput = &(mtx.vout[nOutput]);
193  if (poutput->nValue < nDelta) {
194  errors.push_back("Change output is too small to bump the fee");
195  return Result::WALLET_ERROR;
196  }
197 
198  // If the output would become dust, discard it (converting the dust to fee)
199  poutput->nValue -= nDelta;
200  if (poutput->nValue <= GetDustThreshold(*poutput, GetDiscardRate(*wallet, ::feeEstimator))) {
201  wallet->WalletLogPrintf("Bumping fee and discarding dust output\n");
202  new_fee += poutput->nValue;
203  mtx.vout.erase(mtx.vout.begin() + nOutput);
204  }
205 
206  // Mark new tx not replaceable, if requested.
207  if (!coin_control.m_signal_bip125_rbf.get_value_or(wallet->m_signal_rbf)) {
208  for (auto& input : mtx.vin) {
209  if (input.nSequence < 0xfffffffe) input.nSequence = 0xfffffffe;
210  }
211  }
212 
213 
214  return Result::OK;
215 }

223 Result CommitTransaction(CWallet* wallet, const uint256& txid, CMutableTransaction&& mtx, std::vector<std::string>& errors, uint256& bumped_txid)
224 {
225  auto locked_chain = wallet->chain().lock();
226  LOCK(wallet->cs_wallet);
227  if (!errors.empty()) {
228  return Result::MISC_ERROR;
229  }
230  auto it = txid.IsNull() ? wallet->mapWallet.end() : wallet->mapWallet.find(txid);
231  if (it == wallet->mapWallet.end()) {
232  errors.push_back("Invalid or non-wallet transaction id");
233  return Result::MISC_ERROR;
234  }
235  CWalletTx& oldWtx = it->second;
236 
237  // make sure the transaction still has no descendants and hasn't been mined in the meantime
238  Result result = PreconditionChecks(*locked_chain, wallet, oldWtx, errors);
239  if (result != Result::OK) {
240  return result;
241  }
242 
243  // commit/broadcast the tx
244  CTransactionRef tx = MakeTransactionRef(std::move(mtx));
245  mapValue_t mapValue = oldWtx.mapValue;
246  mapValue["replaces_txid"] = oldWtx.GetHash().ToString();
247 
248  CReserveKey reservekey(wallet);
249  CValidationState state;
250  if (!wallet->CommitTransaction(tx, std::move(mapValue), oldWtx.vOrderForm, reservekey, g_connman.get(), state)) {
251  // NOTE: CommitTransaction never returns false, so this should never happen.
252  errors.push_back(strprintf("The transaction was rejected: %s", FormatStateMessage(state)));
253  return Result::WALLET_ERROR;
254  }
255 
256  bumped_txid = tx->GetHash();
257  if (state.IsInvalid()) {
258  // This can happen if the mempool rejected the transaction. Report
259  // what happened in the "errors" response.
260  errors.push_back(strprintf("Error: The transaction was rejected: %s", FormatStateMessage(state)));
261  }
262 
263  // mark the original tx as bumped
264  if (!wallet->MarkReplaced(oldWtx.GetHash(), bumped_txid)) {
265  // TODO: see if JSON-RPC has a standard way of returning a response
266  // along with an exception. It would be good to return information about
267  // wtxBumped to the caller even if marking the original transaction
268  // replaced does not succeed for some reason.
269  errors.push_back("Created new bumpfee transaction but could not mark the original transaction as replaced");
270  }
271  return Result::OK;
272 }
