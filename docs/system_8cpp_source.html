101 class CInit
102 {
103 public:
104  CInit()
105  {
106  // Init OpenSSL library multithreading support
107  ppmutexOpenSSL.reset(new CCriticalSection[CRYPTO_num_locks()]);
108  CRYPTO_set_locking_callback(locking_callback);
109 
110  // OpenSSL can optionally load a config file which lists optional loadable modules and engines.
111  // We don't use them so we don't require the config. However some of our libs may call functions
112  // which attempt to load the config file, possibly resulting in an exit() or crash if it is missing
113  // or corrupt. Explicitly tell OpenSSL not to try to load the file. The result for our libs will be
114  // that the config appears to have been loaded and there are no modules/engines available.
115  OPENSSL_no_config();
116 
117 #ifdef WIN32
118  // Seed OpenSSL PRNG with current contents of the screen
119  RAND_screen();
120 #endif
121 
122  // Seed OpenSSL PRNG with performance counter
123  RandAddSeed();
124  }
125  ~CInit()
126  {
127  // Securely erase the memory used by the PRNG
128  RAND_cleanup();
129  // Shutdown OpenSSL library multithreading support
130  CRYPTO_set_locking_callback(nullptr);
131  // Clear the set of locks now to maintain symmetry with the constructor.
132  ppmutexOpenSSL.reset();
133  }

146 bool LockDirectory(const fs::path& directory, const std::string lockfile_name, bool probe_only)
147 {
148  std::lock_guard<std::mutex> ulock(cs_dir_locks);
149  fs::path pathLockFile = directory / lockfile_name;
150 
151  // If a lock for this directory already exists in the map, don't try to re-lock it
152  if (dir_locks.count(pathLockFile.string())) {
153  return true;
154  }
155 
156  // Create empty lock file if it doesn't exist.
157  FILE* file = fsbridge::fopen(pathLockFile, "a");
158  if (file) fclose(file);
159  auto lock = MakeUnique<fsbridge::FileLock>(pathLockFile);
160  if (!lock->TryLock()) {
161  return error("Error while attempting to lock directory %s: %s", directory.string(), lock->GetReason());
162  }
163  if (!probe_only) {
164  // Lock successful and we're not just probing, put it into the map
165  dir_locks.emplace(pathLockFile.string(), std::move(lock));
166  }
167  return true;
168 }

334 static bool InterpretNegatedOption(std::string& key, std::string& val)
335 {
336  assert(key[0] == '-');
337 
338  size_t option_index = key.find('.');
339  if (option_index == std::string::npos) {
340  option_index = 1;
341  } else {
342  ++option_index;
343  }
344  if (key.substr(option_index, 2) == "no") {
345  bool bool_val = InterpretBool(val);
346  key.erase(option_index, 2);
347  if (!bool_val ) {
348  // Double negatives like -nofoo=0 are supported (but discouraged)
349  LogPrintf("Warning: parsed potentially confusing double-negative %s=%s\n", key, val);
350  val = "1";
351  } else {
352  return true;
353  }
354  }
355  return false;

432 bool ArgsManager::ParseParameters(int argc, const char* const argv[], std::string& error)
433 {
434  LOCK(cs_args);
435  m_override_args.clear();
436 
437  for (int i = 1; i < argc; i++) {
438  std::string key(argv[i]);
439  std::string val;
440  size_t is_index = key.find('=');
441  if (is_index != std::string::npos) {
442  val = key.substr(is_index + 1);
443  key.erase(is_index);
444  }
445 #ifdef WIN32
446  std::transform(key.begin(), key.end(), key.begin(), ::tolower);
447  if (key[0] == '/')
448  key[0] = '-';
449 #endif
450 
451  if (key[0] != '-')
452  break;
453 
454  // Transform --foo to -foo
455  if (key.length() > 1 && key[1] == '-')
456  key.erase(0, 1);
457 
458  // Check for -nofoo
459  if (InterpretNegatedOption(key, val)) {
460  m_override_args[key].clear();
461  } else {
462  m_override_args[key].push_back(val);
463  }
464 
465  // Check that the arg is known
466  if (!(IsSwitchChar(key[0]) && key.size() == 1)) {
467  if (!IsArgKnown(key)) {
468  error = strprintf("Invalid parameter %s", key.c_str());
469  return false;
470  }
471  }
472  }
473 
474  // we do not allow -includeconf from command line, so we clear it here
475  auto it = m_override_args.find("-includeconf");
476  if (it != m_override_args.end()) {
477  if (it->second.size() > 0) {
478  for (const auto& ic : it->second) {
479  error += "-includeconf cannot be used from commandline; -includeconf=" + ic + "\n";
480  }
481  return false;
482  }
483  }
484  return true;
485 }

844 static bool GetConfigOptions(std::istream& stream, std::string& error, std::vector<std::pair<std::string, std::string>>& options, std::set<std::string>& sections)
845 {
846  std::string str, prefix;
847  std::string::size_type pos;
848  int linenr = 1;
849  while (std::getline(stream, str)) {
850  bool used_hash = false;
851  if ((pos = str.find('#')) != std::string::npos) {
852  str = str.substr(0, pos);
853  used_hash = true;
854  }
855  const static std::string pattern = " \t\r\n";
856  str = TrimString(str, pattern);
857  if (!str.empty()) {
858  if (*str.begin() == '[' && *str.rbegin() == ']') {
859  const std::string section = str.substr(1, str.size() - 2);
860  sections.insert(section);
861  prefix = section + '.';
862  } else if (*str.begin() == '-') {
863  error = strprintf("parse error on line %i: %s, options in configuration file must be specified without leading -", linenr, str);
864  return false;
865  } else if ((pos = str.find('=')) != std::string::npos) {
866  std::string name = prefix + TrimString(str.substr(0, pos), pattern);
867  std::string value = TrimString(str.substr(pos + 1), pattern);
868  if (used_hash && name == "rpcpassword") {
869  error = strprintf("parse error on line %i, using # in rpcpassword can be ambiguous and should be avoided", linenr);
870  return false;
871  }
872  options.emplace_back(name, value);
873  if ((pos = name.rfind('.')) != std::string::npos) {
874  sections.insert(name.substr(0, pos));
875  }
876  } else {
877  error = strprintf("parse error on line %i: %s", linenr, str);
878  if (str.size() >= 2 && str.substr(0, 2) == "no") {
879  error += strprintf(", if you intended to specify a negated option, use %s=1 instead", str);
880  }
881  return false;
882  }
883  }
884  ++linenr;
885  }
886  return true;
887 }

1063 {
