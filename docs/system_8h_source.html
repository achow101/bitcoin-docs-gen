333 template <typename Callable> void TraceThread(const char* name, Callable func)
334 {
335  std::string s = strprintf("bitcoin-%s", name);
336  RenameThread(s.c_str());
337  try
338  {
339  LogPrintf("%s thread start\n", name);
340  func();
341  LogPrintf("%s thread exit\n", name);
342  }
343  catch (const boost::thread_interrupted&)

345  LogPrintf("%s thread interrupt\n", name);

348  catch (const std::exception& e) {
