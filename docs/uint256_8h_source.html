17class base_blob
18{
19protected:

22public:

31 bool IsNull() const

33 for (int i = 0; i < WIDTH; i++)

83 uint64_t GetUint64(int pos) const
84 {
85 const uint8_t* ptr = m_data + pos * 8;
86 return ((uint64_t)ptr[0]) | \
87 ((uint64_t)ptr[1]) << 8 | \
88 ((uint64_t)ptr[2]) << 16 | \
