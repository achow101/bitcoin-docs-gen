30 QList<TransactionRecord> TransactionRecord::decomposeTransaction(const interfaces::WalletTx& wtx)
31 {
32  QList<TransactionRecord> parts;
33  int64_t nTime = wtx.time;
34  CAmount nCredit = wtx.credit;
35  CAmount nDebit = wtx.debit;
36  CAmount nNet = nCredit - nDebit;
37  uint256 hash = wtx.tx->GetHash();
38  std::map<std::string, std::string> mapValue = wtx.value_map;
39 
40  if (nNet > 0 || wtx.is_coinbase)
41  {
42  //
43  // Credit
44  //
45  for(unsigned int i = 0; i < wtx.tx->vout.size(); i++)
46  {
47  const CTxOut& txout = wtx.tx->vout[i];
48  isminetype mine = wtx.txout_is_mine[i];
49  if(mine)
50  {
51  TransactionRecord sub(hash, nTime);
52  CTxDestination address;
53  sub.idx = i; // vout index
54  sub.credit = txout.nValue;
55  sub.involvesWatchAddress = mine & ISMINE_WATCH_ONLY;
56  if (wtx.txout_address_is_mine[i])
57  {
58  // Received by Bitcoin Address
59  sub.type = TransactionRecord::RecvWithAddress;
60  sub.address = EncodeDestination(wtx.txout_address[i]);
61  }
62  else
63  {
64  // Received by IP connection (deprecated features), or a multisignature or other non-simple transaction
65  sub.type = TransactionRecord::RecvFromOther;
66  sub.address = mapValue["from"];
67  }
68  if (wtx.is_coinbase)
69  {
70  // Generated
71  sub.type = TransactionRecord::Generated;
72  }
73 
74  parts.append(sub);
75  }
76  }
77  }
78  else
79  {
80  bool involvesWatchAddress = false;
81  isminetype fAllFromMe = ISMINE_SPENDABLE;
82  for (const isminetype mine : wtx.txin_is_mine)
83  {
84  if(mine & ISMINE_WATCH_ONLY) involvesWatchAddress = true;
85  if(fAllFromMe > mine) fAllFromMe = mine;
86  }
87 
88  isminetype fAllToMe = ISMINE_SPENDABLE;
89  for (const isminetype mine : wtx.txout_is_mine)
90  {
91  if(mine & ISMINE_WATCH_ONLY) involvesWatchAddress = true;
92  if(fAllToMe > mine) fAllToMe = mine;
93  }
94 
95  if (fAllFromMe && fAllToMe)
96  {
97  // Payment to self
98  CAmount nChange = wtx.change;
99 
100  parts.append(TransactionRecord(hash, nTime, TransactionRecord::SendToSelf, "",
101  -(nDebit - nChange), nCredit - nChange));
102  parts.last().involvesWatchAddress = involvesWatchAddress; // maybe pass to TransactionRecord as constructor argument
103  }
104  else if (fAllFromMe)
105  {
106  //
107  // Debit
108  //
109  CAmount nTxFee = nDebit - wtx.tx->GetValueOut();
110 
111  for (unsigned int nOut = 0; nOut < wtx.tx->vout.size(); nOut++)
112  {
113  const CTxOut& txout = wtx.tx->vout[nOut];
114  TransactionRecord sub(hash, nTime);
115  sub.idx = nOut;
116  sub.involvesWatchAddress = involvesWatchAddress;
117 
118  if(wtx.txout_is_mine[nOut])
119  {
120  // Ignore parts sent to self, as this is usually the change
121  // from a transaction sent back to our own address.
122  continue;
123  }
124 
125  if (!boost::get<CNoDestination>(&wtx.txout_address[nOut]))
126  {
127  // Sent to Bitcoin Address
128  sub.type = TransactionRecord::SendToAddress;
129  sub.address = EncodeDestination(wtx.txout_address[nOut]);
130  }
131  else
132  {
133  // Sent to IP, or other non-address transaction like OP_EVAL
134  sub.type = TransactionRecord::SendToOther;
135  sub.address = mapValue["to"];
136  }
137 
138  CAmount nValue = txout.nValue;
139  /* Add fee to first output */
140  if (nTxFee > 0)
141  {
142  nValue += nTxFee;
143  nTxFee = 0;
144  }
145  sub.debit = -nValue;
146 
147  parts.append(sub);
148  }
149  }
150  else
151  {
152  //
153  // Mixed debit transaction, can't break down payees
154  //
155  parts.append(TransactionRecord(hash, nTime, TransactionRecord::Other, "", nNet, 0));
156  parts.last().involvesWatchAddress = involvesWatchAddress;
157  }
158  }
159 
160  return parts;
161 }

163 void TransactionRecord::updateStatus(const interfaces::WalletTxStatus& wtx, int numBlocks, int64_t block_time)
164 {
165  // Determine transaction status
166 
167  // Sort order, unrecorded transactions sort to the top
168  status.sortKey = strprintf("%010d-%01d-%010u-%03d",
169  wtx.block_height,
170  wtx.is_coinbase ? 1 : 0,
171  wtx.time_received,
172  idx);
173  status.countsForBalance = wtx.is_trusted && !(wtx.blocks_to_maturity > 0);
174  status.depth = wtx.depth_in_main_chain;
175  status.cur_num_blocks = numBlocks;
176 
177  const bool up_to_date = ((int64_t)QDateTime::currentMSecsSinceEpoch() / 1000 - block_time < MAX_BLOCK_TIME_GAP);
178  if (up_to_date && !wtx.is_final) {
179  if (wtx.lock_time < LOCKTIME_THRESHOLD) {
180  status.status = TransactionStatus::OpenUntilBlock;
181  status.open_for = wtx.lock_time - numBlocks;
182  }
183  else
184  {
185  status.status = TransactionStatus::OpenUntilDate;
186  status.open_for = wtx.lock_time;
187  }
188  }
189  // For generated transactions, determine maturity
190  else if(type == TransactionRecord::Generated)
191  {
192  if (wtx.blocks_to_maturity > 0)
193  {
194  status.status = TransactionStatus::Immature;
195 
196  if (wtx.is_in_main_chain)
197  {
198  status.matures_in = wtx.blocks_to_maturity;
199  }
200  else
201  {
202  status.status = TransactionStatus::NotAccepted;
203  }
204  }
205  else
206  {
207  status.status = TransactionStatus::Confirmed;
208  }
209  }
210  else
211  {
212  if (status.depth < 0)
213  {
214  status.status = TransactionStatus::Conflicted;
215  }
216  else if (status.depth == 0)
217  {
218  status.status = TransactionStatus::Unconfirmed;
219  if (wtx.is_abandoned)
220  status.status = TransactionStatus::Abandoned;
221  }
222  else if (status.depth < RecommendedNumConfirmations)
223  {
224  status.status = TransactionStatus::Confirming;
225  }
226  else
227  {
228  status.status = TransactionStatus::Confirmed;
229  }
230  }
231  status.needsUpdate = false;
232 }
