173  std::string strMessage = request.params[2].get_str();
174 
175  CTxDestination destination = DecodeDestination(strAddress);
176  if (!IsValidDestination(destination)) {
177  throw JSONRPCError(RPC_TYPE_ERROR, "Invalid address");
178  }
179 
180  const CKeyID *keyID = boost::get<CKeyID>(&destination);
181  if (!keyID) {
182  throw JSONRPCError(RPC_TYPE_ERROR, "Address does not refer to key");
183  }
184 
185  bool fInvalid = false;
186  std::vector<unsigned char> vchSig = DecodeBase64(strSign.c_str(), &fInvalid);
187 
188  if (fInvalid)
189  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Malformed base64 encoding");
190 
191  CHashWriter ss(SER_GETHASH, 0);
192  ss << strMessageMagic;
193  ss << strMessage;
194 
195  CPubKey pubkey;
196  if (!pubkey.RecoverCompact(ss.GetHash(), vchSig))
197  return false;
198 
199  return (pubkey.GetID() == *keyID);
