68 class WorkQueue
69 {
70 private:
72  Mutex cs;
73  std::condition_variable cond;
74  std::deque<std::unique_ptr<WorkItem>> queue;
75  bool running;
76  size_t maxDepth;
77 
78 public:
79  explicit WorkQueue(size_t _maxDepth) : running(true),
80  maxDepth(_maxDepth)
81  {
82  }
85  ~WorkQueue()
86  {
87  }
89  bool Enqueue(WorkItem* item)
90  {
91  LOCK(cs);
92  if (queue.size() >= maxDepth) {
93  return false;
94  }
95  queue.emplace_back(std::unique_ptr<WorkItem>(item));
96  cond.notify_one();
97  return true;
98  }
100  void Run()
101  {
102  while (true) {
103  std::unique_ptr<WorkItem> i;
104  {
105  WAIT_LOCK(cs, lock);
106  while (running && queue.empty())
107  cond.wait(lock);
108  if (!running)
109  break;
110  i = std::move(queue.front());
111  queue.pop_front();
112  }
113  (*i)();
114  }
115  }
117  void Interrupt()
118  {
119  LOCK(cs);
120  running = false;
121  cond.notify_all();
122  }

212 static void http_request_cb(struct evhttp_request* req, void* arg)
213 {
214  // Disable reading to work around a libevent bug, fixed in 2.2.0.
215  if (event_get_version_number() >= 0x02010600 && event_get_version_number() < 0x02020001) {
216  evhttp_connection* conn = evhttp_request_get_connection(req);
217  if (conn) {
218  bufferevent* bev = evhttp_connection_get_bufferevent(conn);
219  if (bev) {
220  bufferevent_disable(bev, EV_READ);
221  }
222  }
223  }
224  std::unique_ptr<HTTPRequest> hreq(new HTTPRequest(req));
225 
226  // Early address-based allow check
227  if (!ClientAllowed(hreq->GetPeer())) {
228  LogPrint(BCLog::HTTP, "HTTP request from %s rejected: Client network is not allowed RPC access\n",
229  hreq->GetPeer().ToString());
230  hreq->WriteReply(HTTP_FORBIDDEN);
231  return;
232  }
233 
234  // Early reject unknown HTTP methods
235  if (hreq->GetRequestMethod() == HTTPRequest::UNKNOWN) {
236  LogPrint(BCLog::HTTP, "HTTP request from %s rejected: Unknown HTTP request method\n",
237  hreq->GetPeer().ToString());
238  hreq->WriteReply(HTTP_BADMETHOD);
239  return;
240  }
241 
242  LogPrint(BCLog::HTTP, "Received a %s request for %s from %s\n",
243  RequestMethodString(hreq->GetRequestMethod()), SanitizeString(hreq->GetURI(), SAFE_CHARS_URI).substr(0, 100), hreq->GetPeer().ToString());
244 
245  // Find registered handler for prefix
246  std::string strURI = hreq->GetURI();
247  std::string path;
248  std::vector<HTTPPathHandler>::const_iterator i = pathHandlers.begin();
249  std::vector<HTTPPathHandler>::const_iterator iend = pathHandlers.end();
250  for (; i != iend; ++i) {
251  bool match = false;
252  if (i->exactMatch)
253  match = (strURI == i->prefix);
254  else
255  match = (strURI.substr(0, i->prefix.size()) == i->prefix);
256  if (match) {
257  path = strURI.substr(i->prefix.size());
258  break;
259  }
260  }
261 
262  // Dispatch to worker thread
263  if (i != iend) {
264  std::unique_ptr<HTTPWorkItem> item(new HTTPWorkItem(std::move(hreq), path, i->handler));
265  assert(workQueue);
266  if (workQueue->Enqueue(item.get()))
267  item.release(); /* if true, queue took ownership */
268  else {
269  LogPrintf("WARNING: request rejected because http work queue depth exceeded, it can be increased with the -rpcworkqueue= setting\n");
270  item->req->WriteReply(HTTP_INTERNAL, "Work queue depth exceeded");
271  }
272  } else {
273  hreq->WriteReply(HTTP_NOTFOUND);
274  }
275 }

296 static bool HTTPBindAddresses(struct evhttp* http)
297 {
298  int http_port = gArgs.GetArg("-rpcport", BaseParams().RPCPort());
299  std::vector<std::pair<std::string, uint16_t> > endpoints;
300 
301  // Determine what addresses to bind to
302  if (!(gArgs.IsArgSet("-rpcallowip") && gArgs.IsArgSet("-rpcbind"))) { // Default to loopback if not allowing external IPs
303  endpoints.push_back(std::make_pair("::1", http_port));
304  endpoints.push_back(std::make_pair("127.0.0.1", http_port));
305  if (gArgs.IsArgSet("-rpcallowip")) {
306  LogPrintf("WARNING: option -rpcallowip was specified without -rpcbind; this doesn't usually make sense\n");
307  }
308  if (gArgs.IsArgSet("-rpcbind")) {
309  LogPrintf("WARNING: option -rpcbind was ignored because -rpcallowip was not specified, refusing to allow everyone to connect\n");
310  }
311  } else if (gArgs.IsArgSet("-rpcbind")) { // Specific bind address
312  for (const std::string& strRPCBind : gArgs.GetArgs("-rpcbind")) {
313  int port = http_port;
314  std::string host;
315  SplitHostPort(strRPCBind, port, host);
316  endpoints.push_back(std::make_pair(host, port));
317  }
318  }
319 
320  // Bind addresses
321  for (std::vector<std::pair<std::string, uint16_t> >::iterator i = endpoints.begin(); i != endpoints.end(); ++i) {
322  LogPrint(BCLog::HTTP, "Binding RPC on address %s port %i\n", i->first, i->second);
323  evhttp_bound_socket *bind_handle = evhttp_bind_socket_with_handle(http, i->first.empty() ? nullptr : i->first.c_str(), i->second);
324  if (bind_handle) {
325  CNetAddr addr;
326  if (i->first.empty() || (LookupHost(i->first.c_str(), addr, false) && addr.IsBindAny())) {
327  LogPrintf("WARNING: the RPC server is not safe to expose to untrusted networks such as the public internet\n");
328  }
329  boundSockets.push_back(bind_handle);
330  } else {
331  LogPrintf("Binding RPC on address %s port %i failed.\n", i->first, i->second);
332  }
333  }
334  return !boundSockets.empty();
335 }

357 bool InitHTTPServer()
358 {
359  if (!InitHTTPAllowList())
360  return false;
361 
362  // Redirect libevent's logging to our own log
363  event_set_log_callback(&libevent_log_cb);
364  // Update libevent's log handling. Returns false if our version of
365  // libevent doesn't support debug logging, in which case we should
366  // clear the BCLog::LIBEVENT flag.
367  if (!UpdateHTTPServerLogging(LogInstance().WillLogCategory(BCLog::LIBEVENT))) {
368  LogInstance().DisableCategory(BCLog::LIBEVENT);
369  }
370 
371 #ifdef WIN32
372  evthread_use_windows_threads();
373 #else
374  evthread_use_pthreads();
375 #endif
376 
377  raii_event_base base_ctr = obtain_event_base();
378 
379  /* Create a new evhttp object to handle requests. */
380  raii_evhttp http_ctr = obtain_evhttp(base_ctr.get());
381  struct evhttp* http = http_ctr.get();
382  if (!http) {
383  LogPrintf("couldn't create evhttp. Exiting.\n");
384  return false;
385  }
386 
387  evhttp_set_timeout(http, gArgs.GetArg("-rpcservertimeout", DEFAULT_HTTP_SERVER_TIMEOUT));
388  evhttp_set_max_headers_size(http, MAX_HEADERS_SIZE);
389  evhttp_set_max_body_size(http, MAX_SIZE);
390  evhttp_set_gencb(http, http_request_cb, nullptr);
391 
392  if (!HTTPBindAddresses(http)) {
393  LogPrintf("Unable to bind any endpoint for RPC server\n");
394  return false;
395  }
396 
397  LogPrint(BCLog::HTTP, "Initialized HTTP server\n");
398  int workQueueDepth = std::max((long)gArgs.GetArg("-rpcworkqueue", DEFAULT_HTTP_WORKQUEUE), 1L);
399  LogPrintf("HTTP: creating work queue of depth %d\n", workQueueDepth);
400 
401  workQueue = new WorkQueue<HTTPClosure>(workQueueDepth);
402  // transfer ownership to eventBase/HTTP via .release()
403  eventBase = base_ctr.release();
404  eventHTTP = http_ctr.release();
405  return true;
406 }

448 void StopHTTPServer()
449 {
450  LogPrint(BCLog::HTTP, "Stopping HTTP server\n");
451  if (workQueue) {
452  LogPrint(BCLog::HTTP, "Waiting for HTTP worker threads to exit\n");
453  for (auto& thread: g_thread_http_workers) {
454  thread.join();
455  }
456  g_thread_http_workers.clear();
457  delete workQueue;
458  workQueue = nullptr;
459  }
460  // Unlisten sockets, these are what make the event loop running, which means
461  // that after this and all connections are closed the event loop will quit.
462  for (evhttp_bound_socket *socket : boundSockets) {
463  evhttp_del_accept_socket(eventHTTP, socket);
464  }
465  boundSockets.clear();
466  if (eventBase) {
467  LogPrint(BCLog::HTTP, "Waiting for HTTP event thread to exit\n");
468  threadHTTP.join();
469  }
470  if (eventHTTP) {
471  evhttp_free(eventHTTP);
472  eventHTTP = nullptr;
473  }
474  if (eventBase) {
475  event_base_free(eventBase);
476  eventBase = nullptr;
477  }
478  LogPrint(BCLog::HTTP, "Stopped HTTP server\n");
479 }
