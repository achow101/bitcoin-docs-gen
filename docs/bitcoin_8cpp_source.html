84 static QString GetLangTerritory()
85 {
86  QSettings settings;
87  // Get desired locale (e.g. "de_DE")
88  // 1) System default language
89  QString lang_territory = QLocale::system().name();

96  return lang_territory;

100 static void initTranslations(QTranslator &qtTranslatorBase, QTranslator &qtTranslator, QTranslator &translatorBase, QTranslator &translator)
101 {
102  // Remove old translators
103  QApplication::removeTranslator(&qtTranslatorBase);
104  QApplication::removeTranslator(&qtTranslator);
105  QApplication::removeTranslator(&translatorBase);
106  QApplication::removeTranslator(&translator);
107 
108  // Get desired locale (e.g. "de_DE")
109  // 1) System default language
110  QString lang_territory = GetLangTerritory();
111 
112  // Convert to "de" only by truncating "_DE"
113  QString lang = lang_territory;
114  lang.truncate(lang_territory.lastIndexOf('_'));
115 
116  // Load language files for configured locale:
117  // - First load the translator for the base language, without territory
118  // - Then load the more specific locale translator
119 
120  // Load e.g. qt_de.qm
121  if (qtTranslatorBase.load("qt_" + lang, QLibraryInfo::location(QLibraryInfo::TranslationsPath)))
122  QApplication::installTranslator(&qtTranslatorBase);
123 
124  // Load e.g. qt_de_DE.qm
125  if (qtTranslator.load("qt_" + lang_territory, QLibraryInfo::location(QLibraryInfo::TranslationsPath)))
126  QApplication::installTranslator(&qtTranslator);
127 
128  // Load e.g. bitcoin_de.qm (shortcut "de" needs to be defined in bitcoin.qrc)
129  if (translatorBase.load(lang, ":/translations/"))
130  QApplication::installTranslator(&translatorBase);
131 
132  // Load e.g. bitcoin_de_DE.qm (shortcut "de_DE" needs to be defined in bitcoin.qrc)
133  if (translator.load(lang_territory, ":/translations/"))
134  QApplication::installTranslator(&translator);
135 }

366 void BitcoinApplication::startThread()
367 {
368  if(coreThread)
369  return;
370  coreThread = new QThread(this);
371  BitcoinCore *executor = new BitcoinCore(m_node);
372  executor->moveToThread(coreThread);
373 
374  /* communication to and from thread */
375  connect(executor, &BitcoinCore::initializeResult, this, &BitcoinApplication::initializeResult);
376  connect(executor, &BitcoinCore::shutdownResult, this, &BitcoinApplication::shutdownResult);
377  connect(executor, &BitcoinCore::runawayException, this, &BitcoinApplication::handleRunawayException);
378  connect(this, &BitcoinApplication::requestedInitialize, executor, &BitcoinCore::initialize);
379  connect(this, &BitcoinApplication::requestedShutdown, executor, &BitcoinCore::shutdown);
380  /* make sure executor object is deleted in its own thread */
381  connect(this, &BitcoinApplication::stopThread, executor, &QObject::deleteLater);
382  connect(this, &BitcoinApplication::stopThread, coreThread, &QThread::quit);
383 
384  coreThread->start();
385 }

460 void BitcoinApplication::initializeResult(bool success)
461 {
462  qDebug() << __func__ << ": Initialization result: " << success;
463  // Set exit result.
464  returnValue = success ? EXIT_SUCCESS : EXIT_FAILURE;
465  if(success)
466  {
467  // Log this only after AppInitMain finishes, as then logging setup is guaranteed complete
468  qWarning() << "Platform customization:" << platformStyle->getName();
469 #ifdef ENABLE_WALLET
470  PaymentServer::LoadRootCAs();
471  paymentServer->setOptionsModel(optionsModel);
472 #endif
473 
474  clientModel = new ClientModel(m_node, optionsModel);
475  window->setClientModel(clientModel);
476 
477 #ifdef ENABLE_WALLET
478  m_handler_load_wallet = m_node.handleLoadWallet([this](std::unique_ptr<interfaces::Wallet> wallet) {
479  WalletModel* wallet_model = new WalletModel(std::move(wallet), m_node, platformStyle, optionsModel, nullptr);
480  // Fix wallet model thread affinity.
481  wallet_model->moveToThread(thread());
482  QMetaObject::invokeMethod(this, "addWallet", Qt::QueuedConnection, Q_ARG(WalletModel*, wallet_model));
483  });
484 
485  for (auto& wallet : m_node.getWallets()) {
486  addWallet(new WalletModel(std::move(wallet), m_node, platformStyle, optionsModel));
487  }
488 #endif
489 
490  // If -min option passed, start window minimized.
491  if(gArgs.GetBoolArg("-min", false))
492  {
493  window->showMinimized();
494  }
495  else
496  {
497  window->show();
498  }
499  Q_EMIT splashFinished(window);
500 
501 #ifdef ENABLE_WALLET
502  // Now that initialization/startup is done, process any command-line
503  // bitcoin: URIs or payment requests:
504  connect(paymentServer, &PaymentServer::receivedPaymentRequest, window, &BitcoinGUI::handlePaymentRequest);
505  connect(window, &BitcoinGUI::receivedURI, paymentServer, &PaymentServer::handleURIOrFile);
506  connect(paymentServer, &PaymentServer::message, [this](const QString& title, const QString& message, unsigned int style) {
507  window->message(title, message, style);
508  });
509  QTimer::singleShot(100, paymentServer, &PaymentServer::uiReady);
510 #endif
511  pollShutdownTimer->start(200);
512  } else {
513  Q_EMIT splashFinished(window); // Make sure splash screen doesn't stick around during shutdown
514  quit(); // Exit first main loop invocation
515  }
516 }

577  // so set SSL protocols to TLS1.0+.

579  sslconf.setProtocol(QSsl::TlsV1_0OrLater);

581 #endif

583  // Register meta types used for QMetaObject::invokeMethod
584  qRegisterMetaType< bool* >();
585  // Need to pass name here as CAmount is a typedef (see http://qt-project.org/doc/qt-5/qmetatype.html#qRegisterMetaType)
586  // IMPORTANT if it is no longer a typedef use the normal variant above
587  qRegisterMetaType< CAmount >("CAmount");
588  qRegisterMetaType< std::function<void(void)> >("std::function<void(void)>");
589 #ifdef ENABLE_WALLET
590  qRegisterMetaType<WalletModel*>("WalletModel*");
591 #endif
592 
594  // Command-line options take precedence:
595  node->setupServerArgs();
596  SetupUIArgs();
597  std::string error;
598  if (!node->parseParameters(argc, argv, error)) {
599  QMessageBox::critical(0, QObject::tr(PACKAGE_NAME),
600  QObject::tr("Error parsing command line arguments: %1.").arg(QString::fromStdString(error)));
601  return EXIT_FAILURE;
602  }
603 
604  // Now that the QApplication is setup and we have parsed our parameters, we can set the platform style
605  app.setupPlatformStyle();
606 
608  // must be set before OptionsModel is initialized or translations are loaded,
609  // as it is used to locate QSettings
610  QApplication::setOrganizationName(QAPP_ORG_NAME);
611  QApplication::setOrganizationDomain(QAPP_ORG_DOMAIN);
612  QApplication::setApplicationName(QAPP_APP_NAME_DEFAULT);
613 
615  // Now that QSettings are accessible, initialize translations
616  QTranslator qtTranslatorBase, qtTranslator, translatorBase, translator;
617  initTranslations(qtTranslatorBase, qtTranslator, translatorBase, translator);
618 
619  // Show help message immediately after parsing command-line options (for "-lang") and setting locale,
620  // but before showing splash screen.
621  if (HelpRequested(gArgs) || gArgs.IsArgSet("-version")) {
622  HelpMessageDialog help(*node, nullptr, gArgs.IsArgSet("-version"));
623  help.showOrPrint();
624  return EXIT_SUCCESS;
625  }
626 
628  // User language is set up: pick a data directory
629  if (!Intro::pickDataDirectory(*node))
630  return EXIT_SUCCESS;
631 
634  if (!fs::is_directory(GetDataDir(false)))
635  {
636  QMessageBox::critical(0, QObject::tr(PACKAGE_NAME),
637  QObject::tr("Error: Specified data directory \"%1\" does not exist.").arg(QString::fromStdString(gArgs.GetArg("-datadir", ""))));
638  return EXIT_FAILURE;
639  }
640  if (!node->readConfigFiles(error)) {
641  QMessageBox::critical(0, QObject::tr(PACKAGE_NAME),
642  QObject::tr("Error: Cannot parse configuration file: %1.").arg(QString::fromStdString(error)));
643  return EXIT_FAILURE;
644  }
645 
647  // - Do not call Params() before this step
648  // - Do this after parsing the configuration file, as the network can be switched there
649  // - QSettings() will use the new application name after this, resulting in network-specific settings
650  // - Needs to be done before createOptionsModel
651 
652  // Check for -testnet or -regtest parameter (Params() calls are only valid after this clause)
653  try {
654  node->selectParams(gArgs.GetChainName());
655  } catch(std::exception &e) {
656  QMessageBox::critical(0, QObject::tr(PACKAGE_NAME), QObject::tr("Error: %1").arg(e.what()));
657  return EXIT_FAILURE;
658  }

661  PaymentServer::ipcParseCommandLine(*node, argc, argv);
662 #endif
663 
664  QScopedPointer<const NetworkStyle> networkStyle(NetworkStyle::instantiate(QString::fromStdString(Params().NetworkIDString())));
665  assert(!networkStyle.isNull());
666  // Allow for separate UI settings for testnets
667  QApplication::setApplicationName(networkStyle->getAppName());
668  // Re-initialize translations after changing application name (language in network-specific settings can be different)
669  initTranslations(qtTranslatorBase, qtTranslator, translatorBase, translator);
670 
671 #ifdef ENABLE_WALLET
672  // - Do this early as we don't want to bother initializing if we are just calling IPC
