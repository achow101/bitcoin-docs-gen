455 // Explicit calculation which is used to test the wallet constant
456 // We get the same virtual size due to rounding(weight/4) for both use_max_sig values
457 static size_t CalculateNestedKeyhashInputSize(bool use_max_sig)
458 {
459  // Generate ephemeral valid pubkey
460  CKey key;
461  key.MakeNewKey(true);
462  CPubKey pubkey = key.GetPubKey();
463 
464  // Generate pubkey hash
465  uint160 key_hash(Hash160(pubkey.begin(), pubkey.end()));
466 
467  // Create inner-script to enter into keystore. Key hash can't be 0...
468  CScript inner_script = CScript() << OP_0 << std::vector<unsigned char>(key_hash.begin(), key_hash.end());
469 
470  // Create outer P2SH script for the output
471  uint160 script_id(Hash160(inner_script.begin(), inner_script.end()));
472  CScript script_pubkey = CScript() << OP_HASH160 << std::vector<unsigned char>(script_id.begin(), script_id.end()) << OP_EQUAL;
473 
474  // Add inner-script to key store and key to watchonly
475  CBasicKeyStore keystore;
476  keystore.AddCScript(inner_script);
477  keystore.AddKeyPubKey(key, pubkey);
478 
479  // Fill in dummy signatures for fee calculation.
480  SignatureData sig_data;
481 
482  if (!ProduceSignature(keystore, use_max_sig ? DUMMY_MAXIMUM_SIGNATURE_CREATOR : DUMMY_SIGNATURE_CREATOR, script_pubkey, sig_data)) {
483  // We're hand-feeding it correct arguments; shouldn't happen
484  assert(false);
485  }
486 
487  CTxIn tx_in;
488  UpdateInput(tx_in, sig_data);
489  return (size_t)GetVirtualTransactionInputSize(tx_in);
490 }
