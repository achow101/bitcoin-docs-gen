18 template <typename OStream>
19 static void GolombRiceEncode(BitStreamWriter<OStream>& bitwriter, uint8_t P, uint64_t x)
20 {
21  // Write quotient as unary-encoded: q 1's followed by one 0.
22  uint64_t q = x >> P;
23  while (q > 0) {
24  int nbits = q <= 64 ? static_cast<int>(q) : 64;
25  bitwriter.Write(~0ULL, nbits);
26  q -= nbits;
27  }
28  bitwriter.Write(0, 1);
29 
30  // Write the remainder in P bits. Since the remainder is just the bottom
31  // P bits of x, there is no need to mask first.
32  bitwriter.Write(x, P);
33 }

35 template <typename IStream>
36 static uint64_t GolombRiceDecode(BitStreamReader<IStream>& bitreader, uint8_t P)
37 {
38  // Read unary-encoded quotient: q 1's followed by one 0.
39  uint64_t q = 0;
40  while (bitreader.Read(1) == 1) {
41  ++q;
42  }
43 
44  uint64_t r = bitreader.Read(P);
45 
46  return (q << P) + r;
47 }

49 // Map a value x that is uniformly distributed in the range [0, 2^64) to a
50 // value uniformly distributed in [0, n) by returning the upper 64 bits of
51 // x * n.
52 //
53 // See: https://lemire.me/blog/2016/06/27/a-fast-alternative-to-the-modulo-reduction/
54 static uint64_t MapIntoRange(uint64_t x, uint64_t n)
55 {
56 #ifdef __SIZEOF_INT128__
57  return (static_cast<unsigned __int128>(x) * static_cast<unsigned __int128>(n)) >> 64;
58 #else
59  // To perform the calculation on 64-bit numbers without losing the
60  // result to overflow, split the numbers into the most significant and
61  // least significant 32 bits and perform multiplication piece-wise.
62  //
63  // See: https://stackoverflow.com/a/26855440
64  uint64_t x_hi = x >> 32;
65  uint64_t x_lo = x & 0xFFFFFFFF;
66  uint64_t n_hi = n >> 32;
67  uint64_t n_lo = n & 0xFFFFFFFF;
68 
69  uint64_t ac = x_hi * n_hi;
70  uint64_t ad = x_hi * n_lo;
71  uint64_t bc = x_lo * n_hi;
72  uint64_t bd = x_lo * n_lo;
73 
74  uint64_t mid34 = (bd >> 32) + (bc & 0xFFFFFFFF) + (ad & 0xFFFFFFFF);
75  uint64_t upper64 = ac + (bc >> 32) + (ad >> 32) + (mid34 >> 32);
76  return upper64;
77 #endif
78 }

103 GCSFilter::GCSFilter(const Params& params, std::vector<unsigned char> encoded_filter)
104  : m_params(params), m_encoded(std::move(encoded_filter))
105 {
106  VectorReader stream(GCS_SER_TYPE, GCS_SER_VERSION, m_encoded, 0);
107 
108  uint64_t N = ReadCompactSize(stream);
109  m_N = static_cast<uint32_t>(N);
110  if (m_N != N) {
111  throw std::ios_base::failure("N must be <2^32");
112  }
113  m_F = static_cast<uint64_t>(m_N) * static_cast<uint64_t>(m_params.m_M);
114 
115  // Verify that the encoded filter contains exactly N elements. If it has too much or too little
116  // data, a std::ios_base::failure exception will be raised.
117  BitStreamReader<VectorReader> bitreader(stream);
118  for (uint64_t i = 0; i < m_N; ++i) {
119  GolombRiceDecode(bitreader, m_params.m_P);
120  }
121  if (!stream.empty()) {
122  throw std::ios_base::failure("encoded_filter contains excess data");
123  }

126 GCSFilter::GCSFilter(const Params& params, const ElementSet& elements)
127  : m_params(params)
128 {
129  size_t N = elements.size();
130  m_N = static_cast<uint32_t>(N);
131  if (m_N != N) {
132  throw std::invalid_argument("N must be <2^32");
133  }
134  m_F = static_cast<uint64_t>(m_N) * static_cast<uint64_t>(m_params.m_M);
135 
136  CVectorWriter stream(GCS_SER_TYPE, GCS_SER_VERSION, m_encoded, 0);
137 
138  WriteCompactSize(stream, m_N);
139 
140  if (elements.empty()) {
141  return;
142  }
143 
144  BitStreamWriter<CVectorWriter> bitwriter(stream);
145 
146  uint64_t last_value = 0;
147  for (uint64_t value : BuildHashedSet(elements)) {
148  uint64_t delta = value - last_value;
149  GolombRiceEncode(bitwriter, m_params.m_P, delta);
150  last_value = value;
151  }
152

154 }

156 bool GCSFilter::MatchInternal(const uint64_t* element_hashes, size_t size) const
157 {
158  VectorReader stream(GCS_SER_TYPE, GCS_SER_VERSION, m_encoded, 0);
159 
160  // Seek forward by size of N
161  uint64_t N = ReadCompactSize(stream);
162  assert(N == m_N);
163 
164  BitStreamReader<VectorReader> bitreader(stream);
165 
166  uint64_t value = 0;
167  size_t hashes_index = 0;
168  for (uint32_t i = 0; i < m_N; ++i) {
169  uint64_t delta = GolombRiceDecode(bitreader, m_params.m_P);
170  value += delta;
171 
172  while (true) {
173  if (hashes_index == size) {
174  return false;
175  } else if (element_hashes[hashes_index] == value) {
176  return true;
177  } else if (element_hashes[hashes_index] > value) {
178  break;
179  }
180 
181  hashes_index++;
182  }
183  }
184 
185  return false;
186 }
