48 static inline size_t MallocUsage(size_t alloc)
49 {
50  // Measured on libc6 2.19 on Linux.
51  if (alloc == 0) {
52  return 0;
53  } else if (sizeof(void*) == 8) {
54  return ((alloc + 31) >> 4) << 4;
55  } else if (sizeof(void*) == 4) {
56  return ((alloc + 15) >> 3) << 3;
57  } else {
58  assert(0);
59  }
60 }
