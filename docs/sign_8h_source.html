154 // Takes a stream and multiple arguments and unserializes them first as a vector then each object individually in the order provided in the arguments
155 template<typename Stream, typename... X>
156 void UnserializeFromVector(Stream& s, X&... args)
157 {
158  size_t expected_size = ReadCompactSize(s);
159  size_t remaining_before = s.size();
160  UnserializeMany(s, args...);
161  size_t remaining_after = s.size();
162  if (remaining_after + expected_size != remaining_before) {
163  throw std::ios_base::failure("Size of value was not the stated size");
164  }
165 }

167 // Deserialize HD keypaths into a map
168 template<typename Stream>
169 void DeserializeHDKeypaths(Stream& s, const std::vector<unsigned char>& key, std::map<CPubKey, KeyOriginInfo>& hd_keypaths)
170 {
171  // Make sure that the key is the size of pubkey + 1
172  if (key.size() != CPubKey::PUBLIC_KEY_SIZE + 1 && key.size() != CPubKey::COMPRESSED_PUBLIC_KEY_SIZE + 1) {
173  throw std::ios_base::failure("Size of key was not the expected size for the type BIP32 keypath");
174  }
175  // Read in the pubkey from key
176  CPubKey pubkey(key.begin() + 1, key.end());
177  if (!pubkey.IsFullyValid()) {
178  throw std::ios_base::failure("Invalid pubkey");
179  }
180  if (hd_keypaths.count(pubkey) > 0) {
181  throw std::ios_base::failure("Duplicate Key, pubkey derivation path already provided");
182  }
183 
184  // Read in key path
185  uint64_t value_len = ReadCompactSize(s);
186  if (value_len % 4 || value_len == 0) {
187  throw std::ios_base::failure("Invalid length for HD key path");
188  }
189 
190  KeyOriginInfo keypath;
191  s >> keypath.fingerprint;
192  for (unsigned int i = 4; i < value_len; i += sizeof(uint32_t)) {
193  uint32_t index;
194  s >> index;
195  keypath.path.push_back(index);
196  }
197 
198  // Add to map
199  hd_keypaths.emplace(pubkey, std::move(keypath));
200 }

202 // Serialize HD keypaths to a stream from a map
203 template<typename Stream>
204 void SerializeHDKeypaths(Stream& s, const std::map<CPubKey, KeyOriginInfo>& hd_keypaths, uint8_t type)
205 {
206  for (auto keypath_pair : hd_keypaths) {
207  SerializeToVector(s, type, MakeSpan(keypath_pair.first));
208  WriteCompactSize(s, (keypath_pair.second.path.size() + 1) * sizeof(uint32_t));
209  s << keypath_pair.second.fingerprint;
210  for (const auto& path : keypath_pair.second.path) {
211  s << path;
212  }
213  }

237  template <typename Stream>
238  inline void Serialize(Stream& s) const {
239  // Write the utxo
240  // If there is a non-witness utxo, then don't add the witness one.
241  if (non_witness_utxo) {
242  SerializeToVector(s, PSBT_IN_NON_WITNESS_UTXO);
243  OverrideStream<Stream> os(&s, s.GetType(), s.GetVersion() | SERIALIZE_TRANSACTION_NO_WITNESS);
244  SerializeToVector(os, non_witness_utxo);
245  } else if (!witness_utxo.IsNull()) {
246  SerializeToVector(s, PSBT_IN_WITNESS_UTXO);
247  SerializeToVector(s, witness_utxo);
248  }
249 
250  if (final_script_sig.empty() && final_script_witness.IsNull()) {
251  // Write any partial signatures
252  for (auto sig_pair : partial_sigs) {
253  SerializeToVector(s, PSBT_IN_PARTIAL_SIG, MakeSpan(sig_pair.second.first));
254  s << sig_pair.second.second;
255  }
256 
257  // Write the sighash type
258  if (sighash_type > 0) {
259  SerializeToVector(s, PSBT_IN_SIGHASH);
260  SerializeToVector(s, sighash_type);
261  }
262 
263  // Write the redeem script
264  if (!redeem_script.empty()) {
265  SerializeToVector(s, PSBT_IN_REDEEMSCRIPT);
266  s << redeem_script;
267  }
268 
269  // Write the witness script
270  if (!witness_script.empty()) {
271  SerializeToVector(s, PSBT_IN_WITNESSSCRIPT);
272  s << witness_script;
273  }
274 
275  // Write any hd keypaths
276  SerializeHDKeypaths(s, hd_keypaths, PSBT_IN_BIP32_DERIVATION);
277  }
278 
279  // Write script sig
280  if (!final_script_sig.empty()) {
281  SerializeToVector(s, PSBT_IN_SCRIPTSIG);
282  s << final_script_sig;
283  }
284  // write script witness
285  if (!final_script_witness.IsNull()) {
286  SerializeToVector(s, PSBT_IN_SCRIPTWITNESS);
287  SerializeToVector(s, final_script_witness.stack);
288  }
289 
290  // Write unknown things
291  for (auto& entry : unknown) {
292  s << entry.first;
293  s << entry.second;
294  }
295 
296  s << PSBT_SEPARATOR;
297  }

300  template <typename Stream>
301  inline void Unserialize(Stream& s) {
302  // Read loop
303  while(!s.empty()) {
304  // Read
305  std::vector<unsigned char> key;
306  s >> key;
307 
308  // the key is empty if that was actually a separator byte
309  // This is a special case for key lengths 0 as those are not allowed (except for separator)
310  if (key.empty()) return;
311 
312  // First byte of key is the type
313  unsigned char type = key[0];
314 
315  // Do stuff based on type
316  switch(type) {
317  case PSBT_IN_NON_WITNESS_UTXO:
318  {
319  if (non_witness_utxo) {
320  throw std::ios_base::failure("Duplicate Key, input non-witness utxo already provided");
321  } else if (key.size() != 1) {
322  throw std::ios_base::failure("Non-witness utxo key is more than one byte type");
323  }
324  // Set the stream to unserialize with witness since this is always a valid network transaction
325  OverrideStream<Stream> os(&s, s.GetType(), s.GetVersion() & ~SERIALIZE_TRANSACTION_NO_WITNESS);
326  UnserializeFromVector(os, non_witness_utxo);
327  break;
328  }
329  case PSBT_IN_WITNESS_UTXO:
330  if (!witness_utxo.IsNull()) {
331  throw std::ios_base::failure("Duplicate Key, input witness utxo already provided");
332  } else if (key.size() != 1) {
333  throw std::ios_base::failure("Witness utxo key is more than one byte type");
334  }
335  UnserializeFromVector(s, witness_utxo);
336  break;
337  case PSBT_IN_PARTIAL_SIG:
338  {
339  // Make sure that the key is the size of pubkey + 1
340  if (key.size() != CPubKey::PUBLIC_KEY_SIZE + 1 && key.size() != CPubKey::COMPRESSED_PUBLIC_KEY_SIZE + 1) {
341  throw std::ios_base::failure("Size of key was not the expected size for the type partial signature pubkey");
342  }
343  // Read in the pubkey from key
344  CPubKey pubkey(key.begin() + 1, key.end());
345  if (!pubkey.IsFullyValid()) {
346  throw std::ios_base::failure("Invalid pubkey");
347  }
348  if (partial_sigs.count(pubkey.GetID()) > 0) {
349  throw std::ios_base::failure("Duplicate Key, input partial signature for pubkey already provided");
350  }
351 
352  // Read in the signature from value
353  std::vector<unsigned char> sig;
354  s >> sig;
355 
356  // Add to list
357  partial_sigs.emplace(pubkey.GetID(), SigPair(pubkey, std::move(sig)));
358  break;
359  }
360  case PSBT_IN_SIGHASH:
361  if (sighash_type > 0) {
362  throw std::ios_base::failure("Duplicate Key, input sighash type already provided");
363  } else if (key.size() != 1) {
364  throw std::ios_base::failure("Sighash type key is more than one byte type");
365  }
366  UnserializeFromVector(s, sighash_type);
367  break;
368  case PSBT_IN_REDEEMSCRIPT:
369  {
370  if (!redeem_script.empty()) {
371  throw std::ios_base::failure("Duplicate Key, input redeemScript already provided");
372  } else if (key.size() != 1) {
373  throw std::ios_base::failure("Input redeemScript key is more than one byte type");
374  }
375  s >> redeem_script;
376  break;
377  }
378  case PSBT_IN_WITNESSSCRIPT:
379  {
380  if (!witness_script.empty()) {
381  throw std::ios_base::failure("Duplicate Key, input witnessScript already provided");
382  } else if (key.size() != 1) {
383  throw std::ios_base::failure("Input witnessScript key is more than one byte type");
384  }
385  s >> witness_script;
386  break;
387  }
388  case PSBT_IN_BIP32_DERIVATION:
389  {
390  DeserializeHDKeypaths(s, key, hd_keypaths);
391  break;
392  }
393  case PSBT_IN_SCRIPTSIG:
394  {
395  if (!final_script_sig.empty()) {
396  throw std::ios_base::failure("Duplicate Key, input final scriptSig already provided");
397  } else if (key.size() != 1) {
398  throw std::ios_base::failure("Final scriptSig key is more than one byte type");
399  }
400  s >> final_script_sig;
401  break;
402  }
403  case PSBT_IN_SCRIPTWITNESS:
404  {
405  if (!final_script_witness.IsNull()) {
406  throw std::ios_base::failure("Duplicate Key, input final scriptWitness already provided");
407  } else if (key.size() != 1) {
408  throw std::ios_base::failure("Final scriptWitness key is more than one byte type");
409  }
410  UnserializeFromVector(s, final_script_witness.stack);
411  break;
412  }
413  // Unknown stuff
414  default:
415  if (unknown.count(key) > 0) {
416  throw std::ios_base::failure("Duplicate Key, key for unknown value already provided");
417  }
418  // Read in the value
419  std::vector<unsigned char> val_bytes;
420  s >> val_bytes;
421  unknown.emplace(std::move(key), std::move(val_bytes));
422  break;
423  }
424  }
425  }

595  template <typename Stream>
596  inline void Unserialize(Stream& s) {
597  // Read the magic bytes
598  uint8_t magic[5];
599  s >> magic;
600  if (!std::equal(magic, magic + 5, PSBT_MAGIC_BYTES)) {
601  throw std::ios_base::failure("Invalid PSBT magic bytes");
602  }
603 
604  // Read global data
605  while(!s.empty()) {
606  // Read
607  std::vector<unsigned char> key;
608  s >> key;
609 
610  // the key is empty if that was actually a separator byte
611  // This is a special case for key lengths 0 as those are not allowed (except for separator)
612  if (key.empty()) break;
613 
614  // First byte of key is the type
615  unsigned char type = key[0];
616 
617  // Do stuff based on type
618  switch(type) {
619  case PSBT_GLOBAL_UNSIGNED_TX:
620  {
621  if (tx) {
622  throw std::ios_base::failure("Duplicate Key, unsigned tx already provided");
623  } else if (key.size() != 1) {
624  throw std::ios_base::failure("Global unsigned tx key is more than one byte type");
625  }
626  CMutableTransaction mtx;

628  OverrideStream<Stream> os(&s, s.GetType(), s.GetVersion() | SERIALIZE_TRANSACTION_NO_WITNESS);

632  for (const CTxIn& txin : tx->vin) {
633  if (!txin.scriptSig.empty() || !txin.scriptWitness.IsNull()) {
634  throw std::ios_base::failure("Unsigned tx does not have empty scriptSigs and scriptWitnesses.");
635  }
636  }
637  break;
638  }
639  // Unknown stuff
640  default: {
641  if (unknown.count(key) > 0) {
642  throw std::ios_base::failure("Duplicate Key, key for unknown value already provided");
643  }
644  // Read in the value
645  std::vector<unsigned char> val_bytes;
646  s >> val_bytes;
647  unknown.emplace(std::move(key), std::move(val_bytes));
648  }
649  }
650  }
651 
652  // Make sure that we got an unsigned tx
653  if (!tx) {
654  throw std::ios_base::failure("No unsigned transcation was provided");
655  }
656 
657  // Read input data
658  unsigned int i = 0;
659  while (!s.empty() && i < tx->vin.size()) {
660  PSBTInput input;
661  s >> input;
662  inputs.push_back(input);
663 
664  // Make sure the non-witness utxo matches the outpoint
665  if (input.non_witness_utxo && input.non_witness_utxo->GetHash() != tx->vin[i].prevout.hash) {
666  throw std::ios_base::failure("Non-witness UTXO does not match outpoint hash");
667  }
668  ++i;
669  }
670  // Make sure that the number of inputs matches the number of inputs in the transaction
671  if (inputs.size() != tx->vin.size()) {
672  throw std::ios_base::failure("Inputs provided does not match the number of inputs in transaction.");
673  }
674 
675  // Read output data
676  i = 0;
677  while (!s.empty() && i < tx->vout.size()) {
678  PSBTOutput output;
679  s >> output;
680  outputs.push_back(output);
681  ++i;
682  }
683  // Make sure that the number of outputs matches the number of outputs in the transaction
684  if (outputs.size() != tx->vout.size()) {
685  throw std::ios_base::failure("Outputs provided does not match the number of outputs in transaction.");
686  }
687  // Sanity check
688  if (!IsSane()) {
689  throw std::ios_base::failure("PSBT is not sane.");
690  }
691  }
