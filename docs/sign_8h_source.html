135 // Takes a stream and multiple arguments and unserializes them first as a vector then each object individually in the order provided in the arguments
136 template<typename Stream, typename... X>
137 void UnserializeFromVector(Stream& s, X&... args)
138 {
139  size_t expected_size = ReadCompactSize(s);

148 // Deserialize HD keypaths into a map
149 template<typename Stream>
150 void DeserializeHDKeypaths(Stream& s, const std::vector<unsigned char>& key, std::map<CPubKey, KeyOriginInfo>& hd_keypaths)
151 {
152  // Make sure that the key is the size of pubkey + 1
153  if (key.size() != CPubKey::PUBLIC_KEY_SIZE + 1 && key.size() != CPubKey::COMPRESSED_PUBLIC_KEY_SIZE + 1) {
154  throw std::ios_base::failure("Size of key was not the expected size for the type BIP32 keypath");
155  }
156  // Read in the pubkey from key
157  CPubKey pubkey(key.begin() + 1, key.end());
158  if (!pubkey.IsFullyValid()) {
159  throw std::ios_base::failure("Invalid pubkey");
160  }
161  if (hd_keypaths.count(pubkey) > 0) {
162  throw std::ios_base::failure("Duplicate Key, pubkey derivation path already provided");
163  }
164 
165  // Read in key path
166  uint64_t value_len = ReadCompactSize(s);
167  if (value_len % 4 || value_len == 0) {
168  throw std::ios_base::failure("Invalid length for HD key path");
169  }
170 
171  KeyOriginInfo keypath;
172  s >> keypath.fingerprint;
173  for (unsigned int i = 4; i < value_len; i += sizeof(uint32_t)) {
174  uint32_t index;
175  s >> index;
176  keypath.path.push_back(index);
177  }
178 
179  // Add to map
180  hd_keypaths.emplace(pubkey, std::move(keypath));
181 }

183 // Serialize HD keypaths to a stream from a map
184 template<typename Stream>
185 void SerializeHDKeypaths(Stream& s, const std::map<CPubKey, KeyOriginInfo>& hd_keypaths, uint8_t type)
186 {
187  for (auto keypath_pair : hd_keypaths) {
188  if (!keypath_pair.first.IsValid()) {
189  throw std::ios_base::failure("Invalid CPubKey being serialized");
190  }
191  SerializeToVector(s, type, MakeSpan(keypath_pair.first));
192  WriteCompactSize(s, (keypath_pair.second.path.size() + 1) * sizeof(uint32_t));
193  s << keypath_pair.second.fingerprint;
194  for (const auto& path : keypath_pair.second.path) {
195  s << path;
196  }
197  }
198 }
