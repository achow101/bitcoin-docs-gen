355struct WalletBalances
356{
357 CAmount balance = 0;
358 CAmount unconfirmed_balance = 0;
359 CAmount immature_balance = 0;
360 bool have_watch_only = false;
361 CAmount watch_only_balance = 0;
362 CAmount unconfirmed_watch_only_balance = 0;
363 CAmount immature_watch_only_balance = 0;
364
365 bool balanceChanged(const WalletBalances& prev) const
366 {
367 return balance != prev.balance || unconfirmed_balance != prev.unconfirmed_balance ||
368 immature_balance != prev.immature_balance || watch_only_balance != prev.watch_only_balance ||
369 unconfirmed_watch_only_balance != prev.unconfirmed_watch_only_balance ||
370 immature_watch_only_balance != prev.immature_watch_only_balance;
371 }
372};
