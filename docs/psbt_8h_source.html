63  template <typename Stream>
64  inline void Serialize(Stream& s) const {
65  // Write the utxo
66  // If there is a non-witness utxo, then don't add the witness one.
67  if (non_witness_utxo) {
68  SerializeToVector(s, PSBT_IN_NON_WITNESS_UTXO);
69  OverrideStream<Stream> os(&s, s.GetType(), s.GetVersion() | SERIALIZE_TRANSACTION_NO_WITNESS);
70  SerializeToVector(os, non_witness_utxo);
71  } else if (!witness_utxo.IsNull()) {
72  SerializeToVector(s, PSBT_IN_WITNESS_UTXO);
73  SerializeToVector(s, witness_utxo);
74  }
75 
76  if (final_script_sig.empty() && final_script_witness.IsNull()) {
77  // Write any partial signatures
78  for (auto sig_pair : partial_sigs) {
79  SerializeToVector(s, PSBT_IN_PARTIAL_SIG, MakeSpan(sig_pair.second.first));
80  s << sig_pair.second.second;
81  }
82 
83  // Write the sighash type
84  if (sighash_type > 0) {
85  SerializeToVector(s, PSBT_IN_SIGHASH);
86  SerializeToVector(s, sighash_type);
87  }
88 
89  // Write the redeem script
90  if (!redeem_script.empty()) {
91  SerializeToVector(s, PSBT_IN_REDEEMSCRIPT);
92  s << redeem_script;
93  }
94 
95  // Write the witness script
96  if (!witness_script.empty()) {
97  SerializeToVector(s, PSBT_IN_WITNESSSCRIPT);
98  s << witness_script;
99  }
100 
101  // Write any hd keypaths
102  SerializeHDKeypaths(s, hd_keypaths, PSBT_IN_BIP32_DERIVATION);
103  }
104 
105  // Write script sig
106  if (!final_script_sig.empty()) {
107  SerializeToVector(s, PSBT_IN_SCRIPTSIG);
108  s << final_script_sig;
109  }
110  // write script witness
111  if (!final_script_witness.IsNull()) {
112  SerializeToVector(s, PSBT_IN_SCRIPTWITNESS);
113  SerializeToVector(s, final_script_witness.stack);
114  }
115 
116  // Write unknown things
117  for (auto& entry : unknown) {
118  s << entry.first;
119  s << entry.second;
120  }
121 
122  s << PSBT_SEPARATOR;
123  }

126  template <typename Stream>
127  inline void Unserialize(Stream& s) {
128  // Read loop
129  bool found_sep = false;
130  while(!s.empty()) {
131  // Read
132  std::vector<unsigned char> key;
133  s >> key;
134 
135  // the key is empty if that was actually a separator byte
136  // This is a special case for key lengths 0 as those are not allowed (except for separator)
137  if (key.empty()) {
138  found_sep = true;
139  break;
140  }
141 
142  // First byte of key is the type
143  unsigned char type = key[0];
144 
145  // Do stuff based on type
146  switch(type) {
147  case PSBT_IN_NON_WITNESS_UTXO:
148  {
149  if (non_witness_utxo) {
150  throw std::ios_base::failure("Duplicate Key, input non-witness utxo already provided");
151  } else if (key.size() != 1) {
152  throw std::ios_base::failure("Non-witness utxo key is more than one byte type");
153  }
154  // Set the stream to unserialize with witness since this is always a valid network transaction
155  OverrideStream<Stream> os(&s, s.GetType(), s.GetVersion() & ~SERIALIZE_TRANSACTION_NO_WITNESS);
156  UnserializeFromVector(os, non_witness_utxo);
157  break;
158  }
159  case PSBT_IN_WITNESS_UTXO:
160  if (!witness_utxo.IsNull()) {
161  throw std::ios_base::failure("Duplicate Key, input witness utxo already provided");
162  } else if (key.size() != 1) {
163  throw std::ios_base::failure("Witness utxo key is more than one byte type");
164  }
165  UnserializeFromVector(s, witness_utxo);
166  break;
167  case PSBT_IN_PARTIAL_SIG:
168  {
169  // Make sure that the key is the size of pubkey + 1
170  if (key.size() != CPubKey::PUBLIC_KEY_SIZE + 1 && key.size() != CPubKey::COMPRESSED_PUBLIC_KEY_SIZE + 1) {
171  throw std::ios_base::failure("Size of key was not the expected size for the type partial signature pubkey");
172  }
173  // Read in the pubkey from key
174  CPubKey pubkey(key.begin() + 1, key.end());
175  if (!pubkey.IsFullyValid()) {
176  throw std::ios_base::failure("Invalid pubkey");
177  }
178  if (partial_sigs.count(pubkey.GetID()) > 0) {
179  throw std::ios_base::failure("Duplicate Key, input partial signature for pubkey already provided");
180  }
181 
182  // Read in the signature from value
183  std::vector<unsigned char> sig;
184  s >> sig;
185 
186  // Add to list
187  partial_sigs.emplace(pubkey.GetID(), SigPair(pubkey, std::move(sig)));
188  break;
189  }
190  case PSBT_IN_SIGHASH:
191  if (sighash_type > 0) {
192  throw std::ios_base::failure("Duplicate Key, input sighash type already provided");
193  } else if (key.size() != 1) {
194  throw std::ios_base::failure("Sighash type key is more than one byte type");
195  }
196  UnserializeFromVector(s, sighash_type);
197  break;
198  case PSBT_IN_REDEEMSCRIPT:
199  {
200  if (!redeem_script.empty()) {
201  throw std::ios_base::failure("Duplicate Key, input redeemScript already provided");
202  } else if (key.size() != 1) {
203  throw std::ios_base::failure("Input redeemScript key is more than one byte type");
204  }
205  s >> redeem_script;
206  break;
207  }
208  case PSBT_IN_WITNESSSCRIPT:
209  {
210  if (!witness_script.empty()) {
211  throw std::ios_base::failure("Duplicate Key, input witnessScript already provided");
212  } else if (key.size() != 1) {
213  throw std::ios_base::failure("Input witnessScript key is more than one byte type");
214  }
215  s >> witness_script;
216  break;
217  }
218  case PSBT_IN_BIP32_DERIVATION:
219  {
220  DeserializeHDKeypaths(s, key, hd_keypaths);
221  break;
222  }
223  case PSBT_IN_SCRIPTSIG:
224  {
225  if (!final_script_sig.empty()) {
226  throw std::ios_base::failure("Duplicate Key, input final scriptSig already provided");
227  } else if (key.size() != 1) {
228  throw std::ios_base::failure("Final scriptSig key is more than one byte type");
229  }
230  s >> final_script_sig;
231  break;
232  }
233  case PSBT_IN_SCRIPTWITNESS:
234  {
235  if (!final_script_witness.IsNull()) {
236  throw std::ios_base::failure("Duplicate Key, input final scriptWitness already provided");
237  } else if (key.size() != 1) {
238  throw std::ios_base::failure("Final scriptWitness key is more than one byte type");
239  }
240  UnserializeFromVector(s, final_script_witness.stack);
241  break;
242  }
243  // Unknown stuff
244  default:
245  if (unknown.count(key) > 0) {
246  throw std::ios_base::failure("Duplicate Key, key for unknown value already provided");
247  }
248  // Read in the value
249  std::vector<unsigned char> val_bytes;
250  s >> val_bytes;
251  unknown.emplace(std::move(key), std::move(val_bytes));
252  break;
253  }
254  }
255 
256  if (!found_sep) {
257  throw std::ios_base::failure("Separator is missing at the end of an input map");
258  }
259  }

263  Unserialize(s);
264  }

441  template <typename Stream>
442  inline void Unserialize(Stream& s) {
443  // Read the magic bytes
444  uint8_t magic[5];
445  s >> magic;
446  if (!std::equal(magic, magic + 5, PSBT_MAGIC_BYTES)) {
447  throw std::ios_base::failure("Invalid PSBT magic bytes");
448  }
449 
450  // Read global data
451  bool found_sep = false;
452  while(!s.empty()) {
453  // Read
454  std::vector<unsigned char> key;
455  s >> key;
456 
457  // the key is empty if that was actually a separator byte
458  // This is a special case for key lengths 0 as those are not allowed (except for separator)
459  if (key.empty()) {
460  found_sep = true;
461  break;
462  }
463 
464  // First byte of key is the type
465  unsigned char type = key[0];
466 
467  // Do stuff based on type
468  switch(type) {
469  case PSBT_GLOBAL_UNSIGNED_TX:
470  {
471  if (tx) {
472  throw std::ios_base::failure("Duplicate Key, unsigned tx already provided");
473  } else if (key.size() != 1) {
474  throw std::ios_base::failure("Global unsigned tx key is more than one byte type");
475  }
476  CMutableTransaction mtx;
477  // Set the stream to serialize with non-witness since this should always be non-witness
478  OverrideStream<Stream> os(&s, s.GetType(), s.GetVersion() | SERIALIZE_TRANSACTION_NO_WITNESS);
479  UnserializeFromVector(os, mtx);
480  tx = std::move(mtx);
481  // Make sure that all scriptSigs and scriptWitnesses are empty
482  for (const CTxIn& txin : tx->vin) {
483  if (!txin.scriptSig.empty() || !txin.scriptWitness.IsNull()) {
484  throw std::ios_base::failure("Unsigned tx does not have empty scriptSigs and scriptWitnesses.");
485  }
486  }
487  break;
488  }
489  // Unknown stuff
490  default: {
491  if (unknown.count(key) > 0) {
492  throw std::ios_base::failure("Duplicate Key, key for unknown value already provided");
493  }
494  // Read in the value
495  std::vector<unsigned char> val_bytes;
496  s >> val_bytes;
497  unknown.emplace(std::move(key), std::move(val_bytes));
498  }
499  }
500  }
501 
502  if (!found_sep) {
503  throw std::ios_base::failure("Separator is missing at the end of the global map");
504  }
505 
506  // Make sure that we got an unsigned tx
507  if (!tx) {
508  throw std::ios_base::failure("No unsigned transcation was provided");
509  }
510 
511  // Read input data
512  unsigned int i = 0;
513  while (!s.empty() && i < tx->vin.size()) {
514  PSBTInput input;
515  s >> input;
516  inputs.push_back(input);
517 
518  // Make sure the non-witness utxo matches the outpoint
519  if (input.non_witness_utxo && input.non_witness_utxo->GetHash() != tx->vin[i].prevout.hash) {
520  throw std::ios_base::failure("Non-witness UTXO does not match outpoint hash");
521  }
522  ++i;
523  }
524  // Make sure that the number of inputs matches the number of inputs in the transaction
525  if (inputs.size() != tx->vin.size()) {
526  throw std::ios_base::failure("Inputs provided does not match the number of inputs in transaction.");
527  }
528 
529  // Read output data
530  i = 0;
531  while (!s.empty() && i < tx->vout.size()) {
532  PSBTOutput output;
533  s >> output;
534  outputs.push_back(output);
535  ++i;
536  }
537  // Make sure that the number of outputs matches the number of outputs in the transaction
538  if (outputs.size() != tx->vout.size()) {
539  throw std::ios_base::failure("Outputs provided does not match the number of outputs in transaction.");
540  }
541  // Sanity check
542  if (!IsSane()) {
543  throw std::ios_base::failure("PSBT is not sane.");
544  }
545  }
