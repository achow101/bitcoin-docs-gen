324 bool BerkeleyBatch::Recover(const fs::path& file_path, void *callbackDataIn, bool (*recoverKVcallback)(void* callbackData, CDataStream ssKey, CDataStream ssValue), std::string& newFilename)
325 {
326  std::string filename;
327  std::shared_ptr<BerkeleyEnvironment> env = GetWalletEnv(file_path, filename);
328 
329  // Recovery procedure:
330  // move wallet file to walletfilename.timestamp.bak
331  // Call Salvage with fAggressive=true to
332  // get as much data as possible.
333  // Rewrite salvaged data to fresh wallet file
334  // Set -rescan so any missing transactions will be
335  // found.
336  int64_t now = GetTime();
337  newFilename = strprintf("%s.%d.bak", filename, now);
338 
339  int result = env->dbenv->dbrename(nullptr, filename.c_str(), nullptr,
340  newFilename.c_str(), DB_AUTO_COMMIT);
341  if (result == 0)
342  LogPrintf("Renamed %s to %s\n", filename, newFilename);
343  else
344  {
345  LogPrintf("Failed to rename %s to %s\n", filename, newFilename);
346  return false;
347  }
348 
349  std::vector<BerkeleyEnvironment::KeyValPair> salvagedData;
350  bool fSuccess = env->Salvage(newFilename, true, salvagedData);
351  if (salvagedData.empty())
352  {
353  LogPrintf("Salvage(aggressive) found no records in %s.\n", newFilename);
354  return false;
355  }
356  LogPrintf("Salvage(aggressive) found %u records\n", salvagedData.size());
357 
358  std::unique_ptr<Db> pdbCopy = MakeUnique<Db>(env->dbenv.get(), 0);
359  int ret = pdbCopy->open(nullptr, // Txn pointer
360  filename.c_str(), // Filename
361  "main", // Logical db name
362  DB_BTREE, // Database type
363  DB_CREATE, // Flags
364  0);
365  if (ret > 0) {
366  LogPrintf("Cannot create database file %s\n", filename);
367  pdbCopy->close(0);
368  return false;
369  }
370 
371  DbTxn* ptxn = env->TxnBegin();
372  for (BerkeleyEnvironment::KeyValPair& row : salvagedData)
373  {
374  if (recoverKVcallback)
375  {
376  CDataStream ssKey(row.first, SER_DISK, CLIENT_VERSION);
377  CDataStream ssValue(row.second, SER_DISK, CLIENT_VERSION);
378  if (!(*recoverKVcallback)(callbackDataIn, ssKey, ssValue))
379  continue;
380  }
381  Dbt datKey(&row.first[0], row.first.size());
382  Dbt datValue(&row.second[0], row.second.size());
383  int ret2 = pdbCopy->put(ptxn, &datKey, &datValue, DB_NOOVERWRITE);
384  if (ret2 > 0)
385  fSuccess = false;
386  }
387  ptxn->commit(0);
388  pdbCopy->close(0);
389 
390  return fSuccess;
391 }

450 bool BerkeleyEnvironment::Salvage(const std::string& strFile, bool fAggressive, std::vector<BerkeleyEnvironment::KeyValPair>& vResult)
451 {
452  LOCK(cs_db);
453  assert(mapFileUseCount.count(strFile) == 0);
454 
455  u_int32_t flags = DB_SALVAGE;
456  if (fAggressive)
457  flags |= DB_AGGRESSIVE;
458 
459  std::stringstream strDump;
460 
461  Db db(dbenv.get(), 0);
462  int result = db.verify(strFile.c_str(), nullptr, &strDump, flags);
463  if (result == DB_VERIFY_BAD) {
464  LogPrintf("BerkeleyEnvironment::Salvage: Database salvage found errors, all data may not be recoverable.\n");
465  if (!fAggressive) {
466  LogPrintf("BerkeleyEnvironment::Salvage: Rerun with aggressive mode to ignore errors and continue.\n");
467  return false;
468  }
469  }
470  if (result != 0 && result != DB_VERIFY_BAD) {
471  LogPrintf("BerkeleyEnvironment::Salvage: Database salvage failed with result %d.\n", result);
472  return false;
473  }
474 
475  // Format of bdb dump is ascii lines:
476  // header lines...
477  // HEADER=END
478  // hexadecimal key
479  // hexadecimal value
480  // ... repeated
481  // DATA=END
482 
483  std::string strLine;
484  while (!strDump.eof() && strLine != HEADER_END)
485  getline(strDump, strLine); // Skip past header
486 
487  std::string keyHex, valueHex;
488  while (!strDump.eof() && keyHex != DATA_END) {
489  getline(strDump, keyHex);
490  if (keyHex != DATA_END) {
491  if (strDump.eof())
492  break;
493  getline(strDump, valueHex);
494  if (valueHex == DATA_END) {
495  LogPrintf("BerkeleyEnvironment::Salvage: WARNING: Number of keys in data does not match number of values.\n");
496  break;
497  }
498  vResult.push_back(make_pair(ParseHex(keyHex), ParseHex(valueHex)));
499  }
500  }
501 
502  if (keyHex != DATA_END) {
503  LogPrintf("BerkeleyEnvironment::Salvage: WARNING: Unexpected end of file while reading salvage output.\n");
504  return false;
505  }
506 
507  return (result == 0);
508 }

520 BerkeleyBatch::BerkeleyBatch(BerkeleyDatabase& database, const char* pszMode, bool fFlushOnCloseIn) : pdb(nullptr), activeTxn(nullptr)
521 {
522  fReadOnly = (!strchr(pszMode, '+') && !strchr(pszMode, 'w'));
523  fFlushOnClose = fFlushOnCloseIn;
524  env = database.env.get();
525  if (database.IsDummy()) {
526  return;
527  }
528  const std::string &strFilename = database.strFile;
529 
530  bool fCreate = strchr(pszMode, 'c') != nullptr;
531  unsigned int nFlags = DB_THREAD;
532  if (fCreate)
533  nFlags |= DB_CREATE;
534 
535  {
536  LOCK(cs_db);
537  if (!env->Open(false /* retry */))
538  throw std::runtime_error("BerkeleyBatch: Failed to open database environment.");
539 
540  pdb = database.m_db.get();
541  if (pdb == nullptr) {
542  int ret;
543  std::unique_ptr<Db> pdb_temp = MakeUnique<Db>(env->dbenv.get(), 0);
544 
545  bool fMockDb = env->IsMock();
546  if (fMockDb) {
547  DbMpoolFile* mpf = pdb_temp->get_mpf();
548  ret = mpf->set_flags(DB_MPOOL_NOFILE, 1);
549  if (ret != 0) {
550  throw std::runtime_error(strprintf("BerkeleyBatch: Failed to configure for no temp file backing for database %s", strFilename));
551  }
552  }
553 
554  ret = pdb_temp->open(nullptr, // Txn pointer
555  fMockDb ? nullptr : strFilename.c_str(), // Filename
556  fMockDb ? strFilename.c_str() : "main", // Logical db name
557  DB_BTREE, // Database type
558  nFlags, // Flags
559  0);
560 
561  if (ret != 0) {
562  throw std::runtime_error(strprintf("BerkeleyBatch: Error %d, can't open database %s", ret, strFilename));
563  }
564 
565  // Call CheckUniqueFileid on the containing BDB environment to
566  // avoid BDB data consistency bugs that happen when different data
567  // files in the same environment have the same fileid.
568  //
569  // Also call CheckUniqueFileid on all the other g_dbenvs to prevent
570  // bitcoin from opening the same data file through another
571  // environment when the file is referenced through equivalent but
572  // not obviously identical symlinked or hard linked or bind mounted
573  // paths. In the future a more relaxed check for equal inode and
574  // device ids could be done instead, which would allow opening
575  // different backup copies of a wallet at the same time. Maybe even
576  // more ideally, an exclusive lock for accessing the database could
577  // be implemented, so no equality checks are needed at all. (Newer
578  // versions of BDB have an set_lk_exclusive method for this
579  // purpose, but the older version we use does not.)
580  for (const auto& env : g_dbenvs) {
581  CheckUniqueFileid(*env.second.lock().get(), strFilename, *pdb_temp, this->env->m_fileids[strFilename]);
582  }
583 
584  pdb = pdb_temp.release();
585  database.m_db.reset(pdb);
586 
587  if (fCreate && !Exists(std::string("version"))) {
588  bool fTmp = fReadOnly;
589  fReadOnly = false;
590  WriteVersion(CLIENT_VERSION);
591  fReadOnly = fTmp;
592  }
593  }
594  ++env->mapFileUseCount[strFilename];
595  strFile = strFilename;
596  }
597 }

677 bool BerkeleyBatch::Rewrite(BerkeleyDatabase& database, const char* pszSkip)
678 {
679  if (database.IsDummy()) {
680  return true;
681  }
682  BerkeleyEnvironment *env = database.env.get();
683  const std::string& strFile = database.strFile;
684  while (true) {
685  {
686  LOCK(cs_db);
687  if (!env->mapFileUseCount.count(strFile) || env->mapFileUseCount[strFile] == 0) {
688  // Flush log data to the dat file
689  env->CloseDb(strFile);
690  env->CheckpointLSN(strFile);
691  env->mapFileUseCount.erase(strFile);
692 
693  bool fSuccess = true;
694  LogPrintf("BerkeleyBatch::Rewrite: Rewriting %s...\n", strFile);
695  std::string strFileRes = strFile + ".rewrite";
696  { // surround usage of db with extra {}
697  BerkeleyBatch db(database, "r");
698  std::unique_ptr<Db> pdbCopy = MakeUnique<Db>(env->dbenv.get(), 0);
699 
700  int ret = pdbCopy->open(nullptr, // Txn pointer
701  strFileRes.c_str(), // Filename
702  "main", // Logical db name
703  DB_BTREE, // Database type
704  DB_CREATE, // Flags
705  0);
706  if (ret > 0) {
707  LogPrintf("BerkeleyBatch::Rewrite: Can't create database file %s\n", strFileRes);
708  fSuccess = false;
709  }
710 
711  Dbc* pcursor = db.GetCursor();
712  if (pcursor)
713  while (fSuccess) {
714  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
715  CDataStream ssValue(SER_DISK, CLIENT_VERSION);
716  int ret1 = db.ReadAtCursor(pcursor, ssKey, ssValue);
717  if (ret1 == DB_NOTFOUND) {
718  pcursor->close();
719  break;
720  } else if (ret1 != 0) {
721  pcursor->close();
722  fSuccess = false;
723  break;
724  }
725  if (pszSkip &&
726  strncmp(ssKey.data(), pszSkip, std::min(ssKey.size(), strlen(pszSkip))) == 0)
727  continue;
728  if (strncmp(ssKey.data(), "\x07version", 8) == 0) {
729  // Update version:
730  ssValue.clear();
731  ssValue << CLIENT_VERSION;
732  }
733  Dbt datKey(ssKey.data(), ssKey.size());
734  Dbt datValue(ssValue.data(), ssValue.size());
735  int ret2 = pdbCopy->put(nullptr, &datKey, &datValue, DB_NOOVERWRITE);
736  if (ret2 > 0)
737  fSuccess = false;
738  }
739  if (fSuccess) {
740  db.Close();
741  env->CloseDb(strFile);
742  if (pdbCopy->close(0))
743  fSuccess = false;
744  } else {
745  pdbCopy->close(0);
746  }
747  }
748  if (fSuccess) {
749  Db dbA(env->dbenv.get(), 0);
750  if (dbA.remove(strFile.c_str(), nullptr, 0))
751  fSuccess = false;
752  Db dbB(env->dbenv.get(), 0);
753  if (dbB.rename(strFileRes.c_str(), nullptr, strFile.c_str(), 0))
754  fSuccess = false;
755  }
756  if (!fSuccess)
757  LogPrintf("BerkeleyBatch::Rewrite: Failed to rewrite database file %s\n", strFileRes);
758  return fSuccess;
759  }
760  }
761  MilliSleep(100);
762  }
