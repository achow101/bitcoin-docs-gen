95 //
96 // Thread management and startup/shutdown:
97 //
98 // The network-processing threads are all part of a thread group
99 // created by AppInit() or the Qt main() function.
100 //
101 // A clean exit happens when StartShutdown() or the SIGTERM
102 // signal handler sets ShutdownRequested(), which makes main thread's
103 // WaitForShutdown() interrupts the thread group.
104 // And then, WaitForShutdown() makes all other on-going threads
105 // in the thread group join the main thread.
106 // Shutdown() is then called to clean up database connections, and stop other
107 // threads that should only be stopped after the main network-processing
108 // threads have exited.
109 //
110 // Shutdown for Qt is very similar, only it uses a QTimer to detect
111 // ShutdownRequested() getting set, and then does the normal Qt
112 // shutdown thing.
113 //

161 void Shutdown(InitInterfaces& interfaces)
162 {
163  LogPrintf("%s: In progress...\n", __func__);
164  static CCriticalSection cs_Shutdown;
165  TRY_LOCK(cs_Shutdown, lockShutdown);
166  if (!lockShutdown)
167  return;
168 
173  RenameThread("bitcoin-shutoff");
174  mempool.AddTransactionsUpdated(1);
175 
176  StopHTTPRPC();
177  StopREST();
178  StopRPC();
179  StopHTTPServer();
180  for (const auto& client : interfaces.chain_clients) {
181  client->flush();
182  }
183  StopMapPort();
184 
185  // Because these depend on each-other, we make sure that neither can be
186  // using the other before destroying them.
187  if (peerLogic) UnregisterValidationInterface(peerLogic.get());
188  if (g_connman) g_connman->Stop();
189  if (g_txindex) g_txindex->Stop();
190 
191  StopTorControl();
192 
193  // After everything has been shut down, but before things get flushed, stop the
194  // CScheduler/checkqueue threadGroup
195  threadGroup.interrupt_all();
196  threadGroup.join_all();
197 
198  // After the threads that potentially access these pointers have been stopped,
199  // destruct and reset all to nullptr.
200  peerLogic.reset();
201  g_connman.reset();
202  g_txindex.reset();
203 
204  if (g_is_mempool_loaded && gArgs.GetArg("-persistmempool", DEFAULT_PERSIST_MEMPOOL)) {
205  DumpMempool();
206  }
207 
208  if (fFeeEstimatesInitialized)
209  {
210  ::feeEstimator.FlushUnconfirmed();
211  fs::path est_path = GetDataDir() / FEE_ESTIMATES_FILENAME;
212  CAutoFile est_fileout(fsbridge::fopen(est_path, "wb"), SER_DISK, CLIENT_VERSION);
213  if (!est_fileout.IsNull())
214  ::feeEstimator.Write(est_fileout);
215  else
216  LogPrintf("%s: Failed to write fee estimates to %s\n", __func__, est_path.string());
217  fFeeEstimatesInitialized = false;
218  }
219 
220  // FlushStateToDisk generates a ChainStateFlushed callback, which we should avoid missing
221  if (pcoinsTip != nullptr) {
222  FlushStateToDisk();
223  }
224 
225  // After there are no more peers/RPC left to give us new data which may generate
226  // CValidationInterface callbacks, flush them...
227  GetMainSignals().FlushBackgroundCallbacks();
228 
229  // Any future callbacks will be dropped. This should absolutely be safe - if
230  // missing a callback results in an unrecoverable situation, unclean shutdown
231  // would too. The only reason to do the above flushes is to let the wallet catch
232  // up with our current chain to avoid any strange pruning edge cases and make
233  // next startup faster by avoiding rescan.
234 
235  {
236  LOCK(cs_main);
237  if (pcoinsTip != nullptr) {
238  FlushStateToDisk();
239  }
240  pcoinsTip.reset();
241  pcoinscatcher.reset();
242  pcoinsdbview.reset();
243  pblocktree.reset();
244  }
245  for (const auto& client : interfaces.chain_clients) {
246  client->stop();
247  }
248 
249 #if ENABLE_ZMQ
250  if (g_zmq_notification_interface) {
251  UnregisterValidationInterface(g_zmq_notification_interface);
252  delete g_zmq_notification_interface;
253  g_zmq_notification_interface = nullptr;
254  }
255 #endif
256 
257 #ifndef WIN32
258  try {
259  fs::remove(GetPidFile());
260  } catch (const fs::filesystem_error& e) {
261  LogPrintf("%s: Unable to remove pidfile: %s\n", __func__, e.what());
262  }
263 #endif
264  interfaces.chain_clients.clear();
265  UnregisterAllValidationInterfaces();
266  GetMainSignals().UnregisterBackgroundSignalScheduler();
267  GetMainSignals().UnregisterWithMempoolSignals(mempool);
268  globalVerifyHandle.reset();
269  ECC_Stop();
270  LogPrintf("%s: done\n", __func__);
271 }

588 // If we're using -prune with -reindex, then delete block files that will be ignored by the
589 // reindex. Since reindexing works by starting at block file 0 and looping until a blockfile
590 // is missing, do the same here to delete any later block files after a gap. Also delete all
591 // rev files since they'll be rewritten by the reindex anyway. This ensures that vinfoBlockFile
592 // is in sync with what's actually on disk by the time we start downloading, so that pruning
593 // works correctly.
594 static void CleanupBlockRevFiles()
595 {
596  std::map<std::string, fs::path> mapBlockFiles;
597 
598  // Glob all blk?????.dat and rev?????.dat files from the blocks directory.
599  // Remove the rev files immediately and insert the blk file paths into an
600  // ordered map keyed by block file index.
601  LogPrintf("Removing unusable blk?????.dat and rev?????.dat files for -reindex with -prune\n");
602  fs::path blocksdir = GetBlocksDir();
603  for (fs::directory_iterator it(blocksdir); it != fs::directory_iterator(); it++) {
604  if (fs::is_regular_file(*it) &&
605  it->path().filename().string().length() == 12 &&
606  it->path().filename().string().substr(8,4) == ".dat")
607  {
608  if (it->path().filename().string().substr(0,3) == "blk")
609  mapBlockFiles[it->path().filename().string().substr(3,5)] = it->path();
610  else if (it->path().filename().string().substr(0,3) == "rev")
611  remove(it->path());
612  }
613  }
614 
615  // Remove all block files that aren't part of a contiguous set starting at
616  // zero by walking the ordered map (keys are block file indices) by
617  // keeping a separate counter. Once we hit a gap (or if 0 doesn't exist)
618  // start removing block files.
619  int nContigCounter = 0;
620  for (const std::pair<const std::string, fs::path>& item : mapBlockFiles) {
621  if (atoi(item.first) == nContigCounter) {
622  nContigCounter++;
623  continue;
624  }
625  remove(item.second);
626  }
627 }

629 static void ThreadImport(std::vector<fs::path> vImportFiles)
630 {
631  const CChainParams& chainparams = Params();
632  RenameThread("bitcoin-loadblk");
633  ScheduleBatchPriority();
634 
635  {
636  CImportingNow imp;
637 
638  // -reindex
639  if (fReindex) {
640  int nFile = 0;
641  while (true) {
642  CDiskBlockPos pos(nFile, 0);
643  if (!fs::exists(GetBlockPosFilename(pos, "blk")))
644  break; // No block files left to reindex
645  FILE *file = OpenBlockFile(pos, true);
646  if (!file)
647  break; // This error is logged in OpenBlockFile
648  LogPrintf("Reindexing block file blk%05u.dat...\n", (unsigned int)nFile);
649  LoadExternalBlockFile(chainparams, file, &pos);
650  nFile++;
651  }
652  pblocktree->WriteReindexing(false);
653  fReindex = false;
654  LogPrintf("Reindexing finished\n");
655  // To avoid ending up in a situation without genesis block, re-try initializing (no-op if reindexing worked):
656  LoadGenesisBlock(chainparams);
657  }
658 
659  // hardcoded $DATADIR/bootstrap.dat
660  fs::path pathBootstrap = GetDataDir() / "bootstrap.dat";
661  if (fs::exists(pathBootstrap)) {
662  FILE *file = fsbridge::fopen(pathBootstrap, "rb");
663  if (file) {
664  fs::path pathBootstrapOld = GetDataDir() / "bootstrap.dat.old";
665  LogPrintf("Importing bootstrap.dat...\n");
666  LoadExternalBlockFile(chainparams, file);
667  RenameOver(pathBootstrap, pathBootstrapOld);
668  } else {
669  LogPrintf("Warning: Could not open bootstrap file %s\n", pathBootstrap.string());
670  }
671  }
672 
673  // -loadblock=
674  for (const fs::path& path : vImportFiles) {
675  FILE *file = fsbridge::fopen(path, "rb");
676  if (file) {
677  LogPrintf("Importing blocks file %s...\n", path.string());
678  LoadExternalBlockFile(chainparams, file);
679  } else {
680  LogPrintf("Warning: Could not open blocks file %s\n", path.string());
681  }
682  }
683 
684  // scan for better chains in the block chain database, that are not yet connected in the active best chain
685  CValidationState state;
686  if (!ActivateBestChain(state, chainparams)) {
687  LogPrintf("Failed to connect best block (%s)\n", FormatStateMessage(state));
688  StartShutdown();
689  return;
690  }
691 
692  if (gArgs.GetBoolArg("-stopafterblockimport", DEFAULT_STOPAFTERBLOCKIMPORT)) {
693  LogPrintf("Stopping after block import\n");
694  StartShutdown();
695  return;
696  }
697  } // End scope of CImportingNow
698  if (gArgs.GetArg("-persistmempool", DEFAULT_PERSIST_MEMPOOL)) {
699  LoadMempool();
700  }
701  g_is_mempool_loaded = !ShutdownRequested();
702 }

763  // to protect privacy, do not listen by default if a default proxy server is specified
764  if (gArgs.SoftSetBoolArg("-listen", false))
765  LogPrintf("%s: parameter interaction: -proxy set -> setting -listen=0\n", __func__);

769  LogPrintf("%s: parameter interaction: -proxy set -> setting -upnp=0\n", __func__);
770  // to protect privacy, do not discover addresses by default
771  if (gArgs.SoftSetBoolArg("-discover", false))
772  LogPrintf("%s: parameter interaction: -proxy set -> setting -discover=0\n", __func__);
773  }
774 
775  if (!gArgs.GetBoolArg("-listen", DEFAULT_LISTEN)) {
776  // do not map ports or try to retrieve public IP when not listening (pointless)
777  if (gArgs.SoftSetBoolArg("-upnp", false))
778  LogPrintf("%s: parameter interaction: -listen=0 -> setting -upnp=0\n", __func__);
