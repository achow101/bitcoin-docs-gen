128  inline void SerializationOp(Stream& s, Operation ser_action) {
129  int nVersion = s.GetVersion();
130  if (!(s.GetType() & SER_GETHASH))
131  READWRITE(nVersion);
132  READWRITE(nTime);
133  READWRITE(vchPubKey);

135  try {
136  READWRITE(fInternal);
