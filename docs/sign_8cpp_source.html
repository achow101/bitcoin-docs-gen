89 static bool SignStep(const SigningProvider& provider, const BaseSignatureCreator& creator, const CScript& scriptPubKey,
90  std::vector<valtype>& ret, txnouttype& whichTypeRet, SigVersion sigversion, SignatureData& sigdata)
91 {
92  CScript scriptRet;
93  uint160 h160;
94  ret.clear();
95  std::vector<unsigned char> sig;
96 
97  std::vector<valtype> vSolutions;
98  if (!Solver(scriptPubKey, whichTypeRet, vSolutions))
99  return false;
100 
101  switch (whichTypeRet)
102  {
103  case TX_NONSTANDARD:
104  case TX_NULL_DATA:
105  case TX_WITNESS_UNKNOWN:
106  return false;
107  case TX_PUBKEY:
108  if (!CreateSig(creator, sigdata, provider, sig, CPubKey(vSolutions[0]).GetID(), scriptPubKey, sigversion)) return false;
109  ret.push_back(std::move(sig));
110  return true;
111  case TX_PUBKEYHASH: {
112  CKeyID keyID = CKeyID(uint160(vSolutions[0]));
113  if (!CreateSig(creator, sigdata, provider, sig, keyID, scriptPubKey, sigversion)) return false;
114  ret.push_back(std::move(sig));
115  CPubKey pubkey;
116  GetPubKey(provider, sigdata, keyID, pubkey);
117  ret.push_back(ToByteVector(pubkey));
118  return true;
119  }
120  case TX_SCRIPTHASH:
121  if (GetCScript(provider, sigdata, uint160(vSolutions[0]), scriptRet)) {
122  ret.push_back(std::vector<unsigned char>(scriptRet.begin(), scriptRet.end()));
123  return true;
124  }
125  return false;
126 
127  case TX_MULTISIG: {
128  size_t required = vSolutions.front()[0];
129  ret.push_back(valtype()); // workaround CHECKMULTISIG bug
130  for (size_t i = 1; i < vSolutions.size() - 1; ++i) {
131  CPubKey pubkey = CPubKey(vSolutions[i]);
132  if (ret.size() < required + 1 && CreateSig(creator, sigdata, provider, sig, pubkey.GetID(), scriptPubKey, sigversion)) {
133  ret.push_back(std::move(sig));
134  }
135  }
136  bool ok = ret.size() == required + 1;
137  for (size_t i = 0; i + ret.size() < required + 1; ++i) {
138  ret.push_back(valtype());
139  }
140  return ok;
141  }
142  case TX_WITNESS_V0_KEYHASH:
143  ret.push_back(vSolutions[0]);
144  return true;
145 
146  case TX_WITNESS_V0_SCRIPTHASH:
147  CRIPEMD160().Write(&vSolutions[0][0], vSolutions[0].size()).Finalize(h160.begin());
148  if (GetCScript(provider, sigdata, h160, scriptRet)) {
149  ret.push_back(std::vector<unsigned char>(scriptRet.begin(), scriptRet.end()));
150  return true;
151  }
152  return false;
153 
154  default:
155  return false;
156  }

174 bool ProduceSignature(const SigningProvider& provider, const BaseSignatureCreator& creator, const CScript& fromPubKey, SignatureData& sigdata)
175 {
176  if (sigdata.complete) return true;
177 
178  std::vector<valtype> result;
179  txnouttype whichType;
180  bool solved = SignStep(provider, creator, fromPubKey, result, whichType, SigVersion::BASE, sigdata);
181  bool P2SH = false;
182  CScript subscript;
183  sigdata.scriptWitness.stack.clear();
184 
185  if (solved && whichType == TX_SCRIPTHASH)
186  {
187  // Solver returns the subscript that needs to be evaluated;
188  // the final scriptSig is the signatures from that
189  // and then the serialized subscript:
190  subscript = CScript(result[0].begin(), result[0].end());
191  sigdata.redeem_script = subscript;
192  solved = solved && SignStep(provider, creator, subscript, result, whichType, SigVersion::BASE, sigdata) && whichType != TX_SCRIPTHASH;
193  P2SH = true;
194  }
195 
196  if (solved && whichType == TX_WITNESS_V0_KEYHASH)
197  {
198  CScript witnessscript;
199  witnessscript << OP_DUP << OP_HASH160 << ToByteVector(result[0]) << OP_EQUALVERIFY << OP_CHECKSIG;
200  txnouttype subType;
201  solved = solved && SignStep(provider, creator, witnessscript, result, subType, SigVersion::WITNESS_V0, sigdata);
202  sigdata.scriptWitness.stack = result;
203  result.clear();
204  }
205  else if (solved && whichType == TX_WITNESS_V0_SCRIPTHASH)
206  {
207  CScript witnessscript(result[0].begin(), result[0].end());
208  sigdata.witness_script = witnessscript;
209  txnouttype subType;
210  solved = solved && SignStep(provider, creator, witnessscript, result, subType, SigVersion::WITNESS_V0, sigdata) && subType != TX_SCRIPTHASH && subType != TX_WITNESS_V0_SCRIPTHASH && subType != TX_WITNESS_V0_KEYHASH;
211  result.push_back(std::vector<unsigned char>(witnessscript.begin(), witnessscript.end()));
212  sigdata.scriptWitness.stack = result;
213  result.clear();
214  }
215 
216  if (P2SH) {
217  result.push_back(std::vector<unsigned char>(subscript.begin(), subscript.end()));
218  }
219  sigdata.scriptSig = PushAll(result);
220 
221  // Test solution
222  sigdata.complete = solved && VerifyScript(sigdata.scriptSig, fromPubKey, &sigdata.scriptWitness, STANDARD_SCRIPT_VERIFY_FLAGS, creator.Checker());
223  return sigdata.complete;
224 }

270 SignatureData DataFromTransaction(const CMutableTransaction& tx, unsigned int nIn, const CTxOut& txout)
271 {
272  SignatureData data;
273  assert(tx.vin.size() > nIn);
274  data.scriptSig = tx.vin[nIn].scriptSig;
275  data.scriptWitness = tx.vin[nIn].scriptWitness;
276  Stacks stack(data);
277 
278  // Get signatures
279  MutableTransactionSignatureChecker tx_checker(&tx, nIn, txout.nValue);
280  SignatureExtractorChecker extractor_checker(data, tx_checker);
281  if (VerifyScript(data.scriptSig, txout.scriptPubKey, &data.scriptWitness, STANDARD_SCRIPT_VERIFY_FLAGS, extractor_checker)) {
282  data.complete = true;
283  return data;
284  }
285 
286  // Get scripts
287  txnouttype script_type;
288  std::vector<std::vector<unsigned char>> solutions;
289  Solver(txout.scriptPubKey, script_type, solutions);
290  SigVersion sigversion = SigVersion::BASE;
291  CScript next_script = txout.scriptPubKey;
292 
293  if (script_type == TX_SCRIPTHASH && !stack.script.empty() && !stack.script.back().empty()) {
294  // Get the redeemScript
295  CScript redeem_script(stack.script.back().begin(), stack.script.back().end());
296  data.redeem_script = redeem_script;
297  next_script = std::move(redeem_script);
298 
299  // Get redeemScript type
300  Solver(next_script, script_type, solutions);
301  stack.script.pop_back();
302  }
303  if (script_type == TX_WITNESS_V0_SCRIPTHASH && !stack.witness.empty() && !stack.witness.back().empty()) {
304  // Get the witnessScript
305  CScript witness_script(stack.witness.back().begin(), stack.witness.back().end());
306  data.witness_script = witness_script;
307  next_script = std::move(witness_script);
308 
309  // Get witnessScript type
310  Solver(next_script, script_type, solutions);
311  stack.witness.pop_back();
312  stack.script = std::move(stack.witness);
313  stack.witness.clear();
314  sigversion = SigVersion::WITNESS_V0;
315  }
316  if (script_type == TX_MULTISIG && !stack.script.empty()) {
317  // Build a map of pubkey -> signature by matching sigs to pubkeys:
318  assert(solutions.size() > 1);
319  unsigned int num_pubkeys = solutions.size()-2;
320  unsigned int last_success_key = 0;
321  for (const valtype& sig : stack.script) {
322  for (unsigned int i = last_success_key; i < num_pubkeys; ++i) {
323  const valtype& pubkey = solutions[i+1];
324  // We either have a signature for this pubkey, or we have found a signature and it is valid
325  if (data.signatures.count(CPubKey(pubkey).GetID()) || extractor_checker.CheckSig(sig, pubkey, next_script, sigversion)) {
326  last_success_key = i + 1;
327  break;
328  }
329  }
330  }
331  }
332 
333  return data;
334 }
