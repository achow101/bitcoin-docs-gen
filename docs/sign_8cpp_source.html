95 static bool SignStep(const SigningProvider& provider, const BaseSignatureCreator& creator, const CScript& scriptPubKey,
96  std::vector<valtype>& ret, txnouttype& whichTypeRet, SigVersion sigversion, SignatureData& sigdata)
97 {
98  CScript scriptRet;
99  uint160 h160;
100  ret.clear();
101  std::vector<unsigned char> sig;
102 
103  std::vector<valtype> vSolutions;
104  whichTypeRet = Solver(scriptPubKey, vSolutions);
105 
106  switch (whichTypeRet)
107  {
108  case TX_NONSTANDARD:
109  case TX_NULL_DATA:
110  case TX_WITNESS_UNKNOWN:
111  return false;
112  case TX_PUBKEY:
113  if (!CreateSig(creator, sigdata, provider, sig, CPubKey(vSolutions[0]), scriptPubKey, sigversion)) return false;
114  ret.push_back(std::move(sig));
115  return true;
116  case TX_PUBKEYHASH: {
117  CKeyID keyID = CKeyID(uint160(vSolutions[0]));
118  CPubKey pubkey;
119  if (!GetPubKey(provider, sigdata, keyID, pubkey)) return false;
120  if (!CreateSig(creator, sigdata, provider, sig, pubkey, scriptPubKey, sigversion)) return false;
121  ret.push_back(std::move(sig));
122  ret.push_back(ToByteVector(pubkey));
123  return true;
124  }
125  case TX_SCRIPTHASH:
126  if (GetCScript(provider, sigdata, uint160(vSolutions[0]), scriptRet)) {
127  ret.push_back(std::vector<unsigned char>(scriptRet.begin(), scriptRet.end()));
128  return true;
129  }
130  return false;
131 
132  case TX_MULTISIG: {
133  size_t required = vSolutions.front()[0];
134  ret.push_back(valtype()); // workaround CHECKMULTISIG bug
135  for (size_t i = 1; i < vSolutions.size() - 1; ++i) {
136  CPubKey pubkey = CPubKey(vSolutions[i]);
137  if (ret.size() < required + 1 && CreateSig(creator, sigdata, provider, sig, pubkey, scriptPubKey, sigversion)) {
138  ret.push_back(std::move(sig));
139  }
140  }
141  bool ok = ret.size() == required + 1;
142  for (size_t i = 0; i + ret.size() < required + 1; ++i) {
143  ret.push_back(valtype());
144  }
145  return ok;
146  }
147  case TX_WITNESS_V0_KEYHASH:
148  ret.push_back(vSolutions[0]);
149  return true;
150 
151  case TX_WITNESS_V0_SCRIPTHASH:
152  CRIPEMD160().Write(&vSolutions[0][0], vSolutions[0].size()).Finalize(h160.begin());
153  if (GetCScript(provider, sigdata, h160, scriptRet)) {
154  ret.push_back(std::vector<unsigned char>(scriptRet.begin(), scriptRet.end()));
155  return true;
156  }
157  return false;
158 
159  default:
160  return false;
161  }

179 bool ProduceSignature(const SigningProvider& provider, const BaseSignatureCreator& creator, const CScript& fromPubKey, SignatureData& sigdata)
180 {
181  if (sigdata.complete) return true;
182 
183  std::vector<valtype> result;
184  txnouttype whichType;
185  bool solved = SignStep(provider, creator, fromPubKey, result, whichType, SigVersion::BASE, sigdata);
186  bool P2SH = false;
187  CScript subscript;
188  sigdata.scriptWitness.stack.clear();
189 
190  if (solved && whichType == TX_SCRIPTHASH)
191  {
192  // Solver returns the subscript that needs to be evaluated;
193  // the final scriptSig is the signatures from that
194  // and then the serialized subscript:
195  subscript = CScript(result[0].begin(), result[0].end());
196  sigdata.redeem_script = subscript;
197  solved = solved && SignStep(provider, creator, subscript, result, whichType, SigVersion::BASE, sigdata) && whichType != TX_SCRIPTHASH;
198  P2SH = true;
199  }
200 
201  if (solved && whichType == TX_WITNESS_V0_KEYHASH)
202  {
203  CScript witnessscript;
204  witnessscript << OP_DUP << OP_HASH160 << ToByteVector(result[0]) << OP_EQUALVERIFY << OP_CHECKSIG;
205  txnouttype subType;
206  solved = solved && SignStep(provider, creator, witnessscript, result, subType, SigVersion::WITNESS_V0, sigdata);
207  sigdata.scriptWitness.stack = result;
208  sigdata.witness = true;
209  result.clear();
210  }
211  else if (solved && whichType == TX_WITNESS_V0_SCRIPTHASH)
212  {
213  CScript witnessscript(result[0].begin(), result[0].end());
214  sigdata.witness_script = witnessscript;
215  txnouttype subType;
216  solved = solved && SignStep(provider, creator, witnessscript, result, subType, SigVersion::WITNESS_V0, sigdata) && subType != TX_SCRIPTHASH && subType != TX_WITNESS_V0_SCRIPTHASH && subType != TX_WITNESS_V0_KEYHASH;
217  result.push_back(std::vector<unsigned char>(witnessscript.begin(), witnessscript.end()));
218  sigdata.scriptWitness.stack = result;
219  sigdata.witness = true;
220  result.clear();
221  } else if (solved && whichType == TX_WITNESS_UNKNOWN) {
222  sigdata.witness = true;
223  }
224 
225  if (P2SH) {
226  result.push_back(std::vector<unsigned char>(subscript.begin(), subscript.end()));
227  }
228  sigdata.scriptSig = PushAll(result);
229 
230  // Test solution
231  sigdata.complete = solved && VerifyScript(sigdata.scriptSig, fromPubKey, &sigdata.scriptWitness, STANDARD_SCRIPT_VERIFY_FLAGS, creator.Checker());
232  return sigdata.complete;
233 }

240 bool SignPSBTInput(const SigningProvider& provider, PartiallySignedTransaction& psbt, int index, int sighash)
241 {
242  PSBTInput& input = psbt.inputs.at(index);
243  const CMutableTransaction& tx = *psbt.tx;
244 
245  if (PSBTInputSigned(input)) {
246  return true;
247  }
248 
249  // Fill SignatureData with input info
250  SignatureData sigdata;
251  input.FillSignatureData(sigdata);
252 
253  // Get UTXO
254  bool require_witness_sig = false;
255  CTxOut utxo;
256 
257  // Verify input sanity, which checks that at most one of witness or non-witness utxos is provided.
258  if (!input.IsSane()) {
259  return false;
260  }
261 
262  if (input.non_witness_utxo) {
263  // If we're taking our information from a non-witness UTXO, verify that it matches the prevout.
264  COutPoint prevout = tx.vin[index].prevout;
265  if (input.non_witness_utxo->GetHash() != prevout.hash) {
266  return false;
267  }
268  utxo = input.non_witness_utxo->vout[prevout.n];
269  } else if (!input.witness_utxo.IsNull()) {
270  utxo = input.witness_utxo;
271  // When we're taking our information from a witness UTXO, we can't verify it is actually data from
272  // the output being spent. This is safe in case a witness signature is produced (which includes this
273  // information directly in the hash), but not for non-witness signatures. Remember that we require
274  // a witness signature in this situation.
275  require_witness_sig = true;
276  } else {
277  return false;
278  }
279 
280  MutableTransactionSignatureCreator creator(&tx, index, utxo.nValue, sighash);
281  sigdata.witness = false;
282  bool sig_complete = ProduceSignature(provider, creator, utxo.scriptPubKey, sigdata);
283  // Verify that a witness signature was produced in case one was required.
284  if (require_witness_sig && !sigdata.witness) return false;
285  input.FromSignatureData(sigdata);
286 
287  // If we have a witness signature, use the smaller witness UTXO.
288  if (sigdata.witness) {
289  input.witness_utxo = utxo;
290  input.non_witness_utxo = nullptr;
291  }
292 
293  return sig_complete;
294 }

333 SignatureData DataFromTransaction(const CMutableTransaction& tx, unsigned int nIn, const CTxOut& txout)
334 {
335  SignatureData data;
336  assert(tx.vin.size() > nIn);
337  data.scriptSig = tx.vin[nIn].scriptSig;
338  data.scriptWitness = tx.vin[nIn].scriptWitness;
339  Stacks stack(data);
340 
341  // Get signatures
342  MutableTransactionSignatureChecker tx_checker(&tx, nIn, txout.nValue);
343  SignatureExtractorChecker extractor_checker(data, tx_checker);
344  if (VerifyScript(data.scriptSig, txout.scriptPubKey, &data.scriptWitness, STANDARD_SCRIPT_VERIFY_FLAGS, extractor_checker)) {
345  data.complete = true;
346  return data;
347  }
348 
349  // Get scripts
350  std::vector<std::vector<unsigned char>> solutions;
351  txnouttype script_type = Solver(txout.scriptPubKey, solutions);
352  SigVersion sigversion = SigVersion::BASE;
353  CScript next_script = txout.scriptPubKey;
354 
355  if (script_type == TX_SCRIPTHASH && !stack.script.empty() && !stack.script.back().empty()) {
356  // Get the redeemScript
357  CScript redeem_script(stack.script.back().begin(), stack.script.back().end());
358  data.redeem_script = redeem_script;
359  next_script = std::move(redeem_script);
360 
361  // Get redeemScript type
362  script_type = Solver(next_script, solutions);
363  stack.script.pop_back();
364  }
365  if (script_type == TX_WITNESS_V0_SCRIPTHASH && !stack.witness.empty() && !stack.witness.back().empty()) {
366  // Get the witnessScript
367  CScript witness_script(stack.witness.back().begin(), stack.witness.back().end());
368  data.witness_script = witness_script;
369  next_script = std::move(witness_script);
370 
371  // Get witnessScript type
372  script_type = Solver(next_script, solutions);
373  stack.witness.pop_back();
374  stack.script = std::move(stack.witness);
375  stack.witness.clear();
376  sigversion = SigVersion::WITNESS_V0;
377  }
378  if (script_type == TX_MULTISIG && !stack.script.empty()) {
379  // Build a map of pubkey -> signature by matching sigs to pubkeys:
380  assert(solutions.size() > 1);
381  unsigned int num_pubkeys = solutions.size()-2;
382  unsigned int last_success_key = 0;
383  for (const valtype& sig : stack.script) {
384  for (unsigned int i = last_success_key; i < num_pubkeys; ++i) {
385  const valtype& pubkey = solutions[i+1];
386  // We either have a signature for this pubkey, or we have found a signature and it is valid
387  if (data.signatures.count(CPubKey(pubkey).GetID()) || extractor_checker.CheckSig(sig, pubkey, next_script, sigversion)) {
388  last_success_key = i + 1;
389  break;
390  }
391  }
392  }
393  }
394 
395  return data;
396 }

493 bool IsSolvable(const SigningProvider& provider, const CScript& script)
494 {
495  // This check is to make sure that the script we created can actually be solved for and signed by us
496  // if we were to have the private keys. This is just to make sure that the script is valid and that,
497  // if found in a transaction, we would still accept and relay that transaction. In particular,
498  // it will reject witness outputs that require signing with an uncompressed public key.
499  SignatureData sigs;
500  // Make sure that STANDARD_SCRIPT_VERIFY_FLAGS includes SCRIPT_VERIFY_WITNESS_PUBKEYTYPE, the most
501  // important property this function is designed to test for.
502  static_assert(STANDARD_SCRIPT_VERIFY_FLAGS & SCRIPT_VERIFY_WITNESS_PUBKEYTYPE, "IsSolvable requires standard script flags to include WITNESS_PUBKEYTYPE");
503  if (ProduceSignature(provider, DUMMY_SIGNATURE_CREATOR, script, sigs)) {
504  // VerifyScript check is just defensive, and should never fail.
505  bool verified = VerifyScript(sigs.scriptSig, script, &sigs.scriptWitness, STANDARD_SCRIPT_VERIFY_FLAGS, DUMMY_CHECKER);
506  assert(verified);
507  return true;
508  }
509  return false;
510 }
