33// parses a comma separated list like "10,20,30,50"
34static std::vector<double> parseAsymptote(const std::string& str) {
35 std::stringstream ss(str);
36 std::vector<double> numbers;
37 double d;
38 char c;
39 while (ss >> d) {
40 numbers.push_back(d);
41 ss >> c;
42 }
43 return numbers;
44}
