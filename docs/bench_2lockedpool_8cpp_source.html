15{

19
20 std::vector<void*> addr;
21 for (int x=0; x<ASIZE; ++x)
22 addr.push_back(nullptr);
23 uint32_t s = 0x12345678;
24 bench.run([&] {

26 if (s & 0x80000000) {
27 b.free(addr[idx]);
28 addr[idx] = nullptr;
29 } else if (!addr[idx]) {
30 addr[idx] = b.alloc((s >> 16) & (MSIZE - 1));
31 }
32 bool lsb = s & 1;
33 s >>= 1;
34 if (lsb)
35 s ^= 0xf00f00f0; // LFSR period 0xf7ffffe0
36 });
37 for (void *ptr: addr)
38 b.free(ptr);
39 addr.clear();
40}
