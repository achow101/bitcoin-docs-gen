238 explicit CScriptNum(const std::vector<unsigned char>& vch, bool fRequireMinimal,
239 const size_t nMaxNumSize = nDefaultMaxNumSize)
240 {
241 if (vch.size() > nMaxNumSize) {
242 throw scriptnum_error("script number overflow");
243 }
244 if (fRequireMinimal && vch.size() > 0) {
245 // Check that the number is encoded with the minimum possible
246 // number of bytes.
247 //
248 // If the most-significant-byte - excluding the sign bit - is zero
249 // then we're not minimal. Note how this test also rejects the
250 // negative-zero encoding, 0x80.
251 if ((vch.back() & 0x7f) == 0) {
252 // One exception: if there's more than one byte and the most
253 // significant bit of the second-most-significant-byte is set
254 // it would conflict with the sign bit. An example of this case
255 // is +-255, which encode to 0xff00 and 0xff80 respectively.
256 // (big-endian).
257 if (vch.size() <= 1 || (vch[vch.size() - 2] & 0x80) == 0) {
258 throw scriptnum_error("non-minimally encoded script number");
259 }
260 }
261 }
262 m_value = set_vch(vch);
263 }

340 static std::vector<unsigned char> serialize(const int64_t& value)
341 {
342 if(value == 0)
343 return std::vector<unsigned char>();
344
345 std::vector<unsigned char> result;
346 const bool neg = value < 0;
347 uint64_t absvalue = neg ? ~static_cast<uint64_t>(value) + 1 : static_cast<uint64_t>(value);
348
349 while(absvalue)
350 {
351 result.push_back(absvalue & 0xff);
352 absvalue >>= 8;
353 }
354
355// - If the most significant byte is >= 0x80 and the value is positive, push a
356// new zero-byte to make the significant byte < 0x80 again.
357
358// - If the most significant byte is >= 0x80 and the value is negative, push a
359// new 0x80 byte that will be popped off when converting to an integral.
360
361// - If the most significant byte is < 0x80 and the value is negative, add
362// 0x80 to it, since it will be subtracted and interpreted as a negative when
363// converting to an integral.
364
365 if (result.back() & 0x80)
366 result.push_back(neg ? 0x80 : 0);
367 else if (neg)
368 result.back() |= 0x80;
369
370 return result;
371 }
372
373private:
374 static int64_t set_vch(const std::vector<unsigned char>& vch)
375 {
376 if (vch.empty())
377 return 0;
378
379 int64_t result = 0;
380 for (size_t i = 0; i != vch.size(); ++i)
381 result |= static_cast<int64_t>(vch[i]) << 8*i;
382
383 // If the input vector's most significant byte is 0x80, remove it from
384 // the result's msb and return a negative.
385 if (vch.back() & 0x80)
386 return -((int64_t)(result & ~(0x80ULL << (8 * (vch.size() - 1)))));
387
388 return result;
389 }

391 int64_t m_value;
