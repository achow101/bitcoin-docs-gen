110 std::string HexStr(const T itbegin, const T itend, bool fSpaces=false)
111 {
112  std::string rv;
113  static const char hexmap[16] = { '0', '1', '2', '3', '4', '5', '6', '7',
114  '8', '9', 'a', 'b', 'c', 'd', 'e', 'f' };
115  rv.reserve((itend-itbegin)*3);
116  for(T it = itbegin; it < itend; ++it)
117  {
118  unsigned char val = (unsigned char)(*it);
119  if(fSpaces && it != itbegin)
120  rv.push_back(' ');
121  rv.push_back(hexmap[val>>4]);
122  rv.push_back(hexmap[val&15]);
123  }
124 
125  return rv;
126 }

163 template<int frombits, int tobits, bool pad, typename O, typename I>
164 bool ConvertBits(const O& outfn, I it, I end) {
165  size_t acc = 0;
166  size_t bits = 0;
167  constexpr size_t maxv = (1 << tobits) - 1;
168  constexpr size_t max_acc = (1 << (frombits + tobits - 1)) - 1;
169  while (it != end) {
170  acc = ((acc << frombits) | *it) & max_acc;
171  bits += frombits;
172  while (bits >= tobits) {
173  bits -= tobits;
174  outfn((acc >> bits) & maxv);
175  }
176  ++it;
177  }
178  if (pad) {
179  if (bits) outfn((acc << (tobits - bits)) & maxv);
180  } else if (bits >= frombits || ((acc << (tobits - bits)) & maxv)) {
181  return false;
182  }
183  return true;
184 }
