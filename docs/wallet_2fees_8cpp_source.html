37 CFeeRate GetMinimumFeeRate(const CWallet& wallet, const CCoinControl& coin_control, FeeCalculation* feeCalc)
38 {
39  /* User control of how to calculate fee uses the following parameter precedence:
40  1. coin_control.m_feerate
41  2. coin_control.m_confirm_target
42  3. m_pay_tx_fee (user-set member variable of wallet)
43  4. m_confirm_target (user-set member variable of wallet)
44  The first parameter that is set is used.
45  */
46  CFeeRate feerate_needed;
47  if (coin_control.m_feerate) { // 1.
48  feerate_needed = *(coin_control.m_feerate);
49  if (feeCalc) feeCalc->reason = FeeReason::PAYTXFEE;
50  // Allow to override automatic min/max check over coin control instance
51  if (coin_control.fOverrideFeeRate) return feerate_needed;
52  }
53  else if (!coin_control.m_confirm_target && wallet.m_pay_tx_fee != CFeeRate(0)) { // 3. TODO: remove magic value of 0 for wallet member m_pay_tx_fee
54  feerate_needed = wallet.m_pay_tx_fee;
55  if (feeCalc) feeCalc->reason = FeeReason::PAYTXFEE;
56  }
57  else { // 2. or 4.
58  // We will use smart fee estimation
59  unsigned int target = coin_control.m_confirm_target ? *coin_control.m_confirm_target : wallet.m_confirm_target;
60  // By default estimates are economical iff we are signaling opt-in-RBF
61  bool conservative_estimate = !coin_control.m_signal_bip125_rbf.get_value_or(wallet.m_signal_rbf);
62  // Allow to override the default fee estimate mode over the CoinControl instance
63  if (coin_control.m_fee_mode == FeeEstimateMode::CONSERVATIVE) conservative_estimate = true;
64  else if (coin_control.m_fee_mode == FeeEstimateMode::ECONOMICAL) conservative_estimate = false;
65 
66  feerate_needed = wallet.chain().estimateSmartFee(target, conservative_estimate, feeCalc);
67  if (feerate_needed == CFeeRate(0)) {
68  // if we don't have enough data for estimateSmartFee, then use fallback fee
69  feerate_needed = wallet.m_fallback_fee;
70  if (feeCalc) feeCalc->reason = FeeReason::FALLBACK;
71 
72  // directly return if fallback fee is disabled (feerate 0 == disabled)
73  if (wallet.m_fallback_fee == CFeeRate(0)) return feerate_needed;
74  }
75  // Obey mempool min fee when using smart fee estimation
76  CFeeRate min_mempool_feerate = wallet.chain().mempoolMinFee();
77  if (feerate_needed < min_mempool_feerate) {
78  feerate_needed = min_mempool_feerate;
79  if (feeCalc) feeCalc->reason = FeeReason::MEMPOOL_MIN;
80  }
81  }
82 
83  // prevent user from paying a fee below the required fee rate
84  CFeeRate required_feerate = GetRequiredFeeRate(wallet);
85  if (required_feerate > feerate_needed) {
86  feerate_needed = required_feerate;
87  if (feeCalc) feeCalc->reason = FeeReason::REQUIRED;
88  }
89  return feerate_needed;
90 }
