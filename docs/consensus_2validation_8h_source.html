88template <typename Result>
89class ValidationState
90{
91private:
92 enum class ModeState {
93 M_VALID,
94 M_INVALID,
95 M_ERROR,
96 } m_mode{ModeState::M_VALID};
97 Result m_result{};
98 std::string m_reject_reason;
99 std::string m_debug_message;
100
101public:
102 bool Invalid(Result result,
103 const std::string& reject_reason = "",
104 const std::string& debug_message = "")
105 {
106 m_result = result;
107 m_reject_reason = reject_reason;
108 m_debug_message = debug_message;
109 if (m_mode != ModeState::M_ERROR) m_mode = ModeState::M_INVALID;
110 return false;
111 }
112 bool Error(const std::string& reject_reason)
113 {
114 if (m_mode == ModeState::M_VALID)
115 m_reject_reason = reject_reason;
116 m_mode = ModeState::M_ERROR;
117 return false;
118 }
119 bool IsValid() const { return m_mode == ModeState::M_VALID; }
120 bool IsInvalid() const { return m_mode == ModeState::M_INVALID; }
121 bool IsError() const { return m_mode == ModeState::M_ERROR; }
122 Result GetResult() const { return m_result; }
123 std::string GetRejectReason() const { return m_reject_reason; }
124 std::string GetDebugMessage() const { return m_debug_message; }
125 std::string ToString() const
126 {
127 if (IsValid()) {
128 return "Valid";
129 }
130
131 if (!m_debug_message.empty()) {
132 return m_reject_reason + ", " + m_debug_message;
133 }
134
135 return m_reject_reason;
136 }

142// These implement the weight = (stripped_size * 4) + witness_size formula,
143// using only serialization with and without witness data. As witness_size
144// is equal to total_size - stripped_size, this formula is identical to:
145// weight = (stripped_size * 3) + total_size.
146static inline int64_t GetTransactionWeight(const CTransaction& tx)
147{
148 return ::GetSerializeSize(tx, PROTOCOL_VERSION | SERIALIZE_TRANSACTION_NO_WITNESS) * (WITNESS_SCALE_FACTOR - 1) + ::GetSerializeSize(tx, PROTOCOL_VERSION);
149}
150static inline int64_t GetBlockWeight(const CBlock& block)
151{
152 return ::GetSerializeSize(block, PROTOCOL_VERSION | SERIALIZE_TRANSACTION_NO_WITNESS) * (WITNESS_SCALE_FACTOR - 1) + ::GetSerializeSize(block, PROTOCOL_VERSION);
153}
154static inline int64_t GetTransactionInputWeight(const CTxIn& txin)
155{
156 // scriptWitness size is added here because witnesses and txins are split up in segwit serialization.
157 return ::GetSerializeSize(txin, PROTOCOL_VERSION | SERIALIZE_TRANSACTION_NO_WITNESS) * (WITNESS_SCALE_FACTOR - 1) + ::GetSerializeSize(txin, PROTOCOL_VERSION) + ::GetSerializeSize(txin.scriptWitness.stack, PROTOCOL_VERSION);
158}

161inline int GetWitnessCommitmentIndex(const CBlock& block)
162{
163 int commitpos = NO_WITNESS_COMMITMENT;
164 if (!block.vtx.empty()) {
165 for (size_t o = 0; o < block.vtx[0]->vout.size(); o++) {
166 const CTxOut& vout = block.vtx[0]->vout[o];
167 if (vout.scriptPubKey.size() >= MINIMUM_WITNESS_COMMITMENT &&
168 vout.scriptPubKey[0] == OP_RETURN &&
169 vout.scriptPubKey[1] == 0x24 &&
170 vout.scriptPubKey[2] == 0xaa &&
171 vout.scriptPubKey[3] == 0x21 &&
172 vout.scriptPubKey[4] == 0xa9 &&
173 vout.scriptPubKey[5] == 0xed) {
174 commitpos = o;
175 }
176 }
177 }
178 return commitpos;
179}
