124  }

176 static void http_request_done(struct evhttp_request *req, void *ctx)
177 {
178  HTTPReply *reply = static_cast<HTTPReply*>(ctx);
179 
180  if (req == nullptr) {
181  /* If req is nullptr, it means an error occurred while connecting: the
182  * error code will have been passed to http_error_cb.
183  */
184  reply->status = 0;
185  return;
186  }
187 
188  reply->status = evhttp_request_get_response_code(req);
189 
190  struct evbuffer *buf = evhttp_request_get_input_buffer(req);
191  if (buf)
192  {
193  size_t size = evbuffer_get_length(buf);
194  const char *data = (const char*)evbuffer_pullup(buf, size);
195  if (data)
196  reply->body = std::string(data, size);
197  evbuffer_drain(buf, size);
198  }
199 }

302 static UniValue CallRPC(BaseRequestHandler *rh, const std::string& strMethod, const std::vector<std::string>& args)
303 {
304  std::string host;
305  // In preference order, we choose the following for the port:
306  // 1. -rpcport
307  // 2. port in -rpcconnect (ie following : in ipv4 or ]: in ipv6)
308  // 3. default port for chain
309  int port = BaseParams().RPCPort();
310  SplitHostPort(gArgs.GetArg("-rpcconnect", DEFAULT_RPCCONNECT), port, host);
311  port = gArgs.GetArg("-rpcport", port);
312 
313  // Obtain event base
314  raii_event_base base = obtain_event_base();
315 
316  // Synchronously look up hostname
317  raii_evhttp_connection evcon = obtain_evhttp_connection_base(base.get(), host, port);
318  evhttp_connection_set_timeout(evcon.get(), gArgs.GetArg("-rpcclienttimeout", DEFAULT_HTTP_CLIENT_TIMEOUT));
319 
320  HTTPReply response;
321  raii_evhttp_request req = obtain_evhttp_request(http_request_done, (void*)&response);
322  if (req == nullptr)
323  throw std::runtime_error("create http request failed");
324 #if LIBEVENT_VERSION_NUMBER >= 0x02010300
325  evhttp_request_set_error_cb(req.get(), http_error_cb);
326 #endif
327 
328  // Get credentials
329  std::string strRPCUserColonPass;
330  bool failedToGetAuthCookie = false;
331  if (gArgs.GetArg("-rpcpassword", "") == "") {
332  // Try fall back to cookie-based authentication if no password is provided
333  if (!GetAuthCookie(&strRPCUserColonPass)) {
334  failedToGetAuthCookie = true;
335  }
336  } else {
337  strRPCUserColonPass = gArgs.GetArg("-rpcuser", "") + ":" + gArgs.GetArg("-rpcpassword", "");
338  }
339 
340  struct evkeyvalq* output_headers = evhttp_request_get_output_headers(req.get());
341  assert(output_headers);
342  evhttp_add_header(output_headers, "Host", host.c_str());
343  evhttp_add_header(output_headers, "Connection", "close");
344  evhttp_add_header(output_headers, "Authorization", (std::string("Basic ") + EncodeBase64(strRPCUserColonPass)).c_str());
345 
346  // Attach request data
347  std::string strRequest = rh->PrepareRequest(strMethod, args).write() + "\n";
348  struct evbuffer* output_buffer = evhttp_request_get_output_buffer(req.get());
349  assert(output_buffer);
350  evbuffer_add(output_buffer, strRequest.data(), strRequest.size());
351 
352  // check if we should use a special wallet endpoint
353  std::string endpoint = "/";
354  if (!gArgs.GetArgs("-rpcwallet").empty()) {
355  std::string walletName = gArgs.GetArg("-rpcwallet", "");
356  char *encodedURI = evhttp_uriencode(walletName.c_str(), walletName.size(), false);
357  if (encodedURI) {
358  endpoint = "/wallet/"+ std::string(encodedURI);
359  free(encodedURI);
360  }
361  else {
362  throw CConnectionFailed("uri-encode failed");
363  }
364  }
365  int r = evhttp_make_request(evcon.get(), req.get(), EVHTTP_REQ_POST, endpoint.c_str());
366  req.release(); // ownership moved to evcon in above call
367  if (r != 0) {
368  throw CConnectionFailed("send http request failed");
369  }
370 
371  event_base_dispatch(base.get());
372 
373  if (response.status == 0) {
374  std::string responseErrorMessage;
375  if (response.error != -1) {
376  responseErrorMessage = strprintf(" (error code %d - \"%s\")", response.error, http_errorstring(response.error));
377  }
378  throw CConnectionFailed(strprintf("Could not connect to the server %s:%d%s\n\nMake sure the bitcoind server is running and that you are connecting to the correct RPC port.", host, port, responseErrorMessage));
379  } else if (response.status == HTTP_UNAUTHORIZED) {
380  if (failedToGetAuthCookie) {
381  throw std::runtime_error(strprintf(
382  "Could not locate RPC credentials. No authentication cookie could be found, and RPC password is not set. See -rpcpassword and -stdinrpcpass. Configuration file: (%s)",
383  GetConfigFile(gArgs.GetArg("-conf", BITCOIN_CONF_FILENAME)).string().c_str()));
384  } else {
385  throw std::runtime_error("Authorization failed: Incorrect rpcuser or rpcpassword");
386  }
387  } else if (response.status >= 400 && response.status != HTTP_BAD_REQUEST && response.status != HTTP_NOT_FOUND && response.status != HTTP_INTERNAL_SERVER_ERROR)
388  throw std::runtime_error(strprintf("server returned HTTP error %d", response.status));
389  else if (response.body.empty())
390  throw std::runtime_error("no response from server");
391 
392  // Parse reply
393  UniValue valReply(UniValue::VSTR);
394  if (!valReply.read(response.body))
395  throw std::runtime_error("couldn't parse reply from server");
396  const UniValue reply = rh->ProcessReply(valReply);
397  if (reply.empty())
398  throw std::runtime_error("expected reply to have result, error and id properties");
399 
400  return reply;
401 }

403 static int CommandLineRPC(int argc, char *argv[])
404 {
405  std::string strPrint;
406  int nRet = 0;
407  try {
408  // Skip switches
409  while (argc > 1 && IsSwitchChar(argv[1][0])) {
410  argc--;
411  argv++;
412  }
413  std::string rpcPass;
414  if (gArgs.GetBoolArg("-stdinrpcpass", false)) {
415  if (!std::getline(std::cin, rpcPass)) {
416  throw std::runtime_error("-stdinrpcpass specified but failed to read from standard input");
417  }
418  gArgs.ForceSetArg("-rpcpassword", rpcPass);
419  }
420  std::vector<std::string> args = std::vector<std::string>(&argv[1], &argv[argc]);
421  if (gArgs.GetBoolArg("-stdin", false)) {
422  // Read one arg per line from stdin and append
423  std::string line;
424  while (std::getline(std::cin, line)) {
425  args.push_back(line);
426  }
427  }
428  std::unique_ptr<BaseRequestHandler> rh;
429  std::string method;
430  if (gArgs.GetBoolArg("-getinfo", false)) {
431  rh.reset(new GetinfoRequestHandler());
432  method = "";
433  } else {
434  rh.reset(new DefaultRequestHandler());
435  if (args.size() < 1) {
436  throw std::runtime_error("too few parameters (need at least command)");
437  }
438  method = args[0];
439  args.erase(args.begin()); // Remove trailing method name from arguments vector
440  }
441 
442  // Execute and handle connection failures with -rpcwait
443  const bool fWait = gArgs.GetBoolArg("-rpcwait", false);
444  do {
445  try {
446  const UniValue reply = CallRPC(rh.get(), method, args);
447 
448  // Parse reply
449  const UniValue& result = find_value(reply, "result");
450  const UniValue& error = find_value(reply, "error");
451 
452  if (!error.isNull()) {
453  // Error
454  int code = error["code"].get_int();
455  if (fWait && code == RPC_IN_WARMUP)
456  throw CConnectionFailed("server in warmup");
457  strPrint = "error: " + error.write();
458  nRet = abs(code);
459  if (error.isObject())
460  {
461  UniValue errCode = find_value(error, "code");
462  UniValue errMsg = find_value(error, "message");
463  strPrint = errCode.isNull() ? "" : "error code: "+errCode.getValStr()+"\n";
464 
465  if (errMsg.isStr())
466  strPrint += "error message:\n"+errMsg.get_str();
467 
468  if (errCode.isNum() && errCode.get_int() == RPC_WALLET_NOT_SPECIFIED) {
469  strPrint += "\nTry adding \"-rpcwallet=<filename>\" option to bitcoin-cli command line.";
470  }
471  }
472  } else {
473  // Result
474  if (result.isNull())
475  strPrint = "";
476  else if (result.isStr())
477  strPrint = result.get_str();
478  else
479  strPrint = result.write(2);
480  }
481  // Connection succeeded, no need to retry.
482  break;
483  }
484  catch (const CConnectionFailed&) {
485  if (fWait)
486  MilliSleep(1000);
487  else
488  throw;
489  }
490  } while (fWait);
491  }
492  catch (const std::exception& e) {
493  strPrint = std::string("error: ") + e.what();
494  nRet = EXIT_FAILURE;
495  }
496  catch (...) {
497  PrintExceptionContinue(nullptr, "CommandLineRPC()");
498  throw;
499  }
500 
501  if (strPrint != "") {
502  fprintf((nRet == 0 ? stdout : stderr), "%s\n", strPrint.c_str());
503  }
504  return nRet;
505 }
