42 static void TxToJSON(const CTransaction& tx, const uint256 hashBlock, UniValue& entry)
43 {
44  // Call into TxToUniv() in bitcoin-common to decode the transaction hex.
45  //
46  // Blockchain contextual information (confirmations and blocktime) is not
47  // available to code in bitcoin-common, so we query them here and push the
48  // data into the returned UniValue.
49  TxToUniv(tx, uint256(), entry, true, RPCSerializationFlags());
50 
51  if (!hashBlock.IsNull()) {
52  LOCK(cs_main);
53 
54  entry.pushKV("blockhash", hashBlock.GetHex());
55  CBlockIndex* pindex = LookupBlockIndex(hashBlock);
56  if (pindex) {
57  if (chainActive.Contains(pindex)) {
58  entry.pushKV("confirmations", 1 + chainActive.Height() - pindex->nHeight);
59  entry.pushKV("time", pindex->GetBlockTime());
60  entry.pushKV("blocktime", pindex->GetBlockTime());
61  }
62  else
63  entry.pushKV("confirmations", 0);
64  }
65  }
66 }

363 
364  CMutableTransaction rawTx;
365 
366  if (!locktime.isNull()) {
367  int64_t nLockTime = locktime.get_int64();
368  if (nLockTime < 0 || nLockTime > LOCKTIME_MAX)
369  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, locktime out of range");

371  }
372 
373  bool rbfOptIn = rbf.isTrue();
374

376  const UniValue& input = inputs[idx];
377  const UniValue& o = input.get_obj();
378 
379  uint256 txid = ParseHashO(o, "txid");
380 
381  const UniValue& vout_v = find_value(o, "vout");
382  if (!vout_v.isNum())
383  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, missing vout key");
384  int nOutput = vout_v.get_int();
385  if (nOutput < 0)
386  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, vout must be positive");
387 
388  uint32_t nSequence;
389  if (rbfOptIn) {
390  nSequence = MAX_BIP125_RBF_SEQUENCE; /* CTxIn::SEQUENCE_FINAL - 2 */
391  } else if (rawTx.nLockTime) {
392  nSequence = CTxIn::SEQUENCE_FINAL - 1;
393  } else {
394  nSequence = CTxIn::SEQUENCE_FINAL;
395  }
396 
397  // set the sequence number if passed in the parameters object
398  const UniValue& sequenceObj = find_value(o, "sequence");
399  if (sequenceObj.isNum()) {
400  int64_t seqNr64 = sequenceObj.get_int64();
401  if (seqNr64 < 0 || seqNr64 > CTxIn::SEQUENCE_FINAL) {
402  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, sequence number is out of range");
403  } else {
404  nSequence = (uint32_t)seqNr64;
405  }
406  }
407

409 
410  rawTx.vin.push_back(in);
411  }
412 
413  if (!outputs_is_obj) {
414  // Translate array of key-value pairs into dict
415  UniValue outputs_dict = UniValue(UniValue::VOBJ);
416  for (size_t i = 0; i < outputs.size(); ++i) {
417  const UniValue& output = outputs[i];
418  if (!output.isObject()) {
419  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, key-value pair not an object as expected");
420  }
421  if (output.size() != 1) {
422  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, key-value pair must contain exactly one key");
423  }

425  }
426  outputs = std::move(outputs_dict);
427  }
428 
429  // Duplicate checking
430  std::set<CTxDestination> destinations;
431  bool has_data{false};
432 
433  for (const std::string& name_ : outputs.getKeys()) {
434  if (name_ == "data") {
435  if (has_data) {
436  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, duplicate key: data");
437  }
438  has_data = true;
439  std::vector<unsigned char> data = ParseHexV(outputs[name_].getValStr(), "Data");
440 
441  CTxOut out(0, CScript() << OP_RETURN << data);
442  rawTx.vout.push_back(out);
443  } else {
444  CTxDestination destination = DecodeDestination(name_);
445  if (!IsValidDestination(destination)) {
446  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, std::string("Invalid Bitcoin address: ") + name_);
447  }
448 
449  if (!destinations.insert(destination).second) {
450  throw JSONRPCError(RPC_INVALID_PARAMETER, std::string("Invalid parameter, duplicated address: ") + name_);
451  }
452 
453  CScript scriptPubKey = GetScriptForDestination(destination);
454  CAmount nAmount = AmountFromValue(outputs[name_]);
455 
456  CTxOut out(nAmount, scriptPubKey);
457  rawTx.vout.push_back(out);
458  }
459  }
460 
461  if (!rbf.isNull() && rawTx.vin.size() > 0 && rbfOptIn != SignalsOptInRBF(CTransaction(rawTx))) {
462  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter combination: Sequence number(s) contradict replaceable option");
463  }
464 
465  return rawTx;
466 }

614  throw std::runtime_error(

635  + HelpExampleRpc("decodescript", "\"hexstring\"")

640 
641  UniValue r(UniValue::VOBJ);
642  CScript script;
643  if (request.params[0].get_str().size() > 0){
644  std::vector<unsigned char> scriptData(ParseHexV(request.params[0], "argument"));
645  script = CScript(scriptData.begin(), scriptData.end());

647  // Empty scripts are valid
648  }

651  UniValue type;
652  type = find_value(r, "type");

656  // don't return the address for a P2SH of the P2SH.
657  r.pushKV("p2sh", EncodeDestination(CScriptID(script)));
658  // P2SH and witness programs cannot be wrapped in P2WSH, if this script
659  // is a witness program, don't return addresses for a segwit programs.
660  if (type.get_str() == "pubkey" || type.get_str() == "pubkeyhash" || type.get_str() == "multisig" || type.get_str() == "nonstandard") {

670  }
671  }

675  segwitScr = GetScriptForDestination(WitnessV0KeyHash(Hash160(solutions_data[0].begin(), solutions_data[0].end())));

679  // Scripts that are not fit for P2WPKH are encoded as P2WSH.
680  // Newer segwit program versions should be considered when then become available.
681  segwitScr = GetScriptForDestination(WitnessV0ScriptHash(script));
682  }
683  ScriptPubKeyToUniv(segwitScr, sr, true);
684  sr.pushKV("p2sh-segwit", EncodeDestination(CScriptID(segwitScr)));
685  r.pushKV("segwit", sr);
686  }
687  }
688 
689  return r;

794 {
795  // Fetch previous transactions (inputs):
796  CCoinsView viewDummy;
797  CCoinsViewCache view(&viewDummy);
798  {
799  LOCK2(cs_main, mempool.cs);
800  CCoinsViewCache &viewChain = *pcoinsTip;
801  CCoinsViewMemPool viewMempool(&viewChain, mempool);
802  view.SetBackend(viewMempool); // temporarily switch cache backend to db+mempool view
803 
804  for (const CTxIn& txin : mtx.vin) {
805  view.AccessCoin(txin.prevout); // Load entries from viewChain into view; can fail.
806  }
807 
808  view.SetBackend(viewDummy); // switch back to avoid locking mempool for too long
809  }
810 
811  // Add previous txouts given in the RPC call:
812  if (!prevTxsUnival.isNull()) {
813  UniValue prevTxs = prevTxsUnival.get_array();
814  for (unsigned int idx = 0; idx < prevTxs.size(); ++idx) {
815  const UniValue& p = prevTxs[idx];
816  if (!p.isObject()) {
817  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "expected object with {\"txid'\",\"vout\",\"scriptPubKey\"}");
818  }
819 
820  UniValue prevOut = p.get_obj();
821 
822  RPCTypeCheckObj(prevOut,
823  {
824  {"txid", UniValueType(UniValue::VSTR)},
825  {"vout", UniValueType(UniValue::VNUM)},
826  {"scriptPubKey", UniValueType(UniValue::VSTR)},
827  });
828 
829  uint256 txid = ParseHashO(prevOut, "txid");
830 
831  int nOut = find_value(prevOut, "vout").get_int();
832  if (nOut < 0) {
833  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "vout must be positive");
834  }
835 
836  COutPoint out(txid, nOut);
837  std::vector<unsigned char> pkData(ParseHexO(prevOut, "scriptPubKey"));
838  CScript scriptPubKey(pkData.begin(), pkData.end());
839 
840  {
841  const Coin& coin = view.AccessCoin(out);
842  if (!coin.IsSpent() && coin.out.scriptPubKey != scriptPubKey) {
843  std::string err("Previous output scriptPubKey mismatch:\n");
844  err = err + ScriptToAsmStr(coin.out.scriptPubKey) + "\nvs:\n"+
845  ScriptToAsmStr(scriptPubKey);
846  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, err);
847  }
848  Coin newcoin;
849  newcoin.out.scriptPubKey = scriptPubKey;
850  newcoin.out.nValue = MAX_MONEY;
851  if (prevOut.exists("amount")) {
852  newcoin.out.nValue = AmountFromValue(find_value(prevOut, "amount"));
853  }
854  newcoin.nHeight = 1;
855  view.AddCoin(out, std::move(newcoin), true);
856  }
857 
858  // if redeemScript and private keys were given, add redeemScript to the keystore so it can be signed
859  if (is_temp_keystore && (scriptPubKey.IsPayToScriptHash() || scriptPubKey.IsPayToWitnessScriptHash())) {
860  RPCTypeCheckObj(prevOut,
861  {
862  {"redeemScript", UniValueType(UniValue::VSTR)},
863  {"witnessScript", UniValueType(UniValue::VSTR)},
864  }, true);
865  UniValue rs = find_value(prevOut, "redeemScript");
866  if (!rs.isNull()) {
867  std::vector<unsigned char> rsData(ParseHexV(rs, "redeemScript"));
868  CScript redeemScript(rsData.begin(), rsData.end());
869  keystore->AddCScript(redeemScript);
870  // Automatically also add the P2WSH wrapped version of the script (to deal with P2SH-P2WSH).
871  // This is only for compatibility, it is encouraged to use the explicit witnessScript field instead.
872  keystore->AddCScript(GetScriptForWitness(redeemScript));
873  }
874  UniValue ws = find_value(prevOut, "witnessScript");
875  if (!ws.isNull()) {
876  std::vector<unsigned char> wsData(ParseHexV(ws, "witnessScript"));
877  CScript witnessScript(wsData.begin(), wsData.end());
878  keystore->AddCScript(witnessScript);
879  // Automatically also add the P2WSH wrapped version of the script (to deal with P2SH-P2WSH).
880  keystore->AddCScript(GetScriptForWitness(witnessScript));
881  }
882  }
883  }
884  }
885 
886  int nHashType = ParseSighashString(hashType);
887 
888  bool fHashSingle = ((nHashType & ~SIGHASH_ANYONECANPAY) == SIGHASH_SINGLE);
889 
890  // Script verification errors
891  UniValue vErrors(UniValue::VARR);
892 
893  // Use CTransaction for the constant parts of the
894  // transaction to avoid rehashing.
895  const CTransaction txConst(mtx);
896  // Sign what we can:
897  for (unsigned int i = 0; i < mtx.vin.size(); i++) {
898  CTxIn& txin = mtx.vin[i];
899  const Coin& coin = view.AccessCoin(txin.prevout);
900  if (coin.IsSpent()) {
901  TxInErrorToJSON(txin, vErrors, "Input not found or already spent");
902  continue;
903  }
904  const CScript& prevPubKey = coin.out.scriptPubKey;
905  const CAmount& amount = coin.out.nValue;
906 
907  SignatureData sigdata = DataFromTransaction(mtx, i, coin.out);
908  // Only sign SIGHASH_SINGLE if there's a corresponding output:
909  if (!fHashSingle || (i < mtx.vout.size())) {
910  ProduceSignature(*keystore, MutableTransactionSignatureCreator(&mtx, i, amount, nHashType), prevPubKey, sigdata);
911  }
912 
913  UpdateInput(txin, sigdata);
914 
915  // amount must be specified for valid segwit signature
916  if (amount == MAX_MONEY && !txin.scriptWitness.IsNull()) {
917  throw JSONRPCError(RPC_TYPE_ERROR, strprintf("Missing amount for %s", coin.out.ToString()));
918  }
919 
920  ScriptError serror = SCRIPT_ERR_OK;
921  if (!VerifyScript(txin.scriptSig, prevPubKey, &txin.scriptWitness, STANDARD_SCRIPT_VERIFY_FLAGS, TransactionSignatureChecker(&txConst, i, amount), &serror)) {
922  if (serror == SCRIPT_ERR_INVALID_STACK_OPERATION) {
923  // Unable to sign input and verification failed (possible attempt to partially sign).
924  TxInErrorToJSON(txin, vErrors, "Unable to sign input, invalid stack size (possibly missing key)");
925  } else {
926  TxInErrorToJSON(txin, vErrors, ScriptErrorString(serror));
927  }
928  }
929  }
930  bool fComplete = vErrors.empty();
931 
932  UniValue result(UniValue::VOBJ);
933  result.pushKV("hex", EncodeHexTx(CTransaction(mtx)));
934  result.pushKV("complete", fComplete);
935  if (!vErrors.empty()) {
936  result.pushKV("errors", vErrors);
937  }
938
