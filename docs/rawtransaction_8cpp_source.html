234  uint256 hash(uint256S(txid.get_str()));
235  if (setTxids.count(hash))
236  throw JSONRPCError(RPC_INVALID_PARAMETER, std::string("Invalid parameter, duplicated txid: ")+txid.get_str());
237  setTxids.insert(hash);

239  }

242  uint256 hashBlock;
243  if (!request.params[1].isNull()) {
244  LOCK(cs_main);
245  hashBlock = uint256S(request.params[1].get_str());

251  LOCK(cs_main);
252

257  pblockindex = chainActive[coin.nHeight];

259  }
260  }
261  }
262 
263

266  g_txindex->BlockUntilSyncedToCurrentChain();

268 
269  LOCK(cs_main);
270

292

573 {

575  throw std::runtime_error(

580  "\nResult:\n"
581  "{\n"
582  " \"asm\":\"asm\", (string) Script public key\n"
583  " \"hex\":\"hex\", (string) hex encoded public key\n"
584  " \"type\":\"type\", (string) The output type\n"
585  " \"reqSigs\": n, (numeric) The required signatures\n"
586  " \"addresses\": [ (json array of string)\n"
587  " \"address\" (string) bitcoin address\n"
588  " ,...\n"
589  " ],\n"
590  " \"p2sh\",\"address\" (string) address of P2SH script wrapping this redeem script (not returned if the script is already a P2SH).\n"
591  "}\n"
592  "\nExamples:\n"
593  + HelpExampleCli("decodescript", "\"hexstring\"")
594  + HelpExampleRpc("decodescript", "\"hexstring\"")
595  );
596 
597  RPCTypeCheck(request.params, {UniValue::VSTR});
598 
599  UniValue r(UniValue::VOBJ);
600  CScript script;
601  if (request.params[0].get_str().size() > 0){
602  std::vector<unsigned char> scriptData(ParseHexV(request.params[0], "argument"));
603  script = CScript(scriptData.begin(), scriptData.end());
604  } else {
605  // Empty scripts are valid
606  }
607  ScriptPubKeyToUniv(script, r, false);
608 
609  UniValue type;
610  type = find_value(r, "type");
611 
612  if (type.isStr() && type.get_str() != "scripthash") {
613  // P2SH cannot be wrapped in a P2SH. If this script is already a P2SH,
614  // don't return the address for a P2SH of the P2SH.
615  r.pushKV("p2sh", EncodeDestination(CScriptID(script)));
616  // P2SH and witness programs cannot be wrapped in P2WSH, if this script
617  // is a witness program, don't return addresses for a segwit programs.
618  if (type.get_str() == "pubkey" || type.get_str() == "pubkeyhash" || type.get_str() == "multisig" || type.get_str() == "nonstandard") {
619  txnouttype which_type;
620  std::vector<std::vector<unsigned char>> solutions_data;
621  Solver(script, which_type, solutions_data);
622  // Uncompressed pubkeys cannot be used with segwit checksigs.
623  // If the script contains an uncompressed pubkey, skip encoding of a segwit program.
624  if ((which_type == TX_PUBKEY) || (which_type == TX_MULTISIG)) {
625  for (const auto& solution : solutions_data) {
626  if ((solution.size() != 1) && !CPubKey(solution).IsCompressed()) {
627  return r;
628  }
629  }
630  }
631  UniValue sr(UniValue::VOBJ);
632  CScript segwitScr;
633  if (which_type == TX_PUBKEY) {
634  segwitScr = GetScriptForDestination(WitnessV0KeyHash(Hash160(solutions_data[0].begin(), solutions_data[0].end())));
635  } else if (which_type == TX_PUBKEYHASH) {
636  segwitScr = GetScriptForDestination(WitnessV0KeyHash(solutions_data[0]));
637  } else {
638  // Scripts that are not fit for P2WPKH are encoded as P2WSH.
639  // Newer segwit program versions should be considered when then become available.
640  uint256 scriptHash;
641  CSHA256().Write(script.data(), script.size()).Finalize(scriptHash.begin());
642  segwitScr = GetScriptForDestination(WitnessV0ScriptHash(scriptHash));
643  }
644  ScriptPubKeyToUniv(segwitScr, sr, true);
645  sr.pushKV("p2sh-segwit", EncodeDestination(CScriptID(segwitScr)));
646  r.pushKV("segwit", sr);
647  }
648  }
649 
650  return r;
651 }

758 {

760  CCoinsView viewDummy;
761  CCoinsViewCache view(&viewDummy);
762  {
763  LOCK2(cs_main, mempool.cs);
764  CCoinsViewCache &viewChain = *pcoinsTip;
765  CCoinsViewMemPool viewMempool(&viewChain, mempool);
766  view.SetBackend(viewMempool); // temporarily switch cache backend to db+mempool view
767 
768  for (const CTxIn& txin : mtx.vin) {
769  view.AccessCoin(txin.prevout); // Load entries from viewChain into view; can fail.
770  }
771 
772  view.SetBackend(viewDummy); // switch back to avoid locking mempool for too long
773  }
774 
775  // Add previous txouts given in the RPC call:
776  if (!prevTxsUnival.isNull()) {
777  UniValue prevTxs = prevTxsUnival.get_array();
778  for (unsigned int idx = 0; idx < prevTxs.size(); ++idx) {
779  const UniValue& p = prevTxs[idx];
780  if (!p.isObject()) {
781  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "expected object with {\"txid'\",\"vout\",\"scriptPubKey\"}");
782  }
783 
784  UniValue prevOut = p.get_obj();
785 
786  RPCTypeCheckObj(prevOut,
787  {
788  {"txid", UniValueType(UniValue::VSTR)},
789  {"vout", UniValueType(UniValue::VNUM)},
790  {"scriptPubKey", UniValueType(UniValue::VSTR)},
791  });
792 
793  uint256 txid = ParseHashO(prevOut, "txid");
794 
795  int nOut = find_value(prevOut, "vout").get_int();
796  if (nOut < 0) {
797  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "vout must be positive");
798  }
799 
800  COutPoint out(txid, nOut);
801  std::vector<unsigned char> pkData(ParseHexO(prevOut, "scriptPubKey"));
802  CScript scriptPubKey(pkData.begin(), pkData.end());
803 
804  {
805  const Coin& coin = view.AccessCoin(out);
806  if (!coin.IsSpent() && coin.out.scriptPubKey != scriptPubKey) {
807  std::string err("Previous output scriptPubKey mismatch:\n");
808  err = err + ScriptToAsmStr(coin.out.scriptPubKey) + "\nvs:\n"+
809  ScriptToAsmStr(scriptPubKey);
810  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, err);
811  }
812  Coin newcoin;
813  newcoin.out.scriptPubKey = scriptPubKey;
814  newcoin.out.nValue = 0;
815  if (prevOut.exists("amount")) {
816  newcoin.out.nValue = AmountFromValue(find_value(prevOut, "amount"));
817  }
818  newcoin.nHeight = 1;
819  view.AddCoin(out, std::move(newcoin), true);
820  }
821 
822  // if redeemScript given and not using the local wallet (private keys
823  // given), add redeemScript to the keystore so it can be signed:
824  if (is_temp_keystore && (scriptPubKey.IsPayToScriptHash() || scriptPubKey.IsPayToWitnessScriptHash())) {
825  RPCTypeCheckObj(prevOut,
826  {
827  {"redeemScript", UniValueType(UniValue::VSTR)},
828  });
829  UniValue v = find_value(prevOut, "redeemScript");
830  if (!v.isNull()) {
831  std::vector<unsigned char> rsData(ParseHexV(v, "redeemScript"));
832  CScript redeemScript(rsData.begin(), rsData.end());
833  keystore->AddCScript(redeemScript);
834  // Automatically also add the P2WSH wrapped version of the script (to deal with P2SH-P2WSH).
835  keystore->AddCScript(GetScriptForWitness(redeemScript));
836  }
837  }
838  }
839  }
840 
841  int nHashType = SIGHASH_ALL;
842  if (!hashType.isNull()) {
843  static std::map<std::string, int> mapSigHashValues = {
844  {std::string("ALL"), int(SIGHASH_ALL)},
845  {std::string("ALL|ANYONECANPAY"), int(SIGHASH_ALL|SIGHASH_ANYONECANPAY)},
846  {std::string("NONE"), int(SIGHASH_NONE)},
847  {std::string("NONE|ANYONECANPAY"), int(SIGHASH_NONE|SIGHASH_ANYONECANPAY)},
848  {std::string("SINGLE"), int(SIGHASH_SINGLE)},
849  {std::string("SINGLE|ANYONECANPAY"), int(SIGHASH_SINGLE|SIGHASH_ANYONECANPAY)},
850  };
851  std::string strHashType = hashType.get_str();
852  if (mapSigHashValues.count(strHashType)) {
853  nHashType = mapSigHashValues[strHashType];
854  } else {
855  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid sighash param");
856  }
857  }
858 
859  bool fHashSingle = ((nHashType & ~SIGHASH_ANYONECANPAY) == SIGHASH_SINGLE);
860 
861  // Script verification errors
862  UniValue vErrors(UniValue::VARR);
863 
864  // Use CTransaction for the constant parts of the
865  // transaction to avoid rehashing.
866  const CTransaction txConst(mtx);
867  // Sign what we can:
868  for (unsigned int i = 0; i < mtx.vin.size(); i++) {
869  CTxIn& txin = mtx.vin[i];
870  const Coin& coin = view.AccessCoin(txin.prevout);
871  if (coin.IsSpent()) {
872  TxInErrorToJSON(txin, vErrors, "Input not found or already spent");
873  continue;
874  }
875  const CScript& prevPubKey = coin.out.scriptPubKey;
876  const CAmount& amount = coin.out.nValue;
877 
878  SignatureData sigdata;
879  // Only sign SIGHASH_SINGLE if there's a corresponding output:
880  if (!fHashSingle || (i < mtx.vout.size())) {
881  ProduceSignature(*keystore, MutableTransactionSignatureCreator(&mtx, i, amount, nHashType), prevPubKey, sigdata);
882  }
883  sigdata = CombineSignatures(prevPubKey, TransactionSignatureChecker(&txConst, i, amount), sigdata, DataFromTransaction(mtx, i));
884 
885  UpdateInput(txin, sigdata);
886 
887  ScriptError serror = SCRIPT_ERR_OK;
888  if (!VerifyScript(txin.scriptSig, prevPubKey, &txin.scriptWitness, STANDARD_SCRIPT_VERIFY_FLAGS, TransactionSignatureChecker(&txConst, i, amount), &serror)) {
889  if (serror == SCRIPT_ERR_INVALID_STACK_OPERATION) {
890  // Unable to sign input and verification failed (possible attempt to partially sign).
891  TxInErrorToJSON(txin, vErrors, "Unable to sign input, invalid stack size (possibly missing key)");
892  } else {
893  TxInErrorToJSON(txin, vErrors, ScriptErrorString(serror));
894  }
895  }
896  }
897  bool fComplete = vErrors.empty();
898 
899  UniValue result(UniValue::VOBJ);
900  result.pushKV("hex", EncodeHexTx(mtx));
901  result.pushKV("complete", fComplete);
902  if (!vErrors.empty()) {
903  result.pushKV("errors", vErrors);
904  }
905 
906  return result;
907 }
