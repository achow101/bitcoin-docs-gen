327 int have_flag(int argc, char** argv, char *flag) {
328  char** argm = argv + argc;
329  argv++;
330  if (argv == argm) {
331  return 1;
332  }
333  while (argv != NULL && argv != argm) {
334  if (strcmp(*argv, flag) == 0) {
