67 void WalletModel::pollBalanceChanged()
68 {
69  // Try to get balances and return early if locks can't be acquired. This
70  // avoids the GUI from getting stuck on periodical polls if the core is
71  // holding the locks for a longer time - for example, during a wallet
72  // rescan.
73  interfaces::WalletBalances new_balances;
74  int numBlocks = -1;
75  if (!m_wallet->tryGetBalances(new_balances, numBlocks)) {
76  return;
77  }
78 
79  if(fForceCheckBalanceChanged || m_node.getNumBlocks() != cachedNumBlocks)
80  {
81  fForceCheckBalanceChanged = false;
82 
83  // Balance and number of transactions might have changed
84  cachedNumBlocks = m_node.getNumBlocks();
85 
86  checkBalanceChanged(new_balances);
87  if(transactionTableModel)
88  transactionTableModel->updateConfirmations();
89  }
90 }

124 WalletModel::SendCoinsReturn WalletModel::prepareTransaction(WalletModelTransaction &transaction, const CCoinControl& coinControl)
125 {
126  CAmount total = 0;
127  bool fSubtractFeeFromAmount = false;
128  QList<SendCoinsRecipient> recipients = transaction.getRecipients();
129  std::vector<CRecipient> vecSend;
130 
131  if(recipients.empty())
132  {
133  return OK;
134  }
135 
136  QSet<QString> setAddress; // Used to detect duplicates
137  int nAddresses = 0;
138 
139  // Pre-check input data for validity
140  for (const SendCoinsRecipient &rcp : recipients)
141  {
142  if (rcp.fSubtractFeeFromAmount)
143  fSubtractFeeFromAmount = true;
144 
145  if (rcp.paymentRequest.IsInitialized())
146  { // PaymentRequest...
147  CAmount subtotal = 0;
148  const payments::PaymentDetails& details = rcp.paymentRequest.getDetails();
149  for (int i = 0; i < details.outputs_size(); i++)
150  {
151  const payments::Output& out = details.outputs(i);
152  if (out.amount() <= 0) continue;
153  subtotal += out.amount();
154  const unsigned char* scriptStr = (const unsigned char*)out.script().data();
155  CScript scriptPubKey(scriptStr, scriptStr+out.script().size());
156  CAmount nAmount = out.amount();
157  CRecipient recipient = {scriptPubKey, nAmount, rcp.fSubtractFeeFromAmount};
158  vecSend.push_back(recipient);
159  }
160  if (subtotal <= 0)
161  {
162  return InvalidAmount;
163  }
164  total += subtotal;
165  }
166  else
167  { // User-entered bitcoin address / amount:
168  if(!validateAddress(rcp.address))
169  {
170  return InvalidAddress;
171  }
172  if(rcp.amount <= 0)
173  {
174  return InvalidAmount;
175  }
176  setAddress.insert(rcp.address);
177  ++nAddresses;
178 
179  CScript scriptPubKey = GetScriptForDestination(DecodeDestination(rcp.address.toStdString()));
180  CRecipient recipient = {scriptPubKey, rcp.amount, rcp.fSubtractFeeFromAmount};
181  vecSend.push_back(recipient);
182 
183  total += rcp.amount;
184  }
185  }
186  if(setAddress.size() != nAddresses)
187  {
188  return DuplicateAddress;
189  }
190 
191  CAmount nBalance = m_wallet->getAvailableBalance(coinControl);
192 
193  if(total > nBalance)
194  {
195  return AmountExceedsBalance;
196  }
197 
198  {
199  CAmount nFeeRequired = 0;
200  int nChangePosRet = -1;
201  std::string strFailReason;
202 
203  auto& newTx = transaction.getWtx();
204  newTx = m_wallet->createTransaction(vecSend, coinControl, true /* sign */, nChangePosRet, nFeeRequired, strFailReason);
205  transaction.setTransactionFee(nFeeRequired);
206  if (fSubtractFeeFromAmount && newTx)
207  transaction.reassignAmounts(nChangePosRet);
208 
209  if(!newTx)
210  {
211  if(!fSubtractFeeFromAmount && (total + nFeeRequired) > nBalance)
212  {
213  return SendCoinsReturn(AmountWithFeeExceedsBalance);
214  }
215  Q_EMIT message(tr("Send Coins"), QString::fromStdString(strFailReason),
216  CClientUIInterface::MSG_ERROR);
217  return TransactionCreationFailed;
218  }
219 
220  // reject absurdly high fee. (This can never happen because the
221  // wallet caps the fee at maxTxFee. This merely serves as a
222  // belt-and-suspenders check)
223  if (nFeeRequired > m_node.getMaxTxFee())
224  return AbsurdFee;
225  }
226 
227  return SendCoinsReturn(OK);
228 }

230 WalletModel::SendCoinsReturn WalletModel::sendCoins(WalletModelTransaction &transaction)
231 {
232  QByteArray transaction_array; /* store serialized transaction */
233 
234  {
235  std::vector<std::pair<std::string, std::string>> vOrderForm;
236  for (const SendCoinsRecipient &rcp : transaction.getRecipients())
237  {
238  if (rcp.paymentRequest.IsInitialized())
239  {
240  // Make sure any payment requests involved are still valid.
241  if (PaymentServer::verifyExpired(rcp.paymentRequest.getDetails())) {
242  return PaymentRequestExpired;
243  }
244 
245  // Store PaymentRequests in wtx.vOrderForm in wallet.
246  std::string value;
247  rcp.paymentRequest.SerializeToString(&value);
248  vOrderForm.emplace_back("PaymentRequest", std::move(value));
249  }
250  else if (!rcp.message.isEmpty()) // Message from normal bitcoin:URI (bitcoin:123...?message=example)
251  vOrderForm.emplace_back("Message", rcp.message.toStdString());
252  }
253 
254  auto& newTx = transaction.getWtx();
255  std::string rejectReason;
256  if (!newTx->commit({} /* mapValue */, std::move(vOrderForm), rejectReason))
257  return SendCoinsReturn(TransactionCommitFailed, QString::fromStdString(rejectReason));
258 
259  CDataStream ssTx(SER_NETWORK, PROTOCOL_VERSION);
260  ssTx << newTx->get();
261  transaction_array.append(&(ssTx[0]), ssTx.size());
262  }
263 
264  // Add addresses / update labels that we've sent to the address book,
265  // and emit coinsSent signal for each recipient
266  for (const SendCoinsRecipient &rcp : transaction.getRecipients())
267  {
268  // Don't touch the address book when we have a payment request
269  if (!rcp.paymentRequest.IsInitialized())
270  {
271  std::string strAddress = rcp.address.toStdString();
272  CTxDestination dest = DecodeDestination(strAddress);
273  std::string strLabel = rcp.label.toStdString();
274  {
275  // Check if we have a new address or an updated label
276  std::string name;
277  if (!m_wallet->getAddress(
278  dest, &name, /* is_mine= */ nullptr, /* purpose= */ nullptr))
279  {
280  m_wallet->setAddressBook(dest, strLabel, "send");
281  }
282  else if (name != strLabel)
283  {
284  m_wallet->setAddressBook(dest, strLabel, ""); // "" means don't change purpose
285  }
286  }
287  }
288  Q_EMIT coinsSent(this, rcp, transaction_array);
289  }
290 
291  checkBalanceChanged(m_wallet->getBalances()); // update balance immediately, otherwise there could be a short noticeable delay until pollBalanceChanged hits
292 
293  return SendCoinsReturn(OK);
294 }

495 bool WalletModel::bumpFee(uint256 hash, uint256& new_hash)
496 {

499  std::vector<std::string> errors;

501  CAmount new_fee;

503  if (!m_wallet->createBumpTransaction(hash, coin_control, 0 /* totalFee */, errors, old_fee, new_fee, mtx)) {
504  QMessageBox::critical(0, tr("Fee bump error"), tr("Increasing transaction fee failed") + "<br />(" +
505  (errors.size() ? QString::fromStdString(errors[0]) : "") +")");

521  questionString.append("</td></tr><tr><td>");
522  questionString.append(tr("New fee:"));
523  questionString.append("</td><td>");
524  questionString.append(BitcoinUnits::formatHtmlWithUnit(getOptionsModel()->getDisplayUnit(), new_fee));
525  questionString.append("</td></tr></table>");
526  SendConfirmationDialog confirmationDialog(tr("Confirm fee bump"), questionString);
527  confirmationDialog.exec();
528  QMessageBox::StandardButton retval = static_cast<QMessageBox::StandardButton>(confirmationDialog.result());
529 
530  // cancel sign&broadcast if user doesn't want to bump the fee
531  if (retval != QMessageBox::Yes) {
532  return false;
533  }
534 
535  WalletModel::UnlockContext ctx(requestUnlock());
536  if(!ctx.isValid())
537  {
538  return false;
539  }
540 
541  // sign bumped transaction
542  if (!m_wallet->signBumpTransaction(mtx)) {
543  QMessageBox::critical(0, tr("Fee bump error"), tr("Can't sign transaction."));
544  return false;
545  }
546  // commit the bumped transaction
547  if(!m_wallet->commitBumpTransaction(hash, std::move(mtx), errors, new_hash)) {
548  QMessageBox::critical(0, tr("Fee bump error"), tr("Could not commit transaction") + "<br />(" +
549  QString::fromStdString(errors[0])+")");
550  return false;
551  }
552  return true;
553 }
