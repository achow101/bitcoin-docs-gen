407 void CoinControlDialog::updateLabels(WalletModel *model, QDialog* dialog)
408 {
409  if (!model)
410  return;
411 
412  // nPayAmount
413  CAmount nPayAmount = 0;
414  bool fDust = false;
415  CMutableTransaction txDummy;
416  for (const CAmount &amount : CoinControlDialog::payAmounts)
417  {
418  nPayAmount += amount;
419 
420  if (amount > 0)
421  {
422  CTxOut txout(amount, static_cast<CScript>(std::vector<unsigned char>(24, 0)));
423  txDummy.vout.push_back(txout);
424  fDust |= IsDust(txout, model->node().getDustRelayFee());
425  }
426  }
427 
428  CAmount nAmount = 0;
429  CAmount nPayFee = 0;
430  CAmount nAfterFee = 0;
431  CAmount nChange = 0;
432  unsigned int nBytes = 0;
433  unsigned int nBytesInputs = 0;
434  unsigned int nQuantity = 0;
435  bool fWitness = false;
436 
437  std::vector<COutPoint> vCoinControl;
438  coinControl()->ListSelected(vCoinControl);
439 
440  size_t i = 0;
441  for (const auto& out : model->wallet().getCoins(vCoinControl)) {
442  if (out.depth_in_main_chain < 0) continue;
443 
444  // unselect already spent, very unlikely scenario, this could happen
445  // when selected are spent elsewhere, like rpc or another computer
446  const COutPoint& outpt = vCoinControl[i++];
447  if (out.is_spent)
448  {
449  coinControl()->UnSelect(outpt);
450  continue;
451  }
452 
453  // Quantity
454  nQuantity++;
455 
456  // Amount
457  nAmount += out.txout.nValue;
458 
459  // Bytes
460  CTxDestination address;
461  int witnessversion = 0;
462  std::vector<unsigned char> witnessprogram;
463  if (out.txout.scriptPubKey.IsWitnessProgram(witnessversion, witnessprogram))
464  {
465  nBytesInputs += (32 + 4 + 1 + (107 / WITNESS_SCALE_FACTOR) + 4);
466  fWitness = true;
467  }
468  else if(ExtractDestination(out.txout.scriptPubKey, address))
469  {
470  CPubKey pubkey;
471  CKeyID *keyid = boost::get<CKeyID>(&address);
472  if (keyid && model->wallet().getPubKey(*keyid, pubkey))
473  {
474  nBytesInputs += (pubkey.IsCompressed() ? 148 : 180);
475  }
476  else
477  nBytesInputs += 148; // in all error cases, simply assume 148 here
478  }
479  else nBytesInputs += 148;
480  }
481 
482  // calculation
483  if (nQuantity > 0)
484  {
485  // Bytes
486  nBytes = nBytesInputs + ((CoinControlDialog::payAmounts.size() > 0 ? CoinControlDialog::payAmounts.size() + 1 : 2) * 34) + 10; // always assume +1 output for change here
487  if (fWitness)
488  {
489  // there is some fudging in these numbers related to the actual virtual transaction size calculation that will keep this estimate from being exact.
490  // usually, the result will be an overestimate within a couple of satoshis so that the confirmation dialog ends up displaying a slightly smaller fee.
491  // also, the witness stack size value is a variable sized integer. usually, the number of stack items will be well under the single byte var int limit.
492  nBytes += 2; // account for the serialized marker and flag bytes
493  nBytes += nQuantity; // account for the witness byte that holds the number of stack items for each input.
494  }
495 
496  // in the subtract fee from amount case, we can tell if zero change already and subtract the bytes, so that fee calculation afterwards is accurate
497  if (CoinControlDialog::fSubtractFeeFromAmount)
498  if (nAmount - nPayAmount == 0)
499  nBytes -= 34;
500 
501  // Fee
502  nPayFee = model->wallet().getMinimumFee(nBytes, *coinControl(), nullptr /* returned_target */, nullptr /* reason */);
503 
504  if (nPayAmount > 0)
505  {
506  nChange = nAmount - nPayAmount;
507  if (!CoinControlDialog::fSubtractFeeFromAmount)
508  nChange -= nPayFee;
509 
510  // Never create dust outputs; if we would, just add the dust to the fee.
511  if (nChange > 0 && nChange < MIN_CHANGE)
512  {
513  CTxOut txout(nChange, static_cast<CScript>(std::vector<unsigned char>(24, 0)));
514  if (IsDust(txout, model->node().getDustRelayFee()))
515  {
516  nPayFee += nChange;
517  nChange = 0;
518  if (CoinControlDialog::fSubtractFeeFromAmount)
519  nBytes -= 34; // we didn't detect lack of change above
520  }
521  }
522 
523  if (nChange == 0 && !CoinControlDialog::fSubtractFeeFromAmount)
524  nBytes -= 34;
525  }
526 
527  // after fee
528  nAfterFee = std::max<CAmount>(nAmount - nPayFee, 0);
529  }
530 
531  // actually update labels
532  int nDisplayUnit = BitcoinUnits::BTC;
533  if (model && model->getOptionsModel())
534  nDisplayUnit = model->getOptionsModel()->getDisplayUnit();
535 
536  QLabel *l1 = dialog->findChild<QLabel *>("labelCoinControlQuantity");
537  QLabel *l2 = dialog->findChild<QLabel *>("labelCoinControlAmount");
538  QLabel *l3 = dialog->findChild<QLabel *>("labelCoinControlFee");
539  QLabel *l4 = dialog->findChild<QLabel *>("labelCoinControlAfterFee");
540  QLabel *l5 = dialog->findChild<QLabel *>("labelCoinControlBytes");
541  QLabel *l7 = dialog->findChild<QLabel *>("labelCoinControlLowOutput");
542  QLabel *l8 = dialog->findChild<QLabel *>("labelCoinControlChange");
543 
544  // enable/disable "dust" and "change"
545  dialog->findChild<QLabel *>("labelCoinControlLowOutputText")->setEnabled(nPayAmount > 0);
546  dialog->findChild<QLabel *>("labelCoinControlLowOutput") ->setEnabled(nPayAmount > 0);
547  dialog->findChild<QLabel *>("labelCoinControlChangeText") ->setEnabled(nPayAmount > 0);
548  dialog->findChild<QLabel *>("labelCoinControlChange") ->setEnabled(nPayAmount > 0);
549 
550  // stats
551  l1->setText(QString::number(nQuantity)); // Quantity
552  l2->setText(BitcoinUnits::formatWithUnit(nDisplayUnit, nAmount)); // Amount
553  l3->setText(BitcoinUnits::formatWithUnit(nDisplayUnit, nPayFee)); // Fee
554  l4->setText(BitcoinUnits::formatWithUnit(nDisplayUnit, nAfterFee)); // After Fee
555  l5->setText(((nBytes > 0) ? ASYMP_UTF8 : "") + QString::number(nBytes)); // Bytes
556  l7->setText(fDust ? tr("yes") : tr("no")); // Dust
557  l8->setText(BitcoinUnits::formatWithUnit(nDisplayUnit, nChange)); // Change
558  if (nPayFee > 0)
559  {
560  l3->setText(ASYMP_UTF8 + l3->text());
561  l4->setText(ASYMP_UTF8 + l4->text());
562  if (nChange > 0 && !CoinControlDialog::fSubtractFeeFromAmount)
563  l8->setText(ASYMP_UTF8 + l8->text());
564  }
565 
566  // turn label red when dust
567  l7->setStyleSheet((fDust) ? "color:red;" : "");
568 
569  // tool tips
570  QString toolTipDust = tr("This label turns red if any recipient receives an amount smaller than the current dust threshold.");

622  if (treeMode)
623  {
624  // wallet address
625  ui->treeWidget->addTopLevelItem(itemWalletAddress);
626 
627  itemWalletAddress->setFlags(flgTristate);
628  itemWalletAddress->setCheckState(COLUMN_CHECKBOX, Qt::Unchecked);
629 
630  // label
631  itemWalletAddress->setText(COLUMN_LABEL, sWalletLabel);
632 
633  // address
634  itemWalletAddress->setText(COLUMN_ADDRESS, sWalletAddress);
635  }
636 
637  CAmount nSum = 0;
638  int nChildren = 0;
639  for (const auto& outpair : coins.second) {
640  const COutPoint& output = std::get<0>(outpair);
641  const interfaces::WalletTxOut& out = std::get<1>(outpair);
642  nSum += out.txout.nValue;

644 
645  CCoinControlWidgetItem *itemOutput;
646  if (treeMode) itemOutput = new CCoinControlWidgetItem(itemWalletAddress);

649  itemOutput->setCheckState(COLUMN_CHECKBOX,Qt::Unchecked);
650 
651  // address
652  CTxDestination outputAddress;
653  QString sAddress = "";
654  if(ExtractDestination(out.txout.scriptPubKey, outputAddress))
655  {
656  sAddress = QString::fromStdString(EncodeDestination(outputAddress));
657 
658  // if listMode or change => show bitcoin address. In tree mode, address is not shown again for direct wallet address outputs
659  if (!treeMode || (!(sAddress == sWalletAddress)))
660  itemOutput->setText(COLUMN_ADDRESS, sAddress);
661  }
662 
663  // label
664  if (!(sAddress == sWalletAddress)) // change
665  {
666  // tooltip from where the change comes from
667  itemOutput->setToolTip(COLUMN_LABEL, tr("change from %1 (%2)").arg(sWalletLabel).arg(sWalletAddress));
668  itemOutput->setText(COLUMN_LABEL, tr("(change)"));

671  {
672  QString sLabel = model->getAddressTableModel()->labelForAddress(sAddress);
673  if (sLabel.isEmpty())

675  itemOutput->setText(COLUMN_LABEL, sLabel);
676  }
677 
678  // amount
679  itemOutput->setText(COLUMN_AMOUNT, BitcoinUnits::format(nDisplayUnit, out.txout.nValue));

692 
693  // vout index
694  itemOutput->setText(COLUMN_VOUT_INDEX, QString::number(output.n));
695 
696  // disable locked coins
697  if (model->wallet().isLockedCoin(output))
698  {
699  coinControl()->UnSelect(output); // just to be sure
700  itemOutput->setDisabled(true);
701  itemOutput->setIcon(COLUMN_CHECKBOX, platformStyle->SingleColorIcon(":/icons/lock_closed"));
702  }
703 
704  // set checkbox
705  if (coinControl()->IsSelected(output))
706  itemOutput->setCheckState(COLUMN_CHECKBOX, Qt::Checked);
707  }
708 
709  // amount
710  if (treeMode)
711  {
712  itemWalletAddress->setText(COLUMN_CHECKBOX, "(" + QString::number(nChildren) + ")");
713  itemWalletAddress->setText(COLUMN_AMOUNT, BitcoinUnits::format(nDisplayUnit, nSum));
714  itemWalletAddress->setData(COLUMN_AMOUNT, Qt::UserRole, QVariant((qlonglong)nSum));
715  }
716  }
717 
718  // expand all partially selected
719  if (treeMode)
720  {
721  for (int i = 0; i < ui->treeWidget->topLevelItemCount(); i++)
722  if (ui->treeWidget->topLevelItem(i)->checkState(COLUMN_CHECKBOX) == Qt::PartiallyChecked)
723  ui->treeWidget->topLevelItem(i)->setExpanded(true);
724  }
725 
726  // sort view
727  sortView(sortColumn, sortOrder);
728  ui->treeWidget->setEnabled(true);
729 }
