195  void Push(const RPCArg& arg, const size_t current_indent = 5, const OuterType outer_type = OuterType::NAMED_ARG)
196  {
197  const auto indent = std::string(current_indent, ' ');
198  const auto indent_next = std::string(current_indent + 2, ' ');
199  switch (arg.m_type) {
200  case RPCArg::Type::STR_HEX:
201  case RPCArg::Type::STR:
202  case RPCArg::Type::NUM:
203  case RPCArg::Type::AMOUNT:
204  case RPCArg::Type::BOOL: {
205  if (outer_type == OuterType::NAMED_ARG) return; // Nothing more to do for non-recursive types on first recursion
206  auto left = indent;
207  if (arg.m_type_str.size() != 0 && outer_type == OuterType::OBJ) {
208  left += "\"" + arg.m_name + "\": " + arg.m_type_str.at(0);
209  } else {
210  left += outer_type == OuterType::OBJ ? arg.ToStringObj(/* oneline */ false) : arg.ToString(/* oneline */ false);
211  }
212  left += ",";
213  PushSection({left, arg.ToDescriptionString()});
214  break;
215  }
216  case RPCArg::Type::OBJ:
217  case RPCArg::Type::OBJ_USER_KEYS: {
218  const auto right = outer_type == OuterType::NAMED_ARG ? "" : arg.ToDescriptionString();
219  PushSection({indent + "{", right});
220  for (const auto& arg_inner : arg.m_inner) {
221  Push(arg_inner, current_indent + 2, OuterType::OBJ);
222  }
223  if (arg.m_type != RPCArg::Type::OBJ) {
224  PushSection({indent_next + "...", ""});
225  }
226  PushSection({indent + "}" + (outer_type != OuterType::NAMED_ARG ? "," : ""), ""});
227  break;
228  }
229  case RPCArg::Type::ARR: {
230  auto left = indent;
231  left += outer_type == OuterType::OBJ ? "\"" + arg.m_name + "\": " : "";
232  left += "[";
233  const auto right = outer_type == OuterType::NAMED_ARG ? "" : arg.ToDescriptionString();
234  PushSection({left, right});
235  for (const auto& arg_inner : arg.m_inner) {
236  Push(arg_inner, current_indent + 2, OuterType::ARR);
237  }
238  PushSection({indent_next + "...", ""});
239  PushSection({indent + "]" + (outer_type != OuterType::NAMED_ARG ? "," : ""), ""});
240  break;
241  }
242 
243  // no default case, so the compiler can warn about missing cases
244  }
245  }

247  std::string ToString() const
248  {
249  std::string ret;
250  const size_t pad = m_max_pad + 4;
251  for (const auto& s : m_sections) {
252  if (s.m_right.empty()) {
253  ret += s.m_left;
254  ret += "\n";
255  continue;
256  }
257 
258  std::string left = s.m_left;
259  left.resize(pad, ' ');
260  ret += left;
261 
262  // Properly pad after newlines
263  std::string right;
264  size_t begin = 0;
265  size_t new_line_pos = s.m_right.find_first_of('\n');
266  while (true) {
267  right += s.m_right.substr(begin, new_line_pos - begin);
268  if (new_line_pos == std::string::npos) {
269  break; //No new line
270  }
271  right += "\n" + std::string(pad, ' ');
272  begin = s.m_right.find_first_not_of(' ', new_line_pos + 1);
273  if (begin == std::string::npos) {
274  break; // Empty line
275  }
276  new_line_pos = s.m_right.find_first_of('\n', begin + 1);
277  }
278  ret += right;
279  ret += "\n";
280  }
281  return ret;
282  }

343  ret += "\n";
344 
345  // Description
346  ret += m_description;
347 
348  // Arguments
349  Sections sections;
350  for (size_t i{0}; i < m_args.size(); ++i) {
351  const auto& arg = m_args.at(i);
352 
353  if (i == 0) ret += "\nArguments:\n";
354 
355  // Push named argument name and description
356  sections.m_sections.emplace_back(std::to_string(i + 1) + ". " + arg.m_name, arg.ToDescriptionString());
357  sections.m_max_pad = std::max(sections.m_max_pad, sections.m_sections.back().m_left.size());
358 
359  // Recursively push nested args
360  sections.Push(arg);
361  }
362  ret += sections.ToString();
363 
364  // Result
365  ret += m_results.ToDescriptionString();
