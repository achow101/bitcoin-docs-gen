311  void Push(const RPCArg& arg, const size_t current_indent = 5, const OuterType outer_type = OuterType::NAMED_ARG)
312  {
313  const auto indent = std::string(current_indent, ' ');
314  const auto indent_next = std::string(current_indent + 2, ' ');
315  const bool push_name{outer_type == OuterType::OBJ}; // Dictionary keys must have a name
316 
317  switch (arg.m_type) {
318  case RPCArg::Type::STR_HEX:
319  case RPCArg::Type::STR:
320  case RPCArg::Type::NUM:
321  case RPCArg::Type::AMOUNT:
322  case RPCArg::Type::RANGE:
323  case RPCArg::Type::BOOL: {
324  if (outer_type == OuterType::NAMED_ARG) return; // Nothing more to do for non-recursive types on first recursion
325  auto left = indent;
326  if (arg.m_type_str.size() != 0 && push_name) {
327  left += "\"" + arg.m_name + "\": " + arg.m_type_str.at(0);
328  } else {
329  left += push_name ? arg.ToStringObj(/* oneline */ false) : arg.ToString(/* oneline */ false);
330  }
331  left += ",";
332  PushSection({left, arg.ToDescriptionString()});
333  break;
334  }
335  case RPCArg::Type::OBJ:
336  case RPCArg::Type::OBJ_USER_KEYS: {
337  const auto right = outer_type == OuterType::NAMED_ARG ? "" : arg.ToDescriptionString();
338  PushSection({indent + (push_name ? "\"" + arg.m_name + "\": " : "") + "{", right});
339  for (const auto& arg_inner : arg.m_inner) {
340  Push(arg_inner, current_indent + 2, OuterType::OBJ);
341  }
342  if (arg.m_type != RPCArg::Type::OBJ) {
343  PushSection({indent_next + "...", ""});
344  }
345  PushSection({indent + "}" + (outer_type != OuterType::NAMED_ARG ? "," : ""), ""});
346  break;
347  }
348  case RPCArg::Type::ARR: {
349  auto left = indent;
350  left += push_name ? "\"" + arg.m_name + "\": " : "";
351  left += "[";
352  const auto right = outer_type == OuterType::NAMED_ARG ? "" : arg.ToDescriptionString();
353  PushSection({left, right});
354  for (const auto& arg_inner : arg.m_inner) {
355  Push(arg_inner, current_indent + 2, OuterType::ARR);
356  }
357  PushSection({indent_next + "...", ""});
358  PushSection({indent + "]" + (outer_type != OuterType::NAMED_ARG ? "," : ""), ""});
359  break;
360  }
361 
362  // no default case, so the compiler can warn about missing cases
363  }
364  }

369  std::string ToString() const
370  {
371  std::string ret;
372  const size_t pad = m_max_pad + 4;
373  for (const auto& s : m_sections) {
374  if (s.m_right.empty()) {
375  ret += s.m_left;
376  ret += "\n";
377  continue;
378  }
379 
380  std::string left = s.m_left;
381  left.resize(pad, ' ');
382  ret += left;
383 
384  // Properly pad after newlines
385  std::string right;
386  size_t begin = 0;
387  size_t new_line_pos = s.m_right.find_first_of('\n');
388  while (true) {
389  right += s.m_right.substr(begin, new_line_pos - begin);
390  if (new_line_pos == std::string::npos) {
391  break; //No new line
392  }
393  right += "\n" + std::string(pad, ' ');
394  begin = s.m_right.find_first_not_of(' ', new_line_pos + 1);
395  if (begin == std::string::npos) {
396  break; // Empty line
397  }
398  new_line_pos = s.m_right.find_first_of('\n', begin + 1);
399  }
400  ret += right;
401  ret += "\n";
402  }
403  return ret;
404  }
