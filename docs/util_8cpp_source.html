28 CPubKey AddrToPubKey(CKeyStore* const keystore, const std::string& addr_in)
29 {
30  CTxDestination dest = DecodeDestination(addr_in);

33  }
34  CKeyID key = GetKeyForDestination(*keystore, dest);

38  CPubKey vchPubKey;

42  if (!vchPubKey.IsFullyValid()) {
43  throw JSONRPCError(RPC_INTERNAL_ERROR, "Wallet contains an invalid public key");
44  }

155  void Push(const RPCArg& arg, const size_t current_indent = 5, const OuterType outer_type = OuterType::NAMED_ARG)
156  {
157  const auto indent = std::string(current_indent, ' ');
158  const auto indent_next = std::string(current_indent + 2, ' ');
159  switch (arg.m_type) {
160  case RPCArg::Type::STR_HEX:
161  case RPCArg::Type::STR:
162  case RPCArg::Type::NUM:
163  case RPCArg::Type::AMOUNT:
164  case RPCArg::Type::BOOL: {
165  if (outer_type == OuterType::NAMED_ARG) return; // Nothing more to do for non-recursive types on first recursion
166  auto left = indent;
167  if (arg.m_type_str.size() != 0 && outer_type == OuterType::OBJ) {
168  left += "\"" + arg.m_name + "\": " + arg.m_type_str.at(0);
169  } else {
170  left += outer_type == OuterType::OBJ ? arg.ToStringObj(/* oneline */ false) : arg.ToString(/* oneline */ false);
171  }
172  left += ",";
173  PushSection({left, arg.ToDescriptionString(/* implicitly_required */ outer_type == OuterType::ARR)});
174  break;
175  }
176  case RPCArg::Type::OBJ:
177  case RPCArg::Type::OBJ_USER_KEYS: {
178  const auto right = outer_type == OuterType::NAMED_ARG ? "" : arg.ToDescriptionString(/* implicitly_required */ outer_type == OuterType::ARR);
179  PushSection({indent + "{", right});
180  for (const auto& arg_inner : arg.m_inner) {
181  Push(arg_inner, current_indent + 2, OuterType::OBJ);
182  }
183  if (arg.m_type != RPCArg::Type::OBJ) {
184  PushSection({indent_next + "...", ""});
185  }
186  PushSection({indent + "}" + (outer_type != OuterType::NAMED_ARG ? "," : ""), ""});
187  break;
188  }
189  case RPCArg::Type::ARR: {
190  auto left = indent;
191  left += outer_type == OuterType::OBJ ? "\"" + arg.m_name + "\": " : "";
192  left += "[";
193  const auto right = outer_type == OuterType::NAMED_ARG ? "" : arg.ToDescriptionString(/* implicitly_required */ outer_type == OuterType::ARR);
194  PushSection({left, right});
195  for (const auto& arg_inner : arg.m_inner) {
196  Push(arg_inner, current_indent + 2, OuterType::ARR);
197  }
198  PushSection({indent_next + "...", ""});
199  PushSection({indent + "]" + (outer_type != OuterType::NAMED_ARG ? "," : ""), ""});
200  break;
201  }
202 
203  // no default case, so the compiler can warn about missing cases
204  }
205  }

207  std::string ToString() const
208  {
209  std::string ret;
210  const size_t pad = m_max_pad + 4;
211  for (const auto& s : m_sections) {
212  if (s.m_right.empty()) {
213  ret += s.m_left;
214  ret += "\n";
215  continue;
216  }
217 
218  std::string left = s.m_left;
219  left.resize(pad, ' ');
220  ret += left;
221 
222  // Properly pad after newlines
223  std::string right;
224  size_t begin = 0;
225  size_t new_line_pos = s.m_right.find_first_of('\n');
226  while (true) {
227  right += s.m_right.substr(begin, new_line_pos - begin);
228  if (new_line_pos == std::string::npos) {
229  break; //No new line
230  }
231  right += "\n" + std::string(pad, ' ');
232  begin = s.m_right.find_first_not_of(' ', new_line_pos + 1);
233  if (begin == std::string::npos) {
234  break; // Empty line
235  }
236  new_line_pos = s.m_right.find_first_of('\n', begin + 1);
237  }
238  ret += right;
239  ret += "\n";
240  }
241  return ret;
242  }
