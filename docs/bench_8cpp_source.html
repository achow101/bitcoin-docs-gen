100 void benchmark::BenchRunner::RunAll(Printer& printer, uint64_t num_evals, double scaling, const std::string& filter, bool is_list_only)
101 {
102  if (!std::ratio_less_equal<benchmark::clock::period, std::micro>::value) {
103  std::cerr << "WARNING: Clock precision is worse than microsecond - benchmarks may be less accurate!\n";
104  }
105 #ifdef DEBUG
106  std::cerr << "WARNING: This is a debug build - may result in slower benchmarks.\n";
107 #endif
108 
109  std::regex reFilter(filter);
110  std::smatch baseMatch;
111 
112  printer.header();
113 
114  for (const auto& p : benchmarks()) {
115  TestingSetup test{CBaseChainParams::REGTEST};
116  {
117  assert(::chainActive.Height() == 0);
118  const bool witness_enabled{IsWitnessEnabled(::chainActive.Tip(), Params().GetConsensus())};
119  assert(witness_enabled);
120  }
121 
122  if (!std::regex_match(p.first, baseMatch, reFilter)) {
123  continue;
124  }
125 
126  uint64_t num_iters = static_cast<uint64_t>(p.second.num_iters_for_one_second * scaling);
127  if (0 == num_iters) {
128  num_iters = 1;
129  }
130  State state(p.first, num_evals, num_iters, printer);
131  if (!is_list_only) {
132  p.second.func(state);
133  }
134  printer.result(state);
135  }
