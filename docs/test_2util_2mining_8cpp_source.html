28std::vector<std::shared_ptr<CBlock>> CreateBlockChain(size_t total_height, const CChainParams& params)
29{
30 std::vector<std::shared_ptr<CBlock>> ret{total_height};
31 auto time{params.GenesisBlock().nTime};
32 for (size_t height{0}; height < total_height; ++height) {
33 CBlock& block{*(ret.at(height) = std::make_shared<CBlock>())};
34
35 CMutableTransaction coinbase_tx;
36 coinbase_tx.vin.resize(1);
37 coinbase_tx.vin[0].prevout.SetNull();
38 coinbase_tx.vout.resize(1);
39 coinbase_tx.vout[0].scriptPubKey = P2WSH_OP_TRUE;
40 coinbase_tx.vout[0].nValue = GetBlockSubsidy(height + 1, params.GetConsensus());
41 coinbase_tx.vin[0].scriptSig = CScript() << (height + 1) << OP_0;
42 block.vtx = {MakeTransactionRef(std::move(coinbase_tx))};
43
44 block.nVersion = VERSIONBITS_LAST_OLD_BLOCK_VERSION;
45 block.hashPrevBlock = (height >= 1 ? *ret.at(height - 1) : params.GenesisBlock()).GetHash();
46 block.hashMerkleRoot = BlockMerkleRoot(block);
47 block.nTime = ++time;
48 block.nBits = params.GenesisBlock().nBits;
49 block.nNonce = 0;
50
51 while (!CheckProofOfWork(block.GetHash(), block.nBits, params.GetConsensus())) {
52 ++block.nNonce;
53 assert(block.nNonce);
54 }
55 }
56 return ret;
57}

59CTxIn MineBlock(const NodeContext& node, const CScript& coinbase_scriptPubKey)
60{
61 auto block = PrepareBlock(node, coinbase_scriptPubKey);
62
63 while (!CheckProofOfWork(block->GetHash(), block->nBits, Params().GetConsensus())) {
64 ++block->nNonce;
65 assert(block->nNonce);
66 }
67
68 bool processed{Assert(node.chainman)->ProcessNewBlock(Params(), block, true, nullptr)};
69 assert(processed);
70
71 return CTxIn{block->vtx[0]->GetHash(), 0};
72}

74std::shared_ptr<CBlock> PrepareBlock(const NodeContext& node, const CScript& coinbase_scriptPubKey)
75{
76 auto block = std::make_shared<CBlock>(
77 BlockAssembler{Assert(node.chainman)->ActiveChainstate(), *Assert(node.mempool), Params()}
78 .CreateNewBlock(coinbase_scriptPubKey)
79 ->block);
80
81 LOCK(cs_main);
82 block->nTime = Assert(node.chainman)->ActiveChain().Tip()->GetMedianTimePast() + 1;
83 block->hashMerkleRoot = BlockMerkleRoot(*block);
84
85 return block;
