144 void ReceiveCoinsDialog::on_receiveButton_clicked()
145 {
146  if(!model || !model->getOptionsModel() || !model->getAddressTableModel() || !model->getRecentRequestsTableModel())
147  return;
148 
149  QString address;
150  QString label = ui->reqLabel->text();
151  /* Generate new receiving address */
152  OutputType address_type;
153  if (ui->useBech32->isChecked()) {
154  address_type = OutputType::BECH32;
155  } else {
156  address_type = model->wallet().getDefaultAddressType();
157  if (address_type == OutputType::BECH32) {
158  address_type = OutputType::P2SH_SEGWIT;
159  }
160  }
161  address = model->getAddressTableModel()->addRow(AddressTableModel::Receive, label, "", address_type);
162  SendCoinsRecipient info(address, label,
163  ui->reqAmount->value(), ui->reqMessage->text());
164  ReceiveRequestDialog *dialog = new ReceiveRequestDialog(this);
165  dialog->setAttribute(Qt::WA_DeleteOnClose);
166  dialog->setModel(model);
167  dialog->setInfo(info);
168  dialog->show();
169  clear();
170 
171  /* Store request for later reference */
172  model->getRecentRequestsTableModel()->addNewRequest(info);
173 }
