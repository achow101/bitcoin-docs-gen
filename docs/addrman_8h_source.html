298  void Serialize(Stream &s) const
299  {
300  LOCK(cs);
301 
302  unsigned char nVersion = 1;
303  s << nVersion;
304  s << ((unsigned char)32);
305  s << nKey;
306  s << nNew;
307  s << nTried;
308 
309  int nUBuckets = ADDRMAN_NEW_BUCKET_COUNT ^ (1 << 30);
310  s << nUBuckets;
311  std::map<int, int> mapUnkIds;
312  int nIds = 0;
313  for (const auto& entry : mapInfo) {
314  mapUnkIds[entry.first] = nIds;
315  const CAddrInfo &info = entry.second;
316  if (info.nRefCount) {
317  assert(nIds != nNew); // this means nNew was wrong, oh ow
318  s << info;
319  nIds++;
320  }
321  }
322  nIds = 0;
323  for (const auto& entry : mapInfo) {
324  const CAddrInfo &info = entry.second;
325  if (info.fInTried) {
326  assert(nIds != nTried); // this means nTried was wrong, oh ow
327  s << info;
328  nIds++;
329  }
330  }
331  for (int bucket = 0; bucket < ADDRMAN_NEW_BUCKET_COUNT; bucket++) {
332  int nSize = 0;
333  for (int i = 0; i < ADDRMAN_BUCKET_SIZE; i++) {
334  if (vvNew[bucket][i] != -1)
335  nSize++;
336  }
337  s << nSize;
338  for (int i = 0; i < ADDRMAN_BUCKET_SIZE; i++) {
339  if (vvNew[bucket][i] != -1) {
340  int nIndex = mapUnkIds[vvNew[bucket][i]];
341  s << nIndex;
342  }
343  }
344  }
345  }

347  template<typename Stream>
348  void Unserialize(Stream& s)
349  {
350  LOCK(cs);
351 
352  Clear();
353 
354  unsigned char nVersion;
355  s >> nVersion;
356  unsigned char nKeySize;
357  s >> nKeySize;
358  if (nKeySize != 32) throw std::ios_base::failure("Incorrect keysize in addrman deserialization");
359  s >> nKey;
360  s >> nNew;
361  s >> nTried;
362  int nUBuckets = 0;
363  s >> nUBuckets;
364  if (nVersion != 0) {
365  nUBuckets ^= (1 << 30);
366  }
367 
368  if (nNew > ADDRMAN_NEW_BUCKET_COUNT * ADDRMAN_BUCKET_SIZE) {
369  throw std::ios_base::failure("Corrupt CAddrMan serialization, nNew exceeds limit.");
370  }
371 
372  if (nTried > ADDRMAN_TRIED_BUCKET_COUNT * ADDRMAN_BUCKET_SIZE) {
373  throw std::ios_base::failure("Corrupt CAddrMan serialization, nTried exceeds limit.");
374  }
375 
376  // Deserialize entries from the new table.
377  for (int n = 0; n < nNew; n++) {
378  CAddrInfo &info = mapInfo[n];
379  s >> info;
380  mapAddr[info] = n;
381  info.nRandomPos = vRandom.size();
382  vRandom.push_back(n);
383  if (nVersion != 1 || nUBuckets != ADDRMAN_NEW_BUCKET_COUNT) {
384  // In case the new table data cannot be used (nVersion unknown, or bucket count wrong),
385  // immediately try to give them a reference based on their primary source address.
386  int nUBucket = info.GetNewBucket(nKey);
387  int nUBucketPos = info.GetBucketPosition(nKey, true, nUBucket);
388  if (vvNew[nUBucket][nUBucketPos] == -1) {
389  vvNew[nUBucket][nUBucketPos] = n;
390  info.nRefCount++;
391  }
392  }
393  }
394  nIdCount = nNew;
395 
396  // Deserialize entries from the tried table.
397  int nLost = 0;
398  for (int n = 0; n < nTried; n++) {
399  CAddrInfo info;
400  s >> info;
401  int nKBucket = info.GetTriedBucket(nKey);
402  int nKBucketPos = info.GetBucketPosition(nKey, false, nKBucket);
403  if (vvTried[nKBucket][nKBucketPos] == -1) {
404  info.nRandomPos = vRandom.size();
405  info.fInTried = true;
406  vRandom.push_back(nIdCount);
407  mapInfo[nIdCount] = info;
408  mapAddr[info] = nIdCount;
409  vvTried[nKBucket][nKBucketPos] = nIdCount;
410  nIdCount++;
411  } else {
412  nLost++;
413  }
414  }
415  nTried -= nLost;
416 
417  // Deserialize positions in the new table (if possible).
418  for (int bucket = 0; bucket < nUBuckets; bucket++) {
419  int nSize = 0;
420  s >> nSize;
421  for (int n = 0; n < nSize; n++) {
422  int nIndex = 0;
423  s >> nIndex;
424  if (nIndex >= 0 && nIndex < nNew) {
425  CAddrInfo &info = mapInfo[nIndex];
426  int nUBucketPos = info.GetBucketPosition(nKey, true, bucket);
427  if (nVersion == 1 && nUBuckets == ADDRMAN_NEW_BUCKET_COUNT && vvNew[bucket][nUBucketPos] == -1 && info.nRefCount < ADDRMAN_NEW_BUCKETS_PER_ADDRESS) {
428  info.nRefCount++;
429  vvNew[bucket][nUBucketPos] = nIndex;
430  }
431  }
432  }
433  }
434 
435  // Prune new entries with refcount 0 (as a result of collisions).
436  int nLostUnk = 0;
437  for (std::map<int, CAddrInfo>::const_iterator it = mapInfo.begin(); it != mapInfo.end(); ) {
438  if (it->second.fInTried == false && it->second.nRefCount == 0) {
439  std::map<int, CAddrInfo>::const_iterator itCopy = it++;
440  Delete(itCopy->first);
441  nLostUnk++;
442  } else {
443  it++;
444  }
445  }
446  if (nLost + nLostUnk > 0) {
447  LogPrint(BCLog::ADDRMAN, "addrman lost %i new and %i tried addresses due to collisions\n", nLostUnk, nLost);
448  }
449 
450  Check();
451  }
