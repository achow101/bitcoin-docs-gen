26{
27 uint64_t t1 = h + Sigma1(e) + Ch(e, f, g) + k + w;
28 uint64_t t2 = Sigma0(a) + Maj(a, b, c);
29 d += t1;

159CSHA512& CSHA512::Write(const unsigned char* data, size_t len)
160{
161 const unsigned char* end = data + len;
162 size_t bufsize = bytes % 128;
163 if (bufsize && bufsize + len >= 128) {
164 // Fill the buffer, and process it.
165 memcpy(buf + bufsize, data, 128 - bufsize);
166 bytes += 128 - bufsize;
167 data += 128 - bufsize;
168 sha512::Transform(s, buf);
169 bufsize = 0;
170 }
171 while (end - data >= 128) {
172 // Process full chunks directly from the source.
173 sha512::Transform(s, data);
174 data += 128;
175 bytes += 128;
176 }
177 if (end > data) {
178 // Fill the buffer with what remains.
179 memcpy(buf + bufsize, data, end - data);
180 bytes += end - data;
181 }
182 return *this;
183}
