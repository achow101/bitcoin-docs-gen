103 bool GetLocal(CService& addr, const CNetAddr *paddrPeer)
104 {

106  return false;

108  int nBestScore = -1;

113  {

129 {
130  // It'll only connect to one or two seed nodes because once it connects,
131  // it'll get a pile of addresses with newer timestamps.
132  // Seed nodes are given a random 'last seen time' of between one and two
133  // weeks ago.

371 CNode* CConnman::ConnectNode(CAddress addrConnect, const char *pszDest, bool fCountFailure, bool manual_connection)
372 {
373  if (pszDest == nullptr) {
374  if (IsLocal(addrConnect))
375  return nullptr;
376 
377  // Look for an existing connection
378  CNode* pnode = FindNode(static_cast<CService>(addrConnect));
379  if (pnode)
380  {
381  LogPrintf("Failed to open new connection, already connected\n");
382  return nullptr;
383  }
384  }
385 
387  LogPrint(BCLog::NET, "trying connection %s lastseen=%.1fhrs\n",
388  pszDest ? pszDest : addrConnect.ToString(),
389  pszDest ? 0.0 : (double)(GetAdjustedTime() - addrConnect.nTime)/3600.0);
390 
391  // Resolve
392  const int default_port = Params().GetDefaultPort();
393  if (pszDest) {
394  std::vector<CService> resolved;
395  if (Lookup(pszDest, resolved, default_port, fNameLookup && !HaveNameProxy(), 256) && !resolved.empty()) {
396  addrConnect = CAddress(resolved[GetRand(resolved.size())], NODE_NONE);
397  if (!addrConnect.IsValid()) {
398  LogPrint(BCLog::NET, "Resolver returned invalid address %s for %s\n", addrConnect.ToString(), pszDest);
399  return nullptr;
400  }
401  // It is possible that we already have a connection to the IP/port pszDest resolved to.
402  // In that case, drop the connection that was just created, and return the existing CNode instead.
403  // Also store the name we used to connect in that CNode, so that future FindNode() calls to that
404  // name catch this early.
405  LOCK(cs_vNodes);
406  CNode* pnode = FindNode(static_cast<CService>(addrConnect));
407  if (pnode)
408  {
409  pnode->MaybeSetAddrName(std::string(pszDest));
410  LogPrintf("Failed to open new connection, already connected\n");
411  return nullptr;
412  }
413  }
414  }
415 
416  // Connect
417  bool connected = false;
418  SOCKET hSocket = INVALID_SOCKET;
419  proxyType proxy;
420  if (addrConnect.IsValid()) {
421  bool proxyConnectionFailed = false;
422 
423  if (GetProxy(addrConnect.GetNetwork(), proxy)) {
424  hSocket = CreateSocket(proxy.proxy);
425  if (hSocket == INVALID_SOCKET) {
426  return nullptr;
427  }
428  connected = ConnectThroughProxy(proxy, addrConnect.ToStringIP(), addrConnect.GetPort(), hSocket, nConnectTimeout, &proxyConnectionFailed);
429  } else {
430  // no proxy needed (none set for target network)
431  hSocket = CreateSocket(addrConnect);
432  if (hSocket == INVALID_SOCKET) {
433  return nullptr;
434  }
435  connected = ConnectSocketDirectly(addrConnect, hSocket, nConnectTimeout, manual_connection);
436  }
437  if (!proxyConnectionFailed) {
438  // If a connection to the node was attempted, and failure (if any) is not caused by a problem connecting to
439  // the proxy, mark this as an attempt.
440  addrman.Attempt(addrConnect, fCountFailure);
441  }
442  } else if (pszDest && GetNameProxy(proxy)) {
443  hSocket = CreateSocket(proxy.proxy);
444  if (hSocket == INVALID_SOCKET) {
445  return nullptr;
446  }
447  std::string host;
448  int port = default_port;
449  SplitHostPort(std::string(pszDest), port, host);
450  connected = ConnectThroughProxy(proxy, host, port, hSocket, nConnectTimeout, nullptr);
451  }
452  if (!connected) {
453  CloseSocket(hSocket);
454  return nullptr;
455  }
456 
457  // Add node
458  NodeId id = GetNewNodeId();
459  uint64_t nonce = GetDeterministicRandomizer(RANDOMIZER_ID_LOCALHOSTNONCE).Write(id).Finalize();
460  CAddress addr_bind = GetBindAddress(hSocket);
461  CNode* pnode = new CNode(id, nLocalServices, GetBestHeight(), hSocket, addrConnect, CalculateKeyedNetGroup(addrConnect), nonce, addr_bind, pszDest ? pszDest : "", false);
462  pnode->AddRef();
463 
464  return pnode;
465 }

996 bool CConnman::AttemptToEvictConnection()
997 {
998  std::vector<NodeEvictionCandidate> vEvictionCandidates;
999  {
1000  LOCK(cs_vNodes);
1001 
1002  for (const CNode* node : vNodes) {
1003  if (node->fWhitelisted)
1004  continue;
1005  if (!node->fInbound)
1006  continue;
1007  if (node->fDisconnect)
1008  continue;
1009  LOCK(node->cs_filter);
1010  NodeEvictionCandidate candidate = {node->GetId(), node->nTimeConnected, node->nMinPingUsecTime,
1011  node->nLastBlockTime, node->nLastTXTime,
1012  HasAllDesirableServiceFlags(node->nServices),
1013  node->fRelayTxes, node->pfilter != nullptr, node->addr, node->nKeyedNetGroup};
1014  vEvictionCandidates.push_back(candidate);
1015  }
1016  }
1017 
1018  // Protect connections with certain characteristics
1019 
1020  // Deterministically select 4 peers to protect by netgroup.
1021  // An attacker cannot predict which netgroups will be protected
1022  EraseLastKElements(vEvictionCandidates, CompareNetGroupKeyed, 4);
1023  // Protect the 8 nodes with the lowest minimum ping time.
1024  // An attacker cannot manipulate this metric without physically moving nodes closer to the target.
1025  EraseLastKElements(vEvictionCandidates, ReverseCompareNodeMinPingTime, 8);
1026  // Protect 4 nodes that most recently sent us transactions.
1027  // An attacker cannot manipulate this metric without performing useful work.
1028  EraseLastKElements(vEvictionCandidates, CompareNodeTXTime, 4);
1029  // Protect 4 nodes that most recently sent us blocks.
1030  // An attacker cannot manipulate this metric without performing useful work.
1031  EraseLastKElements(vEvictionCandidates, CompareNodeBlockTime, 4);
1032  // Protect the half of the remaining nodes which have been connected the longest.
1033  // This replicates the non-eviction implicit behavior, and precludes attacks that start later.
1034  EraseLastKElements(vEvictionCandidates, ReverseCompareNodeTimeConnected, vEvictionCandidates.size() / 2);
1035 
1036  if (vEvictionCandidates.empty()) return false;
1037 
1038  // Identify the network group with the most connections and youngest member.
1039  // (vEvictionCandidates is already sorted by reverse connect time)
1040  uint64_t naMostConnections;
1041  unsigned int nMostConnections = 0;
1042  int64_t nMostConnectionsTime = 0;
1043  std::map<uint64_t, std::vector<NodeEvictionCandidate> > mapNetGroupNodes;
1044  for (const NodeEvictionCandidate &node : vEvictionCandidates) {
1045  std::vector<NodeEvictionCandidate> &group = mapNetGroupNodes[node.nKeyedNetGroup];
1046  group.push_back(node);
1047  int64_t grouptime = group[0].nTimeConnected;
1048 
1049  if (group.size() > nMostConnections || (group.size() == nMostConnections && grouptime > nMostConnectionsTime)) {
1050  nMostConnections = group.size();
1051  nMostConnectionsTime = grouptime;
1052  naMostConnections = node.nKeyedNetGroup;
1053  }
1054  }
1055 
1056  // Reduce to the network group with the most connections
1057  vEvictionCandidates = std::move(mapNetGroupNodes[naMostConnections]);
1058 
1059  // Disconnect from the network group with the most connections
1060  NodeId evicted = vEvictionCandidates.front().id;
1061  LOCK(cs_vNodes);
1062  for (CNode* pnode : vNodes) {
1063  if (pnode->GetId() == evicted) {
1064  pnode->fDisconnect = true;
1065  return true;
1066  }
1067  }
1068  return false;
1069 }

1261 void CConnman::SocketHandler()
1262 {
1263  //
1264  // Find which sockets have data to receive
1265  //
1266  struct timeval timeout;
1267  timeout.tv_sec = 0;
1268  timeout.tv_usec = 50000; // frequency to poll pnode->vSend
1269 
1270  fd_set fdsetRecv;
1271  fd_set fdsetSend;
1272  fd_set fdsetError;
1273  FD_ZERO(&fdsetRecv);
1274  FD_ZERO(&fdsetSend);
1275  FD_ZERO(&fdsetError);
1276  SOCKET hSocketMax = 0;
1277  bool have_fds = false;
1278 
1279  for (const ListenSocket& hListenSocket : vhListenSocket) {
1280  FD_SET(hListenSocket.socket, &fdsetRecv);
1281  hSocketMax = std::max(hSocketMax, hListenSocket.socket);
1282  have_fds = true;
1283  }
1284 
1285  {
1286  LOCK(cs_vNodes);
1287  for (CNode* pnode : vNodes)
1288  {
1289  // Implement the following logic:
1290  // * If there is data to send, select() for sending data. As this only
1291  // happens when optimistic write failed, we choose to first drain the
1292  // write buffer in this case before receiving more. This avoids
1293  // needlessly queueing received data, if the remote peer is not themselves
1294  // receiving data. This means properly utilizing TCP flow control signalling.
1295  // * Otherwise, if there is space left in the receive buffer, select() for
1296  // receiving data.
1297  // * Hand off all complete messages to the processor, to be handled without
1298  // blocking here.
1299 
1300  bool select_recv = !pnode->fPauseRecv;
1301  bool select_send;
1302  {
1303  LOCK(pnode->cs_vSend);
1304  select_send = !pnode->vSendMsg.empty();
1305  }
1306 
1307  LOCK(pnode->cs_hSocket);
1308  if (pnode->hSocket == INVALID_SOCKET)
1309  continue;
1310 
1311  FD_SET(pnode->hSocket, &fdsetError);
1312  hSocketMax = std::max(hSocketMax, pnode->hSocket);
1313  have_fds = true;
1314 
1315  if (select_send) {
1316  FD_SET(pnode->hSocket, &fdsetSend);
1317  continue;
1318  }
1319  if (select_recv) {
1320  FD_SET(pnode->hSocket, &fdsetRecv);
1321  }
1322  }
1323  }
1324 
1325  int nSelect = select(have_fds ? hSocketMax + 1 : 0,
1326  &fdsetRecv, &fdsetSend, &fdsetError, &timeout);
1327  if (interruptNet)
1328  return;
1329 
1330  if (nSelect == SOCKET_ERROR)
1331  {
1332  if (have_fds)
1333  {
1334  int nErr = WSAGetLastError();
1335  LogPrintf("socket select error %s\n", NetworkErrorString(nErr));
1336  for (unsigned int i = 0; i <= hSocketMax; i++)
1337  FD_SET(i, &fdsetRecv);

1366  for (CNode* pnode : vNodesCopy)
1367  {
1368  if (interruptNet)

1372  // Receive
1373  //
1374  bool recvSet = false;
1375  bool sendSet = false;
1376  bool errorSet = false;
1377  {
1378  LOCK(pnode->cs_hSocket);

1380  continue;

1382  sendSet = FD_ISSET(pnode->hSocket, &fdsetSend);
1383  errorSet = FD_ISSET(pnode->hSocket, &fdsetError);
1384  }

1386  {

1388  char pchBuf[0x10000];

1390  {
1391  LOCK(pnode->cs_hSocket);

1407  break;

1411  LOCK(pnode->cs_vProcessMsg);
1412  pnode->vProcessMsg.splice(pnode->vProcessMsg.end(), pnode->vRecvMsg, pnode->vRecvMsg.begin(), it);
1413  pnode->nProcessQueueSize += nSizeAdded;
1414  pnode->fPauseRecv = pnode->nProcessQueueSize > nReceiveFloodSize;
1415  }
1416  WakeMessageHandler();
1417  }
1418  }
1419  else if (nBytes == 0)
1420  {
1421  // socket closed gracefully
1422  if (!pnode->fDisconnect) {
1423  LogPrint(BCLog::NET, "socket closed\n");
1424  }
1425  pnode->CloseSocketDisconnect();
1426  }
1427  else if (nBytes < 0)
1428  {
1429  // error
1430  int nErr = WSAGetLastError();
1431  if (nErr != WSAEWOULDBLOCK && nErr != WSAEMSGSIZE && nErr != WSAEINTR && nErr != WSAEINPROGRESS)

1435  pnode->CloseSocketDisconnect();
1436  }
1437  }
1438  }

1447  if (nBytes) {
1448  RecordBytesSent(nBytes);
1449  }
1450  }
1451 
1452  InactivityCheck(pnode);
1453  }
1454  {
1455  LOCK(cs_vNodes);
1456  for (CNode* pnode : vNodesCopy)
1457  pnode->Release();
1458  }

1753 void CConnman::ThreadOpenConnections(const std::vector<std::string> connect)
1754 {
1755  // Connect to specific addresses
1756  if (!connect.empty())
1757  {
1758  for (int64_t nLoop = 0;; nLoop++)
1759  {
1760  ProcessOneShot();
1761  for (const std::string& strAddr : connect)
1762  {
1763  CAddress addr(CService(), NODE_NONE);
1764  OpenNetworkConnection(addr, false, nullptr, strAddr.c_str(), false, false, true);
1765  for (int i = 0; i < 10 && i < nLoop; i++)
1766  {
1767  if (!interruptNet.sleep_for(std::chrono::milliseconds(500)))
1768  return;
1769  }
1770  }
1771  if (!interruptNet.sleep_for(std::chrono::milliseconds(500)))
1772  return;
1773  }
1774  }
1775 
1776  // Initiate network connections
1777  int64_t nStart = GetTime();
1778 
1779  // Minimum time before next feeler connection (in microseconds).
1780  int64_t nNextFeeler = PoissonNextSend(nStart*1000*1000, FEELER_INTERVAL);
1781  while (!interruptNet)
1782  {
1783  ProcessOneShot();
1784 
1785  if (!interruptNet.sleep_for(std::chrono::milliseconds(500)))
1786  return;
1787 
1788  CSemaphoreGrant grant(*semOutbound);
1789  if (interruptNet)
1790  return;
1791 
1792  // Add seed nodes if DNS seeds are all down (an infrastructure attack?).
1793  if (addrman.size() == 0 && (GetTime() - nStart > 60)) {
1794  static bool done = false;
1795  if (!done) {
1796  LogPrintf("Adding fixed seed nodes as DNS doesn't seem to be available.\n");
1797  CNetAddr local;
1798  local.SetInternal("fixedseeds");
1799  addrman.Add(convertSeed6(Params().FixedSeeds()), local);
1800  done = true;
1801  }
1802  }
1803 
1804  //
1805  // Choose an address to connect to based on most recently seen
1806  //
1807  CAddress addrConnect;
1808 
1809  // Only connect out to one peer per network group (/16 for IPv4).
1810  int nOutbound = 0;
1811  std::set<std::vector<unsigned char> > setConnected;
1812  {
1813  LOCK(cs_vNodes);
1814  for (const CNode* pnode : vNodes) {
1815  if (!pnode->fInbound && !pnode->m_manual_connection) {
1816  // Netgroups for inbound and addnode peers are not excluded because our goal here
1817  // is to not use multiple of our limited outbound slots on a single netgroup
1818  // but inbound and addnode peers do not use our outbound slots. Inbound peers
1819  // also have the added issue that they're attacker controlled and could be used
1820  // to prevent us from connecting to particular hosts if we used them here.
1821  setConnected.insert(pnode->addr.GetGroup());
1822  nOutbound++;
1823  }
1824  }
1825  }
1826 
1827  // Feeler Connections
1828  //
1829  // Design goals:
1830  // * Increase the number of connectable addresses in the tried table.
1831  //
1832  // Method:
1833  // * Choose a random address from new and attempt to connect to it if we can connect
1834  // successfully it is added to tried.
1835  // * Start attempting feeler connections only after node finishes making outbound
1836  // connections.
1837  // * Only make a feeler connection once every few minutes.
1838  //
1839  bool fFeeler = false;
1840 
1841  if (nOutbound >= nMaxOutbound && !GetTryNewOutboundPeer()) {
1842  int64_t nTime = GetTimeMicros(); // The current time right now (in microseconds).
1843  if (nTime > nNextFeeler) {
1844  nNextFeeler = PoissonNextSend(nTime, FEELER_INTERVAL);
1845  fFeeler = true;
1846  } else {
1847  continue;
1848  }
1849  }
1850 
1851  addrman.ResolveCollisions();
1852 
1853  int64_t nANow = GetAdjustedTime();
1854  int nTries = 0;
1855  while (!interruptNet)
1856  {
1857  CAddrInfo addr = addrman.SelectTriedCollision();
1858 
1859  // SelectTriedCollision returns an invalid address if it is empty.
1860  if (!fFeeler || !addr.IsValid()) {
1861  addr = addrman.Select(fFeeler);
1862  }
1863 
1864  // if we selected an invalid address, restart
1865  if (!addr.IsValid() || setConnected.count(addr.GetGroup()) || IsLocal(addr))
1866  break;
1867 
1868  // If we didn't find an appropriate destination after trying 100 addresses fetched from addrman,
1869  // stop this loop, and let the outer loop run again (which sleeps, adds seed nodes, recalculates
1870  // already-connected network ranges, ...) before trying new addrman addresses.
1871  nTries++;
1872  if (nTries > 100)
1873  break;
1874 
1875  if (IsLimited(addr))
1876  continue;
1877 
1878  // only consider very recently tried nodes after 30 failed attempts
1879  if (nANow - addr.nLastTry < 600 && nTries < 30)
1880  continue;
1881 
1882  // for non-feelers, require all the services we'll want,
1883  // for feelers, only require they be a full node (only because most
1884  // SPV clients don't have a good address DB available)
1885  if (!fFeeler && !HasAllDesirableServiceFlags(addr.nServices)) {
1886  continue;
1887  } else if (fFeeler && !MayHaveUsefulAddressDB(addr.nServices)) {
1888  continue;
1889  }
1890 
1891  // do not allow non-default ports, unless after 50 invalid addresses selected already
1892  if (addr.GetPort() != Params().GetDefaultPort() && nTries < 50)
1893  continue;
1894 
1895  addrConnect = addr;
1896  break;
1897  }
1898 
1899  if (addrConnect.IsValid()) {
1900 
1901  if (fFeeler) {
1902  // Add small amount of random noise before connection to avoid synchronization.
1903  int randsleep = GetRandInt(FEELER_SLEEP_WINDOW * 1000);
1904  if (!interruptNet.sleep_for(std::chrono::milliseconds(randsleep)))
1905  return;
1906  LogPrint(BCLog::NET, "Making feeler connection to %s\n", addrConnect.ToString());
1907  }
1908 
1909  OpenNetworkConnection(addrConnect, (int)setConnected.size() >= std::min(nMaxConnections - 1, 2), &grant, nullptr, false, fFeeler);
1910  }
1911  }
