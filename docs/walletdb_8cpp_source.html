170 static bool
171 ReadKeyValue(CWallet* pwallet, CDataStream& ssKey, CDataStream& ssValue,
172  CWalletScanState &wss, std::string& strType, std::string& strErr) EXCLUSIVE_LOCKS_REQUIRED(pwallet->cs_wallet)
173 {
174  try {
175  // Unserialize
176  // Taking advantage of the fact that pair serialization
177  // is just the two items serialized one after the other
178  ssKey >> strType;
179  if (strType == "name")
180  {
181  std::string strAddress;
182  ssKey >> strAddress;
183  ssValue >> pwallet->mapAddressBook[DecodeDestination(strAddress)].name;
184  }
185  else if (strType == "purpose")
186  {
187  std::string strAddress;
188  ssKey >> strAddress;
189  ssValue >> pwallet->mapAddressBook[DecodeDestination(strAddress)].purpose;
190  }
191  else if (strType == "tx")
192  {
193  uint256 hash;
194  ssKey >> hash;
195  CWalletTx wtx(nullptr /* pwallet */, MakeTransactionRef());
196  ssValue >> wtx;
197  CValidationState state;
198  if (!(CheckTransaction(*wtx.tx, state) && (wtx.GetHash() == hash) && state.IsValid()))
199  return false;
200 
201  // Undo serialize changes in 31600
202  if (31404 <= wtx.fTimeReceivedIsTxTime && wtx.fTimeReceivedIsTxTime <= 31703)
203  {
204  if (!ssValue.empty())
205  {
206  char fTmp;
207  char fUnused;
208  std::string unused_string;
209  ssValue >> fTmp >> fUnused >> unused_string;
210  strErr = strprintf("LoadWallet() upgrading tx ver=%d %d %s",
211  wtx.fTimeReceivedIsTxTime, fTmp, hash.ToString());
212  wtx.fTimeReceivedIsTxTime = fTmp;
213  }
214  else
215  {
216  strErr = strprintf("LoadWallet() repairing tx ver=%d %s", wtx.fTimeReceivedIsTxTime, hash.ToString());
217  wtx.fTimeReceivedIsTxTime = 0;
218  }
219  wss.vWalletUpgrade.push_back(hash);
220  }
221 
222  if (wtx.nOrderPos == -1)
223  wss.fAnyUnordered = true;
224 
225  pwallet->LoadToWallet(wtx);
226  }
227  else if (strType == "watchs")
228  {
229  wss.nWatchKeys++;
230  CScript script;
231  ssKey >> script;
232  char fYes;
233  ssValue >> fYes;
234  if (fYes == '1')
235  pwallet->LoadWatchOnly(script);
236  }
237  else if (strType == "key" || strType == "wkey")
238  {
239  CPubKey vchPubKey;
240  ssKey >> vchPubKey;
241  if (!vchPubKey.IsValid())
242  {
243  strErr = "Error reading wallet database: CPubKey corrupt";
244  return false;
245  }
246  CKey key;
247  CPrivKey pkey;
248  uint256 hash;
249 
250  if (strType == "key")
251  {
252  wss.nKeys++;
253  ssValue >> pkey;
254  } else {
255  CWalletKey wkey;
256  ssValue >> wkey;
257  pkey = wkey.vchPrivKey;
258  }
259 
260  // Old wallets store keys as "key" [pubkey] => [privkey]
261  // ... which was slow for wallets with lots of keys, because the public key is re-derived from the private key
262  // using EC operations as a checksum.
263  // Newer wallets store keys as "key"[pubkey] => [privkey][hash(pubkey,privkey)], which is much faster while
264  // remaining backwards-compatible.
265  try
266  {
267  ssValue >> hash;
268  }
269  catch (...) {}
270 
271  bool fSkipCheck = false;
272 
273  if (!hash.IsNull())
274  {
275  // hash pubkey/privkey to accelerate wallet load
276  std::vector<unsigned char> vchKey;
277  vchKey.reserve(vchPubKey.size() + pkey.size());
278  vchKey.insert(vchKey.end(), vchPubKey.begin(), vchPubKey.end());
279  vchKey.insert(vchKey.end(), pkey.begin(), pkey.end());
280 
281  if (Hash(vchKey.begin(), vchKey.end()) != hash)
282  {
283  strErr = "Error reading wallet database: CPubKey/CPrivKey corrupt";
284  return false;
285  }
286 
287  fSkipCheck = true;
288  }
289 
290  if (!key.Load(pkey, vchPubKey, fSkipCheck))
291  {
292  strErr = "Error reading wallet database: CPrivKey corrupt";
293  return false;
294  }
295  if (!pwallet->LoadKey(key, vchPubKey))
296  {
297  strErr = "Error reading wallet database: LoadKey failed";
298  return false;
299  }
300  }
301  else if (strType == "mkey")
302  {
303  unsigned int nID;
304  ssKey >> nID;
305  CMasterKey kMasterKey;
306  ssValue >> kMasterKey;
307  if(pwallet->mapMasterKeys.count(nID) != 0)
308  {
309  strErr = strprintf("Error reading wallet database: duplicate CMasterKey id %u", nID);
310  return false;
311  }
312  pwallet->mapMasterKeys[nID] = kMasterKey;
313  if (pwallet->nMasterKeyMaxID < nID)
314  pwallet->nMasterKeyMaxID = nID;
315  }
316  else if (strType == "ckey")
317  {
318  CPubKey vchPubKey;
319  ssKey >> vchPubKey;
320  if (!vchPubKey.IsValid())
321  {
322  strErr = "Error reading wallet database: CPubKey corrupt";
323  return false;
324  }
325  std::vector<unsigned char> vchPrivKey;
326  ssValue >> vchPrivKey;
327  wss.nCKeys++;
328 
329  if (!pwallet->LoadCryptedKey(vchPubKey, vchPrivKey))
330  {
331  strErr = "Error reading wallet database: LoadCryptedKey failed";
332  return false;
333  }
334  wss.fIsEncrypted = true;
335  }
336  else if (strType == "keymeta")
337  {
338  CPubKey vchPubKey;
339  ssKey >> vchPubKey;
340  CKeyMetadata keyMeta;
341  ssValue >> keyMeta;
342  wss.nKeyMeta++;
343  pwallet->LoadKeyMetadata(vchPubKey.GetID(), keyMeta);
344  }
345  else if (strType == "watchmeta")
346  {
347  CScript script;
348  ssKey >> script;
349  CKeyMetadata keyMeta;
350  ssValue >> keyMeta;
351  wss.nKeyMeta++;
352  pwallet->LoadScriptMetadata(CScriptID(script), keyMeta);
353  }
354  else if (strType == "defaultkey")
355  {
356  // We don't want or need the default key, but if there is one set,
357  // we want to make sure that it is valid so that we can detect corruption
358  CPubKey vchPubKey;
359  ssValue >> vchPubKey;
360  if (!vchPubKey.IsValid()) {
361  strErr = "Error reading wallet database: Default Key corrupt";
362  return false;
363  }
364  }
365  else if (strType == "pool")
366  {
367  int64_t nIndex;
368  ssKey >> nIndex;
369  CKeyPool keypool;
370  ssValue >> keypool;
371 
372  pwallet->LoadKeyPool(nIndex, keypool);
373  }
374  else if (strType == "version")
375  {
376  ssValue >> wss.nFileVersion;
377  if (wss.nFileVersion == 10300)
378  wss.nFileVersion = 300;
379  }
380  else if (strType == "cscript")
381  {
382  uint160 hash;
383  ssKey >> hash;
384  CScript script;
385  ssValue >> script;
386  if (!pwallet->LoadCScript(script))
387  {
388  strErr = "Error reading wallet database: LoadCScript failed";
389  return false;
390  }
391  }
392  else if (strType == "orderposnext")
393  {
394  ssValue >> pwallet->nOrderPosNext;
395  }
396  else if (strType == "destdata")
397  {
398  std::string strAddress, strKey, strValue;
399  ssKey >> strAddress;
400  ssKey >> strKey;
401  ssValue >> strValue;
402  pwallet->LoadDestData(DecodeDestination(strAddress), strKey, strValue);
403  }
404  else if (strType == "hdchain")
405  {
406  CHDChain chain;
407  ssValue >> chain;
408  pwallet->SetHDChain(chain, true);
409  } else if (strType == "flags") {
410  uint64_t flags;
411  ssValue >> flags;
412  if (!pwallet->SetWalletFlags(flags, true)) {
413  strErr = "Error reading wallet database: Unknown non-tolerable wallet flags found";
414  return false;
415  }
416  } else if (strType != "bestblock" && strType != "bestblock_nomerkle" &&
417  strType != "minversion" && strType != "acentry") {
418  wss.m_unknown_records++;
419  }
420  } catch (...)
421  {
422  return false;
423  }
424  return true;
425 }

433 DBErrors WalletBatch::LoadWallet(CWallet* pwallet)
434 {
435  CWalletScanState wss;
436  bool fNoncriticalErrors = false;
437  DBErrors result = DBErrors::LOAD_OK;
438 
439  LOCK(pwallet->cs_wallet);
440  try {
441  int nMinVersion = 0;
442  if (m_batch.Read((std::string)"minversion", nMinVersion))
443  {
444  if (nMinVersion > FEATURE_LATEST)
445  return DBErrors::TOO_NEW;
446  pwallet->LoadMinVersion(nMinVersion);
447  }
448 
449  // Get cursor
450  Dbc* pcursor = m_batch.GetCursor();
451  if (!pcursor)
452  {
453  pwallet->WalletLogPrintf("Error getting wallet database cursor\n");
454  return DBErrors::CORRUPT;
455  }
456 
457  while (true)
458  {
459  // Read next record
460  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
461  CDataStream ssValue(SER_DISK, CLIENT_VERSION);
462  int ret = m_batch.ReadAtCursor(pcursor, ssKey, ssValue);
463  if (ret == DB_NOTFOUND)
464  break;
465  else if (ret != 0)
466  {
467  pwallet->WalletLogPrintf("Error reading next record from wallet database\n");
468  return DBErrors::CORRUPT;
469  }
470 
471  // Try to be tolerant of single corrupt records:
472  std::string strType, strErr;
473  if (!ReadKeyValue(pwallet, ssKey, ssValue, wss, strType, strErr))
474  {
475  // losing keys is considered a catastrophic error, anything else
476  // we assume the user can live with:
477  if (IsKeyType(strType) || strType == "defaultkey") {
478  result = DBErrors::CORRUPT;
479  } else if(strType == "flags") {
480  // reading the wallet flags can only fail if unknown flags are present
481  result = DBErrors::TOO_NEW;
482  } else {
483  // Leave other errors alone, if we try to fix them we might make things worse.
484  fNoncriticalErrors = true; // ... but do warn the user there is something wrong.
485  if (strType == "tx")
486  // Rescan if there is a bad transaction record:
487  gArgs.SoftSetBoolArg("-rescan", true);
488  }
489  }
490  if (!strErr.empty())
491  pwallet->WalletLogPrintf("%s\n", strErr);
492  }
493  pcursor->close();
494  }
495  catch (const boost::thread_interrupted&) {
496  throw;
497  }
498  catch (...) {
499  result = DBErrors::CORRUPT;
500  }
501 
502  if (fNoncriticalErrors && result == DBErrors::LOAD_OK)
503  result = DBErrors::NONCRITICAL_ERROR;
504 
505  // Any wallet corruption at all: skip any rewriting or
506  // upgrading, we don't want to make it worse.
507  if (result != DBErrors::LOAD_OK)
508  return result;
509 
510  pwallet->WalletLogPrintf("nFileVersion = %d\n", wss.nFileVersion);
511 
512  pwallet->WalletLogPrintf("Keys: %u plaintext, %u encrypted, %u w/ metadata, %u total. Unknown wallet records: %u\n",
513  wss.nKeys, wss.nCKeys, wss.nKeyMeta, wss.nKeys + wss.nCKeys, wss.m_unknown_records);
514 
515  // nTimeFirstKey is only reliable if all keys have metadata
516  if ((wss.nKeys + wss.nCKeys + wss.nWatchKeys) != wss.nKeyMeta)
517  pwallet->UpdateTimeFirstKey(1);
518 
519  for (const uint256& hash : wss.vWalletUpgrade)
520  WriteTx(pwallet->mapWallet.at(hash));
521 
522  // Rewrite encrypted wallets of versions 0.4.0 and 0.5.0rc:
523  if (wss.fIsEncrypted && (wss.nFileVersion == 40000 || wss.nFileVersion == 50000))
524  return DBErrors::NEED_REWRITE;
525 
526  if (wss.nFileVersion < CLIENT_VERSION) // Update
527  WriteVersion(CLIENT_VERSION);
528 
529  if (wss.fAnyUnordered)
530  result = pwallet->ReorderTransactions();
531 
532  return result;
533 }

624  }

629  return DBErrors::LOAD_OK;
