97bool WalletInit::ParameterInteraction() const
98{
99#ifdef USE_BDB
100 if (!BerkeleyDatabaseSanityCheck()) {
101 return InitError(Untranslated("A version conflict was detected between the run-time BerkeleyDB library and the one used during compilation."));
102 }
103#endif
104 if (gArgs.GetBoolArg("-disablewallet", DEFAULT_DISABLE_WALLET)) {
105 for (const std::string& wallet : gArgs.GetArgs("-wallet")) {
106 LogPrintf("%s: parameter interaction: -disablewallet -> ignoring -wallet=%s\n", __func__, wallet);
107 }
108
109 return true;
110 }
111
112 if (gArgs.GetBoolArg("-blocksonly", DEFAULT_BLOCKSONLY) && gArgs.SoftSetBoolArg("-walletbroadcast", false)) {
113 LogPrintf("%s: parameter interaction: -blocksonly=1 -> setting -walletbroadcast=0\n", __func__);
114 }
115
116 if (gArgs.IsArgSet("-zapwallettxes")) {
117 return InitError(Untranslated("-zapwallettxes has been removed. If you are attempting to remove a stuck transaction from your wallet, please use abandontransaction instead."));
118 }
119
120 if (gArgs.GetBoolArg("-sysperms", false))
121 return InitError(Untranslated("-sysperms is not allowed in combination with enabled wallet functionality"));
122
123 return true;
124}

126void WalletInit::Construct(NodeContext& node) const
127{
128 ArgsManager& args = *Assert(node.args);
129 if (args.GetBoolArg("-disablewallet", DEFAULT_DISABLE_WALLET)) {
130 LogPrintf("Wallet disabled!\n");
131 return;
132 }
133 auto wallet_client = node.init->makeWalletClient(*node.chain);
134 node.wallet_client = wallet_client.get();
135 node.chain_clients.emplace_back(std::move(wallet_client));
136}
