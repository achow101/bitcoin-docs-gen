76 bool WalletInit::ParameterInteraction() const
77 {
78  if (gArgs.GetBoolArg("-disablewallet", DEFAULT_DISABLE_WALLET)) {
79  for (const std::string& wallet : gArgs.GetArgs("-wallet")) {
80  LogPrintf("%s: parameter interaction: -disablewallet -> ignoring -wallet=%s\n", __func__, wallet);
81  }
82 
83  return true;
84  }
85 
86  const bool is_multiwallet = gArgs.GetArgs("-wallet").size() > 1;
87 
88  if (gArgs.GetBoolArg("-blocksonly", DEFAULT_BLOCKSONLY) && gArgs.SoftSetBoolArg("-walletbroadcast", false)) {
89  LogPrintf("%s: parameter interaction: -blocksonly=1 -> setting -walletbroadcast=0\n", __func__);
90  }
91 
92  if (gArgs.GetBoolArg("-salvagewallet", false)) {
93  if (is_multiwallet) {
94  return InitError(strprintf("%s is only allowed with a single wallet file", "-salvagewallet"));
95  }
96  // Rewrite just private keys: rescan to find transactions
97  if (gArgs.SoftSetBoolArg("-rescan", true)) {
98  LogPrintf("%s: parameter interaction: -salvagewallet=1 -> setting -rescan=1\n", __func__);
99  }
100  }
101 
102  bool zapwallettxes = gArgs.GetBoolArg("-zapwallettxes", false);
103  // -zapwallettxes implies dropping the mempool on startup
104  if (zapwallettxes && gArgs.SoftSetBoolArg("-persistmempool", false)) {
105  LogPrintf("%s: parameter interaction: -zapwallettxes enabled -> setting -persistmempool=0\n", __func__);
106  }
107 
108  // -zapwallettxes implies a rescan
109  if (zapwallettxes) {
110  if (is_multiwallet) {
111  return InitError(strprintf("%s is only allowed with a single wallet file", "-zapwallettxes"));
112  }
113  if (gArgs.SoftSetBoolArg("-rescan", true)) {
114  LogPrintf("%s: parameter interaction: -zapwallettxes enabled -> setting -rescan=1\n", __func__);
115  }
116  }
117 
118  if (is_multiwallet) {
119  if (gArgs.GetBoolArg("-upgradewallet", false)) {
120  return InitError(strprintf("%s is only allowed with a single wallet file", "-upgradewallet"));
121  }
122  }
123 
124  if (gArgs.GetBoolArg("-sysperms", false))
125  return InitError("-sysperms is not allowed in combination with enabled wallet functionality");
126  if (gArgs.GetArg("-prune", 0) && gArgs.GetBoolArg("-rescan", false))
127  return InitError(_("Rescans are not possible in pruned mode. You will need to use -reindex which will download the whole blockchain again."));
128
