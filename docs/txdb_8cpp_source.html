83 bool CCoinsViewDB::BatchWrite(CCoinsMap &mapCoins, const uint256 &hashBlock) {
84  CDBBatch batch(db);
85  size_t count = 0;
86  size_t changed = 0;
87  size_t batch_size = (size_t)gArgs.GetArg("-dbbatchsize", nDefaultDbBatchSize);
88  int crash_simulate = gArgs.GetArg("-dbcrashratio", 0);
89  assert(!hashBlock.IsNull());
90 
91  uint256 old_tip = GetBestBlock();
92  if (old_tip.IsNull()) {
93  // We may be in the middle of replaying.
94  std::vector<uint256> old_heads = GetHeadBlocks();
95  if (old_heads.size() == 2) {
96  assert(old_heads[0] == hashBlock);
97  old_tip = old_heads[1];
98  }
99  }
100 
101  // In the first batch, mark the database as being in the middle of a
102  // transition from old_tip to hashBlock.
103  // A vector is used for future extensibility, as we may want to support
104  // interrupting after partial writes from multiple independent reorgs.
105  batch.Erase(DB_BEST_BLOCK);
106  batch.Write(DB_HEAD_BLOCKS, std::vector<uint256>{hashBlock, old_tip});
107 
108  for (CCoinsMap::iterator it = mapCoins.begin(); it != mapCoins.end();) {
109  if (it->second.flags & CCoinsCacheEntry::DIRTY) {
110  CoinEntry entry(&it->first);
111  if (it->second.coin.IsSpent())
112  batch.Erase(entry);
113  else
114  batch.Write(entry, it->second.coin);
115  changed++;
116  }
117  count++;
118  CCoinsMap::iterator itOld = it++;
119  mapCoins.erase(itOld);
120  if (batch.SizeEstimate() > batch_size) {
121  LogPrint(BCLog::COINDB, "Writing partial batch of %.2f MiB\n", batch.SizeEstimate() * (1.0 / 1048576.0));
122  db.WriteBatch(batch);
123  batch.Clear();
124  if (crash_simulate) {
125  static FastRandomContext rng;
126  if (rng.randrange(crash_simulate) == 0) {
127  LogPrintf("Simulating a crash. Goodbye.\n");
128  _Exit(0);
129  }
130  }
131  }
132  }
133 
134  // In the last batch, mark the database as consistent with hashBlock again.
135  batch.Erase(DB_HEAD_BLOCKS);
136  batch.Write(DB_BEST_BLOCK, hashBlock);
137 
138  LogPrint(BCLog::COINDB, "Writing final batch of %.2f MiB\n", batch.SizeEstimate() * (1.0 / 1048576.0));
139  bool ret = db.WriteBatch(batch);
140  LogPrint(BCLog::COINDB, "Committed %u changed transaction outputs (out of %u) to coin database...\n", (unsigned int)changed, (unsigned int)count);
141  return ret;
142 }

311  template<typename Stream>
312  void Unserialize(Stream &s) {
313  unsigned int nCode = 0;
314  // version
315  unsigned int nVersionDummy;
316  ::Unserialize(s, VARINT(nVersionDummy));
317  // header code
318  ::Unserialize(s, VARINT(nCode));
319  fCoinBase = nCode & 1;
320  std::vector<bool> vAvail(2, false);
321  vAvail[0] = (nCode & 2) != 0;
322  vAvail[1] = (nCode & 4) != 0;
323  unsigned int nMaskCode = (nCode / 8) + ((nCode & 6) != 0 ? 0 : 1);
324  // spentness bitmask
325  while (nMaskCode > 0) {
326  unsigned char chAvail = 0;
327  ::Unserialize(s, chAvail);
328  for (unsigned int p = 0; p < 8; p++) {
329  bool f = (chAvail & (1 << p)) != 0;
330  vAvail.push_back(f);
331  }
332  if (chAvail != 0)
333  nMaskCode--;
334  }
335  // txouts themself
336  vout.assign(vAvail.size(), CTxOut());
337  for (unsigned int i = 0; i < vAvail.size(); i++) {
338  if (vAvail[i])
339  ::Unserialize(s, CTxOutCompressor(vout[i]));
340  }
341  // coinbase height
342  ::Unserialize(s, VARINT(nHeight, VarIntMode::NONNEGATIVE_SIGNED));
343  }
344 };

352 bool CCoinsViewDB::Upgrade() {
353  std::unique_ptr<CDBIterator> pcursor(db.NewIterator());
354  pcursor->Seek(std::make_pair(DB_COINS, uint256()));
355  if (!pcursor->Valid()) {
356  return true;
357  }
358 
359  int64_t count = 0;
360  LogPrintf("Upgrading utxo-set database...\n");
361  LogPrintf("[0%%]..."); /* Continued */
362  uiInterface.ShowProgress(_("Upgrading UTXO database"), 0, true);
363  size_t batch_size = 1 << 24;
364  CDBBatch batch(db);
365  int reportDone = 0;
366  std::pair<unsigned char, uint256> key;
367  std::pair<unsigned char, uint256> prev_key = {DB_COINS, uint256()};
368  while (pcursor->Valid()) {
369  boost::this_thread::interruption_point();
370  if (ShutdownRequested()) {
371  break;
372  }
373  if (pcursor->GetKey(key) && key.first == DB_COINS) {
374  if (count++ % 256 == 0) {
375  uint32_t high = 0x100 * *key.second.begin() + *(key.second.begin() + 1);
376  int percentageDone = (int)(high * 100.0 / 65536.0 + 0.5);
377  uiInterface.ShowProgress(_("Upgrading UTXO database"), percentageDone, true);
378  if (reportDone < percentageDone/10) {
379  // report max. every 10% step
380  LogPrintf("[%d%%]...", percentageDone); /* Continued */
381  reportDone = percentageDone/10;
382  }
383  }
384  CCoins old_coins;
385  if (!pcursor->GetValue(old_coins)) {
386  return error("%s: cannot parse CCoins record", __func__);
387  }
388  COutPoint outpoint(key.second, 0);
389  for (size_t i = 0; i < old_coins.vout.size(); ++i) {
390  if (!old_coins.vout[i].IsNull() && !old_coins.vout[i].scriptPubKey.IsUnspendable()) {
391  Coin newcoin(std::move(old_coins.vout[i]), old_coins.nHeight, old_coins.fCoinBase);
392  outpoint.n = i;
393  CoinEntry entry(&outpoint);
394  batch.Write(entry, newcoin);
395  }
396  }
397  batch.Erase(key);
398  if (batch.SizeEstimate() > batch_size) {
399  db.WriteBatch(batch);
400  batch.Clear();
401  db.CompactRange(prev_key, key);
402  prev_key = key;
403  }
404  pcursor->Next();
405  } else {
406  break;
407  }
408  }
409  db.WriteBatch(batch);
410  db.CompactRange({DB_COINS, uint256()}, key);
411  uiInterface.ShowProgress("", 100, false);
412  LogPrintf("[%s].\n", ShutdownRequested() ? "CANCELLED" : "DONE");
413  return !ShutdownRequested();
414 }
