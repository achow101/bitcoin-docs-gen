290  bool ExpandHelper(int pos, const SigningProvider& arg, Span<const unsigned char>* cache_read, std::vector<CScript>& output_scripts, FlatSigningProvider& out, std::vector<unsigned char>* cache_write) const
291  {
292  std::vector<std::pair<CPubKey, KeyOriginInfo>> entries;
293  entries.reserve(m_pubkey_args.size());
294 
295  // Construct temporary data in `entries` and `subscripts`, to avoid producing output in case of failure.
296  for (const auto& p : m_pubkey_args) {
297  entries.emplace_back();
298  if (!p->GetPubKey(pos, arg, cache_read ? nullptr : &entries.back().first, entries.back().second)) return false;
299  if (cache_read) {
300  // Cached expanded public key exists, use it.
301  if (cache_read->size() == 0) return false;
302  bool compressed = ((*cache_read)[0] == 0x02 || (*cache_read)[0] == 0x03) && cache_read->size() >= 33;
303  bool uncompressed = ((*cache_read)[0] == 0x04) && cache_read->size() >= 65;
304  if (!(compressed || uncompressed)) return false;
305  CPubKey pubkey(cache_read->begin(), cache_read->begin() + (compressed ? 33 : 65));
306  entries.back().first = pubkey;
307  *cache_read = cache_read->subspan(compressed ? 33 : 65);
308  }
309  if (cache_write) {
310  cache_write->insert(cache_write->end(), entries.back().first.begin(), entries.back().first.end());
311  }
312  }
313  std::vector<CScript> subscripts;
314  if (m_script_arg) {
315  FlatSigningProvider subprovider;
316  if (!m_script_arg->ExpandHelper(pos, arg, cache_read, subscripts, subprovider, cache_write)) return false;
317  out = Merge(out, subprovider);
318  }
319 
320  std::vector<CPubKey> pubkeys;
321  pubkeys.reserve(entries.size());
322  for (auto& entry : entries) {
323  pubkeys.push_back(entry.first);
324  out.origins.emplace(entry.first.GetID(), std::move(entry.second));
325  }
326  if (m_script_arg) {
327  for (const auto& subscript : subscripts) {
328  out.scripts.emplace(CScriptID(subscript), subscript);
329  std::vector<CScript> addscripts = MakeScripts(pubkeys, &subscript, out);
330  for (auto& addscript : addscripts) {
331  output_scripts.push_back(std::move(addscript));
332  }
333  }
334  } else {
335  output_scripts = MakeScripts(pubkeys, nullptr, out);
336  }
337  return true;
338  }
339

542 NODISCARD bool ParseKeyPath(const std::vector<Span<const char>>& split, KeyPath& out)
543 {
544  for (size_t i = 1; i < split.size(); ++i) {
545  Span<const char> elem = split[i];
546  bool hardened = false;
547  if (elem.size() > 0 && (elem[elem.size() - 1] == '\'' || elem[elem.size() - 1] == 'h')) {
548  elem = elem.first(elem.size() - 1);
549  hardened = true;
550  }
551  uint32_t p;
552  if (!ParseUInt32(std::string(elem.begin(), elem.end()), &p) || p > 0x7FFFFFFFUL) return false;
553  out.push_back(p | (((uint32_t)hardened) << 31));
554  }
555  return true;
556 }

559 std::unique_ptr<PubkeyProvider> ParsePubkeyInner(const Span<const char>& sp, bool permit_uncompressed, FlatSigningProvider& out)
560 {
561  auto split = Split(sp, '/');
562  std::string str(split[0].begin(), split[0].end());
563  if (split.size() == 1) {
564  if (IsHex(str)) {
565  std::vector<unsigned char> data = ParseHex(str);
566  CPubKey pubkey(data);
567  if (pubkey.IsFullyValid() && (permit_uncompressed || pubkey.IsCompressed())) return MakeUnique<ConstPubkeyProvider>(pubkey);
568  }
569  CKey key = DecodeSecret(str);
570  if (key.IsValid() && (permit_uncompressed || key.IsCompressed())) {
571  CPubKey pubkey = key.GetPubKey();
572  out.keys.emplace(pubkey.GetID(), key);
573  return MakeUnique<ConstPubkeyProvider>(pubkey);
574  }
575  }
576  CExtKey extkey = DecodeExtKey(str);
577  CExtPubKey extpubkey = DecodeExtPubKey(str);
578  if (!extkey.key.IsValid() && !extpubkey.pubkey.IsValid()) return nullptr;
579  KeyPath path;
580  DeriveType type = DeriveType::NO;
581  if (split.back() == MakeSpan("*").first(1)) {
582  split.pop_back();
583  type = DeriveType::UNHARDENED;
584  } else if (split.back() == MakeSpan("*'").first(2) || split.back() == MakeSpan("*h").first(2)) {
585  split.pop_back();
586  type = DeriveType::HARDENED;
587  }
588  if (!ParseKeyPath(split, path)) return nullptr;
589  if (extkey.key.IsValid()) {
590  extpubkey = extkey.Neuter();
591  out.keys.emplace(extpubkey.pubkey.GetID(), extkey.key);
592  }
593  return MakeUnique<BIP32PubkeyProvider>(extpubkey, std::move(path), type);
594 }

597 std::unique_ptr<PubkeyProvider> ParsePubkey(const Span<const char>& sp, bool permit_uncompressed, FlatSigningProvider& out)
598 {
599  auto origin_split = Split(sp, ']');
600  if (origin_split.size() > 2) return nullptr;
601  if (origin_split.size() == 1) return ParsePubkeyInner(origin_split[0], permit_uncompressed, out);
602  if (origin_split[0].size() < 1 || origin_split[0][0] != '[') return nullptr;
603  auto slash_split = Split(origin_split[0].subspan(1), '/');
604  if (slash_split[0].size() != 8) return nullptr;
605  std::string fpr_hex = std::string(slash_split[0].begin(), slash_split[0].end());
606  if (!IsHex(fpr_hex)) return nullptr;
607  auto fpr_bytes = ParseHex(fpr_hex);
608  KeyOriginInfo info;
609  static_assert(sizeof(info.fingerprint) == 4, "Fingerprint must be 4 bytes");
610  assert(fpr_bytes.size() == 4);
611  std::copy(fpr_bytes.begin(), fpr_bytes.end(), info.fingerprint);
612  if (!ParseKeyPath(slash_split, info.path)) return nullptr;
613  auto provider = ParsePubkeyInner(origin_split[1], permit_uncompressed, out);
614  if (!provider) return nullptr;
615  return MakeUnique<OriginPubkeyProvider>(std::move(info), std::move(provider));

619 std::unique_ptr<DescriptorImpl> ParseScript(Span<const char>& sp, ParseScriptContext ctx, FlatSigningProvider& out)
620 {
621  auto expr = Expr(sp);
622  if (Func("pk", expr)) {
623  auto pubkey = ParsePubkey(expr, ctx != ParseScriptContext::P2WSH, out);
624  if (!pubkey) return nullptr;
625  return MakeUnique<PKDescriptor>(std::move(pubkey));
626  }
627  if (Func("pkh", expr)) {
628  auto pubkey = ParsePubkey(expr, ctx != ParseScriptContext::P2WSH, out);
629  if (!pubkey) return nullptr;
630  return MakeUnique<PKHDescriptor>(std::move(pubkey));
631  }
632  if (ctx == ParseScriptContext::TOP && Func("combo", expr)) {
633  auto pubkey = ParsePubkey(expr, true, out);
634  if (!pubkey) return nullptr;
635  return MakeUnique<ComboDescriptor>(std::move(pubkey));
636  }
637  if (Func("multi", expr)) {
638  auto threshold = Expr(expr);
639  uint32_t thres;
640  std::vector<std::unique_ptr<PubkeyProvider>> providers;
641  if (!ParseUInt32(std::string(threshold.begin(), threshold.end()), &thres)) return nullptr;
642  size_t script_size = 0;
643  while (expr.size()) {
644  if (!Const(",", expr)) return nullptr;
645  auto arg = Expr(expr);
646  auto pk = ParsePubkey(arg, ctx != ParseScriptContext::P2WSH, out);
647  if (!pk) return nullptr;
648  script_size += pk->GetSize() + 1;
649  providers.emplace_back(std::move(pk));
650  }
651  if (providers.size() < 1 || providers.size() > 16 || thres < 1 || thres > providers.size()) return nullptr;
652  if (ctx == ParseScriptContext::TOP) {
653  if (providers.size() > 3) return nullptr; // Not more than 3 pubkeys for raw multisig
654  }
655  if (ctx == ParseScriptContext::P2SH) {
656  if (script_size + 3 > 520) return nullptr; // Enforce P2SH script size limit
657  }
658  return MakeUnique<MultisigDescriptor>(thres, std::move(providers));
659  }
660  if (ctx != ParseScriptContext::P2WSH && Func("wpkh", expr)) {
661  auto pubkey = ParsePubkey(expr, false, out);
662  if (!pubkey) return nullptr;
663  return MakeUnique<WPKHDescriptor>(std::move(pubkey));
664  }
665  if (ctx == ParseScriptContext::TOP && Func("sh", expr)) {
666  auto desc = ParseScript(expr, ParseScriptContext::P2SH, out);
667  if (!desc || expr.size()) return nullptr;
668  return MakeUnique<SHDescriptor>(std::move(desc));
669  }
670  if (ctx != ParseScriptContext::P2WSH && Func("wsh", expr)) {
671  auto desc = ParseScript(expr, ParseScriptContext::P2WSH, out);
672  if (!desc || expr.size()) return nullptr;
673  return MakeUnique<WSHDescriptor>(std::move(desc));
674  }
675  if (ctx == ParseScriptContext::TOP && Func("addr", expr)) {
676  CTxDestination dest = DecodeDestination(std::string(expr.begin(), expr.end()));
677  if (!IsValidDestination(dest)) return nullptr;
678  return MakeUnique<AddressDescriptor>(std::move(dest));
679  }
680  if (ctx == ParseScriptContext::TOP && Func("raw", expr)) {
681  std::string str(expr.begin(), expr.end());
682  if (!IsHex(str)) return nullptr;
683  auto bytes = ParseHex(str);
684  return MakeUnique<RawDescriptor>(CScript(bytes.begin(), bytes.end()));
685  }
686  return nullptr;
687 }

699 std::unique_ptr<DescriptorImpl> InferScript(const CScript& script, ParseScriptContext ctx, const SigningProvider& provider)
700 {
701  std::vector<std::vector<unsigned char>> data;
702  txnouttype txntype = Solver(script, data);
703 
704  if (txntype == TX_PUBKEY) {
705  CPubKey pubkey(data[0].begin(), data[0].end());
706  if (pubkey.IsValid()) {
707  return MakeUnique<PKDescriptor>(InferPubkey(pubkey, ctx, provider));
708  }
709  }
710  if (txntype == TX_PUBKEYHASH) {
711  uint160 hash(data[0]);
712  CKeyID keyid(hash);
713  CPubKey pubkey;
714  if (provider.GetPubKey(keyid, pubkey)) {
715  return MakeUnique<PKHDescriptor>(InferPubkey(pubkey, ctx, provider));
716  }
717  }
718  if (txntype == TX_WITNESS_V0_KEYHASH && ctx != ParseScriptContext::P2WSH) {
719  uint160 hash(data[0]);
720  CKeyID keyid(hash);
721  CPubKey pubkey;
722  if (provider.GetPubKey(keyid, pubkey)) {
723  return MakeUnique<WPKHDescriptor>(InferPubkey(pubkey, ctx, provider));
724  }
725  }
726  if (txntype == TX_MULTISIG) {
727  std::vector<std::unique_ptr<PubkeyProvider>> providers;
728  for (size_t i = 1; i + 1 < data.size(); ++i) {
729  CPubKey pubkey(data[i].begin(), data[i].end());
730  providers.push_back(InferPubkey(pubkey, ctx, provider));
731  }
732  return MakeUnique<MultisigDescriptor>((int)data[0][0], std::move(providers));
733  }
734  if (txntype == TX_SCRIPTHASH && ctx == ParseScriptContext::TOP) {
735  uint160 hash(data[0]);
736  CScriptID scriptid(hash);
737  CScript subscript;
738  if (provider.GetCScript(scriptid, subscript)) {
739  auto sub = InferScript(subscript, ParseScriptContext::P2SH, provider);
740  if (sub) return MakeUnique<SHDescriptor>(std::move(sub));
741  }
742  }
743  if (txntype == TX_WITNESS_V0_SCRIPTHASH && ctx != ParseScriptContext::P2WSH) {
744  CScriptID scriptid;
745  CRIPEMD160().Write(data[0].data(), data[0].size()).Finalize(scriptid.begin());
746  CScript subscript;
747  if (provider.GetCScript(scriptid, subscript)) {
748  auto sub = InferScript(subscript, ParseScriptContext::P2WSH, provider);
749  if (sub) return MakeUnique<WSHDescriptor>(std::move(sub));
750  }
751  }
752 
753  CTxDestination dest;
754  if (ExtractDestination(script, dest)) {
755  if (GetScriptForDestination(dest) == script) {
756  return MakeUnique<AddressDescriptor>(std::move(dest));
757  }
758  }
759 
760  return MakeUnique<RawDescriptor>(script);
761 }
