78  template <typename Stream, typename Operation>
79  inline void SerializationOp(Stream& s, Operation ser_action) {
80  std::string sAddress = address.toStdString();
81  std::string sLabel = label.toStdString();
82  std::string sMessage = message.toStdString();
83  std::string sPaymentRequest;
84  if (!ser_action.ForRead() && paymentRequest.IsInitialized())
85  paymentRequest.SerializeToString(&sPaymentRequest);
86  std::string sAuthenticatedMerchant = authenticatedMerchant.toStdString();
87 
88  READWRITE(this->nVersion);
89  READWRITE(sAddress);
90  READWRITE(sLabel);
91  READWRITE(amount);
92  READWRITE(sMessage);
93  READWRITE(sPaymentRequest);
94  READWRITE(sAuthenticatedMerchant);
95 
96  if (ser_action.ForRead())
97  {
98  address = QString::fromStdString(sAddress);
99  label = QString::fromStdString(sLabel);
100  message = QString::fromStdString(sMessage);
101  if (!sPaymentRequest.empty())
102  paymentRequest.parse(QByteArray::fromRawData(sPaymentRequest.data(), sPaymentRequest.size()));
103  authenticatedMerchant = QString::fromStdString(sAuthenticatedMerchant);
104  }
105  }
