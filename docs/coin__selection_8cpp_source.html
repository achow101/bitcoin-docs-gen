22 // Simple benchmark for wallet coin selection. Note that it maybe be necessary
23 // to build up more complicated scenarios in order to get meaningful
24 // measurements of performance. From laanwj, "Wallet coin selection is probably
25 // the hardest, as you need a wider selection of scenarios, just testing the
26 // same one over and over isn't too useful. Generating random isn't useful
27 // either for measurements."
28 // (https://github.com/bitcoin/bitcoin/issues/7883#issuecomment-224807484)
29 static void CoinSelection(benchmark::State& state)
30 {
31  auto chain = interfaces::MakeChain();
32  const CWallet wallet(*chain, WalletLocation(), WalletDatabase::CreateDummy());
33  std::vector<std::unique_ptr<CWalletTx>> wtxs;
34  LOCK(wallet.cs_wallet);
35 
36  // Add coins.
37  for (int i = 0; i < 1000; ++i) {
38  addCoin(1000 * COIN, wallet, wtxs);
39  }
40  addCoin(3 * COIN, wallet, wtxs);
41 
42  // Create groups
43  std::vector<OutputGroup> groups;
44  for (const auto& wtx : wtxs) {
45  COutput output(wtx.get(), 0 /* iIn */, 6 * 24 /* nDepthIn */, true /* spendable */, true /* solvable */, true /* safe */);
46  groups.emplace_back(output.GetInputCoin(), 6, false, 0, 0);
47  }
48 
49  const CoinEligibilityFilter filter_standard(1, 6, 0);
50  const CoinSelectionParams coin_selection_params(true, 34, 148, CFeeRate(0), 0);
51  while (state.KeepRunning()) {
52  std::set<CInputCoin> setCoinsRet;
53  CAmount nValueRet;
54  bool bnb_used;
55  bool success = wallet.SelectCoinsMinConf(1003 * COIN, filter_standard, groups, setCoinsRet, nValueRet, coin_selection_params, bnb_used);
56  assert(success);

58  assert(setCoinsRet.size() == 2);
59  }
60 }

78 static CAmount make_hard_case(int utxos, std::vector<OutputGroup>& utxo_pool)
79 {
80  utxo_pool.clear();
81  CAmount target = 0;
82  for (int i = 0; i < utxos; ++i) {
83  target += (CAmount)1 << (utxos+i);
84  add_coin((CAmount)1 << (utxos+i), 2*i, utxo_pool);
85  add_coin(((CAmount)1 << (utxos+i)) + ((CAmount)1 << (utxos-1-i)), 2*i + 1, utxo_pool);
86  }
87  return target;
