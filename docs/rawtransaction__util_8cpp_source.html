22 CMutableTransaction ConstructTransaction(const UniValue& inputs_in, const UniValue& outputs_in, const UniValue& locktime, const UniValue& rbf)
23 {
24  if (inputs_in.isNull() || outputs_in.isNull())
25  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, arguments 1 and 2 must be non-null");
26 
27  UniValue inputs = inputs_in.get_array();
28  const bool outputs_is_obj = outputs_in.isObject();
29  UniValue outputs = outputs_is_obj ? outputs_in.get_obj() : outputs_in.get_array();
30 
31  CMutableTransaction rawTx;
32 
33  if (!locktime.isNull()) {
34  int64_t nLockTime = locktime.get_int64();
35  if (nLockTime < 0 || nLockTime > LOCKTIME_MAX)
36  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, locktime out of range");
37  rawTx.nLockTime = nLockTime;
38  }
39 
40  bool rbfOptIn = rbf.isTrue();
41 
42  for (unsigned int idx = 0; idx < inputs.size(); idx++) {
43  const UniValue& input = inputs[idx];
44  const UniValue& o = input.get_obj();
45 
46  uint256 txid = ParseHashO(o, "txid");
47 
48  const UniValue& vout_v = find_value(o, "vout");
49  if (!vout_v.isNum())
50  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, missing vout key");
51  int nOutput = vout_v.get_int();
52  if (nOutput < 0)
53  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, vout must be positive");
54 
55  uint32_t nSequence;
56  if (rbfOptIn) {
57  nSequence = MAX_BIP125_RBF_SEQUENCE; /* CTxIn::SEQUENCE_FINAL - 2 */
58  } else if (rawTx.nLockTime) {
59  nSequence = CTxIn::SEQUENCE_FINAL - 1;
60  } else {
61  nSequence = CTxIn::SEQUENCE_FINAL;
62  }
63 
64  // set the sequence number if passed in the parameters object
65  const UniValue& sequenceObj = find_value(o, "sequence");
66  if (sequenceObj.isNum()) {
67  int64_t seqNr64 = sequenceObj.get_int64();
68  if (seqNr64 < 0 || seqNr64 > CTxIn::SEQUENCE_FINAL) {
69  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, sequence number is out of range");
70  } else {
71  nSequence = (uint32_t)seqNr64;
72  }
73  }
74 
75  CTxIn in(COutPoint(txid, nOutput), CScript(), nSequence);
76 
77  rawTx.vin.push_back(in);
78  }
79 
80  if (!outputs_is_obj) {
81  // Translate array of key-value pairs into dict
82  UniValue outputs_dict = UniValue(UniValue::VOBJ);
83  for (size_t i = 0; i < outputs.size(); ++i) {
84  const UniValue& output = outputs[i];
85  if (!output.isObject()) {
86  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, key-value pair not an object as expected");
87  }
88  if (output.size() != 1) {
89  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, key-value pair must contain exactly one key");
90  }
91  outputs_dict.pushKVs(output);
92  }
93  outputs = std::move(outputs_dict);
94  }
95 
96  // Duplicate checking
97  std::set<CTxDestination> destinations;
98  bool has_data{false};
99 
100  for (const std::string& name_ : outputs.getKeys()) {
101  if (name_ == "data") {
102  if (has_data) {
103  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter, duplicate key: data");
104  }
105  has_data = true;
106  std::vector<unsigned char> data = ParseHexV(outputs[name_].getValStr(), "Data");
107 
108  CTxOut out(0, CScript() << OP_RETURN << data);
109  rawTx.vout.push_back(out);
110  } else {
111  CTxDestination destination = DecodeDestination(name_);
112  if (!IsValidDestination(destination)) {
113  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, std::string("Invalid Bitcoin address: ") + name_);
114  }
115 
116  if (!destinations.insert(destination).second) {
117  throw JSONRPCError(RPC_INVALID_PARAMETER, std::string("Invalid parameter, duplicated address: ") + name_);
118  }
119 
120  CScript scriptPubKey = GetScriptForDestination(destination);
121  CAmount nAmount = AmountFromValue(outputs[name_]);
122 
123  CTxOut out(nAmount, scriptPubKey);
124  rawTx.vout.push_back(out);
125  }
126  }
127 
128  if (!rbf.isNull() && rawTx.vin.size() > 0 && rbfOptIn != SignalsOptInRBF(CTransaction(rawTx))) {
129  throw JSONRPCError(RPC_INVALID_PARAMETER, "Invalid parameter combination: Sequence number(s) contradict replaceable option");
130  }
131 
132  return rawTx;
133 }

136 static void TxInErrorToJSON(const CTxIn& txin, UniValue& vErrorsRet, const std::string& strMessage)
137 {
138  UniValue entry(UniValue::VOBJ);
139  entry.pushKV("txid", txin.prevout.hash.ToString());
140  entry.pushKV("vout", (uint64_t)txin.prevout.n);
141  UniValue witness(UniValue::VARR);
142  for (unsigned int i = 0; i < txin.scriptWitness.stack.size(); i++) {
143  witness.push_back(HexStr(txin.scriptWitness.stack[i].begin(), txin.scriptWitness.stack[i].end()));
144  }
145  entry.pushKV("witness", witness);
146  entry.pushKV("scriptSig", HexStr(txin.scriptSig.begin(), txin.scriptSig.end()));
147  entry.pushKV("sequence", (uint64_t)txin.nSequence);
148  entry.pushKV("error", strMessage);
149  vErrorsRet.push_back(entry);
150 }

152 UniValue SignTransaction(CMutableTransaction& mtx, const UniValue& prevTxsUnival, CBasicKeyStore* keystore, std::map<COutPoint, Coin>& coins, bool is_temp_keystore, const UniValue& hashType)
153 {
154  // Add previous txouts given in the RPC call:
155  if (!prevTxsUnival.isNull()) {
156  UniValue prevTxs = prevTxsUnival.get_array();
157  for (unsigned int idx = 0; idx < prevTxs.size(); ++idx) {
158  const UniValue& p = prevTxs[idx];
159  if (!p.isObject()) {
160  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "expected object with {\"txid'\",\"vout\",\"scriptPubKey\"}");
161  }
162 
163  UniValue prevOut = p.get_obj();
164 
165  RPCTypeCheckObj(prevOut,
166  {
167  {"txid", UniValueType(UniValue::VSTR)},
168  {"vout", UniValueType(UniValue::VNUM)},
169  {"scriptPubKey", UniValueType(UniValue::VSTR)},
170  });
171 
172  uint256 txid = ParseHashO(prevOut, "txid");
173 
174  int nOut = find_value(prevOut, "vout").get_int();
175  if (nOut < 0) {
176  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, "vout must be positive");
177  }
178 
179  COutPoint out(txid, nOut);
180  std::vector<unsigned char> pkData(ParseHexO(prevOut, "scriptPubKey"));
181  CScript scriptPubKey(pkData.begin(), pkData.end());
182 
183  {
184  auto coin = coins.find(out);
185  if (coin != coins.end() && !coin->second.IsSpent() && coin->second.out.scriptPubKey != scriptPubKey) {
186  std::string err("Previous output scriptPubKey mismatch:\n");
187  err = err + ScriptToAsmStr(coin->second.out.scriptPubKey) + "\nvs:\n"+
188  ScriptToAsmStr(scriptPubKey);
189  throw JSONRPCError(RPC_DESERIALIZATION_ERROR, err);
190  }
191  Coin newcoin;
192  newcoin.out.scriptPubKey = scriptPubKey;
193  newcoin.out.nValue = MAX_MONEY;
194  if (prevOut.exists("amount")) {
195  newcoin.out.nValue = AmountFromValue(find_value(prevOut, "amount"));
196  }
197  newcoin.nHeight = 1;
198  coins[out] = std::move(newcoin);
199  }
200 
201  // if redeemScript and private keys were given, add redeemScript to the keystore so it can be signed
202  if (is_temp_keystore && (scriptPubKey.IsPayToScriptHash() || scriptPubKey.IsPayToWitnessScriptHash())) {
203  RPCTypeCheckObj(prevOut,
204  {
205  {"redeemScript", UniValueType(UniValue::VSTR)},
206  {"witnessScript", UniValueType(UniValue::VSTR)},
207  }, true);
208  UniValue rs = find_value(prevOut, "redeemScript");
209  if (!rs.isNull()) {
210  std::vector<unsigned char> rsData(ParseHexV(rs, "redeemScript"));
211  CScript redeemScript(rsData.begin(), rsData.end());
212  keystore->AddCScript(redeemScript);
213  // Automatically also add the P2WSH wrapped version of the script (to deal with P2SH-P2WSH).
214  // This is only for compatibility, it is encouraged to use the explicit witnessScript field instead.
215  keystore->AddCScript(GetScriptForWitness(redeemScript));
216  }
217  UniValue ws = find_value(prevOut, "witnessScript");
218  if (!ws.isNull()) {
219  std::vector<unsigned char> wsData(ParseHexV(ws, "witnessScript"));
220  CScript witnessScript(wsData.begin(), wsData.end());
221  keystore->AddCScript(witnessScript);
222  // Automatically also add the P2WSH wrapped version of the script (to deal with P2SH-P2WSH).
223  keystore->AddCScript(GetScriptForWitness(witnessScript));
224  }
225  }
226  }
227  }
228 
229  int nHashType = ParseSighashString(hashType);
230 
231  bool fHashSingle = ((nHashType & ~SIGHASH_ANYONECANPAY) == SIGHASH_SINGLE);
232 
233  // Script verification errors
234  UniValue vErrors(UniValue::VARR);
235 
236  // Use CTransaction for the constant parts of the
237  // transaction to avoid rehashing.
238  const CTransaction txConst(mtx);
239  // Sign what we can:
240  for (unsigned int i = 0; i < mtx.vin.size(); i++) {
241  CTxIn& txin = mtx.vin[i];
242  auto coin = coins.find(txin.prevout);
243  if (coin == coins.end() || coin->second.IsSpent()) {
244  TxInErrorToJSON(txin, vErrors, "Input not found or already spent");
245  continue;
246  }
247  const CScript& prevPubKey = coin->second.out.scriptPubKey;
248  const CAmount& amount = coin->second.out.nValue;
249 
250  SignatureData sigdata = DataFromTransaction(mtx, i, coin->second.out);
251  // Only sign SIGHASH_SINGLE if there's a corresponding output:
252  if (!fHashSingle || (i < mtx.vout.size())) {
253  ProduceSignature(*keystore, MutableTransactionSignatureCreator(&mtx, i, amount, nHashType), prevPubKey, sigdata);
254  }
255 
256  UpdateInput(txin, sigdata);
257 
258  // amount must be specified for valid segwit signature
259  if (amount == MAX_MONEY && !txin.scriptWitness.IsNull()) {
260  throw JSONRPCError(RPC_TYPE_ERROR, strprintf("Missing amount for %s", coin->second.out.ToString()));
261  }
262 
263  ScriptError serror = SCRIPT_ERR_OK;
264  if (!VerifyScript(txin.scriptSig, prevPubKey, &txin.scriptWitness, STANDARD_SCRIPT_VERIFY_FLAGS, TransactionSignatureChecker(&txConst, i, amount), &serror)) {
265  if (serror == SCRIPT_ERR_INVALID_STACK_OPERATION) {
266  // Unable to sign input and verification failed (possible attempt to partially sign).
267  TxInErrorToJSON(txin, vErrors, "Unable to sign input, invalid stack size (possibly missing key)");
268  } else {
269  TxInErrorToJSON(txin, vErrors, ScriptErrorString(serror));
270  }
271  }
272  }
273  bool fComplete = vErrors.empty();
274 
275  UniValue result(UniValue::VOBJ);
276  result.pushKV("hex", EncodeHexTx(CTransaction(mtx)));
277  result.pushKV("complete", fComplete);
278  if (!vErrors.empty()) {
279  result.pushKV("errors", vErrors);
280  }
281 
282  return result;
283 }
