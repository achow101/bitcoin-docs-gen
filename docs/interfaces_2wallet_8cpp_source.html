72 WalletTx MakeWalletTx(interfaces::Chain::Lock& locked_chain, CWallet& wallet, const CWalletTx& wtx)
73 {
74  WalletTx result;
75  result.tx = wtx.tx;
76  result.txin_is_mine.reserve(wtx.tx->vin.size());
77  for (const auto& txin : wtx.tx->vin) {
78  result.txin_is_mine.emplace_back(wallet.IsMine(txin));
79  }
80  result.txout_is_mine.reserve(wtx.tx->vout.size());
81  result.txout_address.reserve(wtx.tx->vout.size());
82  result.txout_address_is_mine.reserve(wtx.tx->vout.size());
83  for (const auto& txout : wtx.tx->vout) {
84  result.txout_is_mine.emplace_back(wallet.IsMine(txout));
85  result.txout_address.emplace_back();
86  result.txout_address_is_mine.emplace_back(ExtractDestination(txout.scriptPubKey, result.txout_address.back()) ?
87  IsMine(wallet, result.txout_address.back()) :
88  ISMINE_NO);
89  }
90  result.credit = wtx.GetCredit(locked_chain, ISMINE_ALL);
91  result.debit = wtx.GetDebit(ISMINE_ALL);
92  result.change = wtx.GetChange();
93  result.time = wtx.GetTxTime();
94  result.value_map = wtx.mapValue;
95  result.is_coinbase = wtx.IsCoinBase();
96  return result;
97 }
