901 // Analyse the provided scriptPubKey, determining which keys and which redeem scripts from the ImportData struct are needed to spend it, and mark them as used.
902 // Returns an error string, or the empty string for success.
903 static std::string RecurseImportData(const CScript& script, ImportData& import_data, const ScriptContext script_ctx)
904 {
905  // Use Solver to obtain script type and parsed pubkeys or hashes:
906  std::vector<std::vector<unsigned char>> solverdata;
907  txnouttype script_type = Solver(script, solverdata);
908 
909  switch (script_type) {
910  case TX_PUBKEY: {
911  CPubKey pubkey(solverdata[0].begin(), solverdata[0].end());
912  import_data.used_keys.emplace(pubkey.GetID(), false);
913  return "";
914  }
915  case TX_PUBKEYHASH: {
916  CKeyID id = CKeyID(uint160(solverdata[0]));
917  import_data.used_keys[id] = true;
918  return "";
919  }
920  case TX_SCRIPTHASH: {
921  if (script_ctx == ScriptContext::P2SH) throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Trying to nest P2SH inside another P2SH");
922  if (script_ctx == ScriptContext::WITNESS_V0) throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Trying to nest P2SH inside a P2WSH");
923  assert(script_ctx == ScriptContext::TOP);
924  CScriptID id = CScriptID(uint160(solverdata[0]));
925  auto subscript = std::move(import_data.redeemscript); // Remove redeemscript from import_data to check for superfluous script later.
926  if (!subscript) return "missing redeemscript";
927  if (CScriptID(*subscript) != id) return "redeemScript does not match the scriptPubKey";
928  import_data.import_scripts.emplace(*subscript);
929  return RecurseImportData(*subscript, import_data, ScriptContext::P2SH);
930  }
931  case TX_MULTISIG: {
932  for (size_t i = 1; i + 1< solverdata.size(); ++i) {
933  CPubKey pubkey(solverdata[i].begin(), solverdata[i].end());
934  import_data.used_keys.emplace(pubkey.GetID(), false);
935  }
936  return "";
937  }
938  case TX_WITNESS_V0_SCRIPTHASH: {
939  if (script_ctx == ScriptContext::WITNESS_V0) throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Trying to nest P2WSH inside another P2WSH");
940  uint256 fullid(solverdata[0]);
941  CScriptID id;
942  CRIPEMD160().Write(fullid.begin(), fullid.size()).Finalize(id.begin());
943  auto subscript = std::move(import_data.witnessscript); // Remove redeemscript from import_data to check for superfluous script later.
944  if (!subscript) return "missing witnessscript";
945  if (CScriptID(*subscript) != id) return "witnessScript does not match the scriptPubKey or redeemScript";
946  if (script_ctx == ScriptContext::TOP) {
947  import_data.import_scripts.emplace(script); // Special rule for IsMine: native P2WSH requires the TOP script imported (see script/ismine.cpp)
948  }
949  import_data.import_scripts.emplace(*subscript);
950  return RecurseImportData(*subscript, import_data, ScriptContext::WITNESS_V0);
951  }
952  case TX_WITNESS_V0_KEYHASH: {
953  if (script_ctx == ScriptContext::WITNESS_V0) throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Trying to nest P2WPKH inside P2WSH");
954  CKeyID id = CKeyID(uint160(solverdata[0]));
955  import_data.used_keys[id] = true;
956  if (script_ctx == ScriptContext::TOP) {
957  import_data.import_scripts.emplace(script); // Special rule for IsMine: native P2WPKH requires the TOP script imported (see script/ismine.cpp)
958  }
959  return "";
960  }
961  case TX_NULL_DATA:
962  return "unspendable script";
963  case TX_NONSTANDARD:
964  case TX_WITNESS_UNKNOWN:
965  default:
966  return "unrecognized script";
967  }
968 }

970 static UniValue ProcessImportLegacy(ImportData& import_data, std::map<CKeyID, CPubKey>& pubkey_map, std::map<CKeyID, CKey>& privkey_map, std::set<CScript>& script_pub_keys, bool& have_solving_data, const UniValue& data, std::vector<CKeyID>& ordered_pubkeys)
971 {
972  UniValue warnings(UniValue::VARR);
973 
974  // First ensure scriptPubKey has either a script or JSON with "address" string
975  const UniValue& scriptPubKey = data["scriptPubKey"];
976  bool isScript = scriptPubKey.getType() == UniValue::VSTR;
977  if (!isScript && !(scriptPubKey.getType() == UniValue::VOBJ && scriptPubKey.exists("address"))) {
978  throw JSONRPCError(RPC_INVALID_PARAMETER, "scriptPubKey must be string with script or JSON with address string");
979  }
980  const std::string& output = isScript ? scriptPubKey.get_str() : scriptPubKey["address"].get_str();
981 
982  // Optional fields.
983  const std::string& strRedeemScript = data.exists("redeemscript") ? data["redeemscript"].get_str() : "";
984  const std::string& witness_script_hex = data.exists("witnessscript") ? data["witnessscript"].get_str() : "";
985  const UniValue& pubKeys = data.exists("pubkeys") ? data["pubkeys"].get_array() : UniValue();
986  const UniValue& keys = data.exists("keys") ? data["keys"].get_array() : UniValue();
987  const bool internal = data.exists("internal") ? data["internal"].get_bool() : false;
988  const bool watchOnly = data.exists("watchonly") ? data["watchonly"].get_bool() : false;
989 
990  if (data.exists("range")) {
991  throw JSONRPCError(RPC_INVALID_PARAMETER, "Range should not be specified for a non-descriptor import");
992  }
993 
994  // Generate the script and destination for the scriptPubKey provided
995  CScript script;
996  if (!isScript) {
997  CTxDestination dest = DecodeDestination(output);
998  if (!IsValidDestination(dest)) {
999  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Invalid address \"" + output + "\"");
1000  }
1001  script = GetScriptForDestination(dest);
1002  } else {
1003  if (!IsHex(output)) {
1004  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Invalid scriptPubKey \"" + output + "\"");
1005  }
1006  std::vector<unsigned char> vData(ParseHex(output));
1007  script = CScript(vData.begin(), vData.end());
1008  CTxDestination dest;
1009  if (!ExtractDestination(script, dest) && !internal) {
1010  throw JSONRPCError(RPC_INVALID_PARAMETER, "Internal must be set to true for nonstandard scriptPubKey imports.");
1011  }
1012  }
1013  script_pub_keys.emplace(script);
1014 
1015  // Parse all arguments
1016  if (strRedeemScript.size()) {
1017  if (!IsHex(strRedeemScript)) {
1018  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Invalid redeem script \"" + strRedeemScript + "\": must be hex string");
1019  }
1020  auto parsed_redeemscript = ParseHex(strRedeemScript);
1021  import_data.redeemscript = MakeUnique<CScript>(parsed_redeemscript.begin(), parsed_redeemscript.end());
1022  }
1023  if (witness_script_hex.size()) {
1024  if (!IsHex(witness_script_hex)) {
1025  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Invalid witness script \"" + witness_script_hex + "\": must be hex string");
1026  }
1027  auto parsed_witnessscript = ParseHex(witness_script_hex);
1028  import_data.witnessscript = MakeUnique<CScript>(parsed_witnessscript.begin(), parsed_witnessscript.end());
1029  }
1030  for (size_t i = 0; i < pubKeys.size(); ++i) {
1031  const auto& str = pubKeys[i].get_str();
1032  if (!IsHex(str)) {
1033  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Pubkey \"" + str + "\" must be a hex string");
1034  }
1035  auto parsed_pubkey = ParseHex(str);
1036  CPubKey pubkey(parsed_pubkey.begin(), parsed_pubkey.end());
1037  if (!pubkey.IsFullyValid()) {
1038  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Pubkey \"" + str + "\" is not a valid public key");
1039  }
1040  pubkey_map.emplace(pubkey.GetID(), pubkey);
1041  ordered_pubkeys.push_back(pubkey.GetID());
1042  }
1043  for (size_t i = 0; i < keys.size(); ++i) {
1044  const auto& str = keys[i].get_str();
1045  CKey key = DecodeSecret(str);
1046  if (!key.IsValid()) {
1047  throw JSONRPCError(RPC_INVALID_ADDRESS_OR_KEY, "Invalid private key encoding");
1048  }
1049  CPubKey pubkey = key.GetPubKey();
1050  CKeyID id = pubkey.GetID();
1051  if (pubkey_map.count(id)) {
1052  pubkey_map.erase(id);
1053  }
1054  privkey_map.emplace(id, key);
1055  }
1056
