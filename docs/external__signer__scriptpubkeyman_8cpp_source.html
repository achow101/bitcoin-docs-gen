16bool ExternalSignerScriptPubKeyMan::SetupDescriptor(std::unique_ptr<Descriptor> desc)
17{
18 LOCK(cs_desc_man);
19 assert(m_storage.IsWalletFlagSet(WALLET_FLAG_DESCRIPTORS));
20 assert(m_storage.IsWalletFlagSet(WALLET_FLAG_EXTERNAL_SIGNER));
21
22 int64_t creation_time = GetTime();
23
24 // Make the descriptor
25 WalletDescriptor w_desc(std::move(desc), creation_time, 0, 0, 0);
26 m_wallet_descriptor = w_desc;
27
28 // Store the descriptor
29 WalletBatch batch(m_storage.GetDatabase());
30 if (!batch.WriteDescriptor(GetID(), m_wallet_descriptor)) {
31 throw std::runtime_error(std::string(__func__) + ": writing descriptor failed");
32 }
33
34 // TopUp
35 TopUp();
36
37 m_storage.UnsetBlankWalletFlag(batch);
38 return true;
39}

41ExternalSigner ExternalSignerScriptPubKeyMan::GetExternalSigner() {
42 const std::string command = gArgs.GetArg("-signer", "");
43 if (command == "") throw std::runtime_error(std::string(__func__) + ": restart bitcoind with -signer=<cmd>");
44 std::vector<ExternalSigner> signers;
45 ExternalSigner::Enumerate(command, signers, Params().NetworkIDString());
46 if (signers.empty()) throw std::runtime_error(std::string(__func__) + ": No external signers found");
47 // TODO: add fingerprint argument in case of multiple signers
48 return signers[0];

63TransactionError ExternalSignerScriptPubKeyMan::FillPSBT(PartiallySignedTransaction& psbt, const PrecomputedTransactionData& txdata, int sighash_type, bool sign, bool bip32derivs, int* n_signed) const
64{
65 if (!sign) {
66 return DescriptorScriptPubKeyMan::FillPSBT(psbt, txdata, sighash_type, false, bip32derivs, n_signed);
67 }
68
69 // Already complete if every input is now signed
70 bool complete = true;
71 for (const auto& input : psbt.inputs) {
72 // TODO: for multisig wallets, we should only care if all _our_ inputs are signed
73 complete &= PSBTInputSigned(input);
74 }
75 if (complete) return TransactionError::OK;
76
77 std::string strFailReason;
78 if(!GetExternalSigner().SignTransaction(psbt, strFailReason)) {
79 tfm::format(std::cerr, "Failed to sign: %s\n", strFailReason);
80 return TransactionError::EXTERNAL_SIGNER_FAILED;
81 }
82 FinalizePSBT(psbt); // This won't work in a multisig setup
83 return TransactionError::OK;
84}
