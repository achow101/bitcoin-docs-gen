75 void RPCTypeCheckObj(const UniValue& o,
76  const std::map<std::string, UniValueType>& typesExpected,
77  bool fAllowNull,
78  bool fStrict)
79 {
80  for (const auto& t : typesExpected) {
81  const UniValue& v = find_value(o, t.first);
82  if (!fAllowNull && v.isNull())
83  throw JSONRPCError(RPC_TYPE_ERROR, strprintf("Missing %s", t.first));
84 
85  if (!(t.second.typeAny || v.type() == t.second.type || (fAllowNull && v.isNull()))) {
86  std::string err = strprintf("Expected type %s for %s, got %s",
87  uvTypeName(t.second.type), t.first, uvTypeName(v.type()));
88  throw JSONRPCError(RPC_TYPE_ERROR, err);
89  }
90  }
91 
92  if (fStrict)

94  for (const std::string& k : o.getKeys())

96  if (typesExpected.count(k) == 0)

144 std::string CRPCTable::help(const std::string& strCommand, const JSONRPCRequest& helpreq) const
145 {
146  std::string strRet;
147  std::string category;
148  std::set<rpcfn_type> setDone;
149  std::vector<std::pair<std::string, const CRPCCommand*> > vCommands;
150 
151  for (const auto& entry : mapCommands)
152  vCommands.push_back(make_pair(entry.second->category + entry.first, entry.second));
153  sort(vCommands.begin(), vCommands.end());
154 
155  JSONRPCRequest jreq(helpreq);
156  jreq.fHelp = true;
157  jreq.params = UniValue();
158 
159  for (const std::pair<std::string, const CRPCCommand*>& command : vCommands)
160  {
161  const CRPCCommand *pcmd = command.second;
162  std::string strMethod = pcmd->name;
163  if ((strCommand != "" || pcmd->category == "hidden") && strMethod != strCommand)
164  continue;
165  jreq.strMethod = strMethod;
166  try
167  {
168  rpcfn_type pfn = pcmd->actor;
169  if (setDone.insert(pfn).second)
170  (*pfn)(jreq);
171  }
172  catch (const std::exception& e)
173  {
174  // Help text is returned in an exception
175  std::string strHelp = std::string(e.what());
176  if (strCommand == "")
177  {
178  if (strHelp.find('\n') != std::string::npos)
179  strHelp = strHelp.substr(0, strHelp.find('\n'));
180 
181  if (category != pcmd->category)
182  {
183  if (!category.empty())
184  strRet += "\n";
185  category = pcmd->category;
186  strRet += "== " + Capitalize(category) + " ==\n";
187  }
188  }
189  strRet += strHelp + "\n";
190  }
191  }
192  if (strRet == "")
193  strRet = strprintf("help: unknown command: %s\n", strCommand);
194  strRet = strRet.substr(0,strRet.size()-1);
195  return strRet;
196 }

350 void JSONRPCRequest::parse(const UniValue& valRequest)
351 {
352  // Parse request
353  if (!valRequest.isObject())
354  throw JSONRPCError(RPC_INVALID_REQUEST, "Invalid Request object");
355  const UniValue& request = valRequest.get_obj();
356 
357  // Parse id now so errors from here on will have the id
358  id = find_value(request, "id");
359 
360  // Parse method
361  UniValue valMethod = find_value(request, "method");
362  if (valMethod.isNull())
363  throw JSONRPCError(RPC_INVALID_REQUEST, "Missing method");
364  if (!valMethod.isStr())
365  throw JSONRPCError(RPC_INVALID_REQUEST, "Method must be a string");
366  strMethod = valMethod.get_str();
367  if (fLogIPs)
368  LogPrint(BCLog::RPC, "ThreadRPCServer method=%s user=%s peeraddr=%s\n", SanitizeString(strMethod),
369  this->authUser, this->peerAddr);
370  else
371  LogPrint(BCLog::RPC, "ThreadRPCServer method=%s user=%s\n", SanitizeString(strMethod), this->authUser);
372 
373  // Parse params
374  UniValue valParams = find_value(request, "params");
375  if (valParams.isArray() || valParams.isObject())
376  params = valParams;
377  else if (valParams.isNull())
378  params = UniValue(UniValue::VARR);
379  else
380  throw JSONRPCError(RPC_INVALID_REQUEST, "Params must be an array or object");
381 }

426 static inline JSONRPCRequest transformNamedArguments(const JSONRPCRequest& in, const std::vector<std::string>& argNames)
427 {
428  JSONRPCRequest out = in;
429  out.params = UniValue(UniValue::VARR);
430  // Build a map of parameters, and remove ones that have been processed, so that we can throw a focused error if
431  // there is an unknown one.
432  const std::vector<std::string>& keys = in.params.getKeys();
433  const std::vector<UniValue>& values = in.params.getValues();
434  std::unordered_map<std::string, const UniValue*> argsIn;
435  for (size_t i=0; i<keys.size(); ++i) {
436  argsIn[keys[i]] = &values[i];
437  }
438  // Process expected parameters.
439  int hole = 0;
440  for (const std::string &argNamePattern: argNames) {
441  std::vector<std::string> vargNames;
442  boost::algorithm::split(vargNames, argNamePattern, boost::algorithm::is_any_of("|"));
443  auto fr = argsIn.end();
444  for (const std::string & argName : vargNames) {
445  fr = argsIn.find(argName);
446  if (fr != argsIn.end()) {
447  break;
448  }
449  }
450  if (fr != argsIn.end()) {
451  for (int i = 0; i < hole; ++i) {
452  // Fill hole between specified parameters with JSON nulls,
453  // but not at the end (for backwards compatibility with calls
454  // that act based on number of specified parameters).
455  out.params.push_back(UniValue());
456  }
457  hole = 0;
458  out.params.push_back(*fr->second);
459  argsIn.erase(fr);
460  } else {
461  hole += 1;
462  }
463  }
464  // If there are still arguments in the argsIn map, this is an error.
465  if (!argsIn.empty()) {
466  throw JSONRPCError(RPC_INVALID_PARAMETER, "Unknown named parameter " + argsIn.begin()->first);
467  }
468  // Return request with named arguments transformed to positional arguments
469  return out;
470 }

472 UniValue CRPCTable::execute(const JSONRPCRequest &request) const
473 {
474  // Return immediately if in warmup
475  {
476  LOCK(cs_rpcWarmup);
477  if (fRPCInWarmup)
478  throw JSONRPCError(RPC_IN_WARMUP, rpcWarmupStatus);
479  }
480 
481  // Find method
482  const CRPCCommand *pcmd = tableRPC[request.strMethod];
483  if (!pcmd)
484  throw JSONRPCError(RPC_METHOD_NOT_FOUND, "Method not found");

490  // Execute, convert arguments to array if necessary
