36 int getConfTargetForIndex(int index) {
37  if (index+1 > static_cast<int>(confTargets.size())) {
38  return confTargets.back();
39  }
40  if (index < 0) {
41  return confTargets[0];

43  return confTargets[index];

45 int getIndexForConfTarget(int target) {
46  for (unsigned int i = 0; i < confTargets.size(); i++) {

216 void SendCoinsDialog::on_sendButton_clicked()
217 {
218  if(!model || !model->getOptionsModel())
219  return;
220 
221  QList<SendCoinsRecipient> recipients;
222  bool valid = true;
223 
224  for(int i = 0; i < ui->entries->count(); ++i)
225  {
226  SendCoinsEntry *entry = qobject_cast<SendCoinsEntry*>(ui->entries->itemAt(i)->widget());
227  if(entry)
228  {
229  if(entry->validate(model->node()))
230  {
231  recipients.append(entry->getValue());
232  }
233  else
234  {
235  valid = false;
236  }
237  }
238  }
239 
240  if(!valid || recipients.isEmpty())
241  {
242  return;
243  }
244 
245  fNewRecipientAllowed = false;
246  WalletModel::UnlockContext ctx(model->requestUnlock());
247  if(!ctx.isValid())
248  {
249  // Unlock wallet was cancelled
250  fNewRecipientAllowed = true;
251  return;
252  }
253 
254  // prepare transaction for getting txFee earlier
255  WalletModelTransaction currentTransaction(recipients);
256  WalletModel::SendCoinsReturn prepareStatus;
257 
258  // Always use a CCoinControl instance, use the CoinControlDialog instance if CoinControl has been enabled
259  CCoinControl ctrl;
260  if (model->getOptionsModel()->getCoinControlFeatures())
261  ctrl = *CoinControlDialog::coinControl();
262 
263  updateCoinControlState(ctrl);
264 
265  prepareStatus = model->prepareTransaction(currentTransaction, ctrl);
266 
267  // process prepareStatus and on error generate message shown to user
268  processSendCoinsReturn(prepareStatus,
269  BitcoinUnits::formatWithUnit(model->getOptionsModel()->getDisplayUnit(), currentTransaction.getTransactionFee()));
270 
271  if(prepareStatus.status != WalletModel::OK) {
272  fNewRecipientAllowed = true;
273  return;
274  }
275 
276  CAmount txFee = currentTransaction.getTransactionFee();
277 
278  // Format confirmation message
279  QStringList formatted;
280  for (const SendCoinsRecipient &rcp : currentTransaction.getRecipients())
281  {
282  // generate bold amount string with wallet name in case of multiwallet
283  QString amount = "<b>" + BitcoinUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), rcp.amount);
284  if (model->isMultiwallet()) {
285  amount.append(" <u>"+tr("from wallet %1").arg(GUIUtil::HtmlEscape(model->getWalletName()))+"</u> ");
286  }
287  amount.append("</b>");
288  // generate monospace address string
289  QString address = "<span style='font-family: monospace;'>" + rcp.address;
290  address.append("</span>");
291 
292  QString recipientElement;
293  recipientElement = "<br />";
294 
295 #ifdef ENABLE_BIP70
296  if (!rcp.paymentRequest.IsInitialized()) // normal payment
297 #endif
298  {
299  if(rcp.label.length() > 0) // label with address
300  {
301  recipientElement.append(tr("%1 to %2").arg(amount, GUIUtil::HtmlEscape(rcp.label)));
302  recipientElement.append(QString(" (%1)").arg(address));
303  }
304  else // just address
305  {
306  recipientElement.append(tr("%1 to %2").arg(amount, address));
307  }
308  }
309 #ifdef ENABLE_BIP70
310  else if(!rcp.authenticatedMerchant.isEmpty()) // authenticated payment request
311  {
312  recipientElement.append(tr("%1 to %2").arg(amount, GUIUtil::HtmlEscape(rcp.authenticatedMerchant)));
313  }
314  else // unauthenticated payment request
315  {
316  recipientElement.append(tr("%1 to %2").arg(amount, address));
317  }
318 #endif
319 
320  formatted.append(recipientElement);
321  }
322 
323  QString questionString = tr("Are you sure you want to send?");
324  questionString.append("<br /><span style='font-size:10pt;'>");
325  questionString.append(tr("Please, review your transaction."));
326  questionString.append("</span><br />%1");
327 
328  if(txFee > 0)
329  {
330  // append fee string if a fee is required
331  questionString.append("<hr /><b>");
332  questionString.append(tr("Transaction fee"));
333  questionString.append("</b>");
334 
335  // append transaction size
336  questionString.append(" (" + QString::number((double)currentTransaction.getTransactionSize() / 1000) + " kB): ");
337 
338  // append transaction fee value
339  questionString.append("<span style='color:#aa0000; font-weight:bold;'>");
340  questionString.append(BitcoinUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), txFee));
341  questionString.append("</span><br />");
342 
343  // append RBF message according to transaction's signalling
344  questionString.append("<span style='font-size:10pt; font-weight:normal;'>");
345  if (ui->optInRBF->isChecked()) {
346  questionString.append(tr("You can increase the fee later (signals Replace-By-Fee, BIP-125)."));
347  } else {
348  questionString.append(tr("Not signalling Replace-By-Fee, BIP-125."));
349  }
350  questionString.append("</span>");
351  }
352 
353  // add total amount in all subdivision units
354  questionString.append("<hr />");
355  CAmount totalAmount = currentTransaction.getTotalTransactionAmount() + txFee;
356  QStringList alternativeUnits;
357  for (const BitcoinUnits::Unit u : BitcoinUnits::availableUnits())
358  {
359  if(u != model->getOptionsModel()->getDisplayUnit())
360  alternativeUnits.append(BitcoinUnits::formatHtmlWithUnit(u, totalAmount));
361  }
362  questionString.append(QString("<b>%1</b>: <b>%2</b>").arg(tr("Total Amount"))
363  .arg(BitcoinUnits::formatHtmlWithUnit(model->getOptionsModel()->getDisplayUnit(), totalAmount)));
364  questionString.append(QString("<br /><span style='font-size:10pt; font-weight:normal;'>(=%1)</span>")
365  .arg(alternativeUnits.join(" " + tr("or") + " ")));
366 
367  SendConfirmationDialog confirmationDialog(tr("Confirm send coins"),
368  questionString.arg(formatted.join("<br />")), SEND_CONFIRM_DELAY, this);
369  confirmationDialog.exec();
370  QMessageBox::StandardButton retval = static_cast<QMessageBox::StandardButton>(confirmationDialog.result());
371 
372  if(retval != QMessageBox::Yes)
373  {
374  fNewRecipientAllowed = true;
375  return;
376  }
377 
378  // now send the prepared transaction
379  WalletModel::SendCoinsReturn sendStatus = model->sendCoins(currentTransaction);
380  // process sendStatus and on error generate message shown to user
381  processSendCoinsReturn(sendStatus);
382 
383  if (sendStatus.status == WalletModel::OK)
384  {
385  accept();
386  CoinControlDialog::coinControl()->UnSelectAll();
387  coinControlUpdateLabels();
388  Q_EMIT coinsSent(currentTransaction.getWtx()->get().GetHash());
389  }
390  fNewRecipientAllowed = true;
391 }

546 void SendCoinsDialog::processSendCoinsReturn(const WalletModel::SendCoinsReturn &sendCoinsReturn, const QString &msgArg)
547 {
548  QPair<QString, CClientUIInterface::MessageBoxFlags> msgParams;
549  // Default to a warning message, override if error message is needed
550  msgParams.second = CClientUIInterface::MSG_WARNING;
551 
552  // This comment is specific to SendCoinsDialog usage of WalletModel::SendCoinsReturn.
553  // WalletModel::TransactionCommitFailed is used only in WalletModel::sendCoins()
554  // all others are used only in WalletModel::prepareTransaction()
555  switch(sendCoinsReturn.status)
556  {
557  case WalletModel::InvalidAddress:
558  msgParams.first = tr("The recipient address is not valid. Please recheck.");
559  break;
560  case WalletModel::InvalidAmount:
561  msgParams.first = tr("The amount to pay must be larger than 0.");
562  break;
563  case WalletModel::AmountExceedsBalance:
564  msgParams.first = tr("The amount exceeds your balance.");
565  break;
566  case WalletModel::AmountWithFeeExceedsBalance:
567  msgParams.first = tr("The total exceeds your balance when the %1 transaction fee is included.").arg(msgArg);
568  break;
569  case WalletModel::DuplicateAddress:
570  msgParams.first = tr("Duplicate address found: addresses should only be used once each.");
571  break;
572  case WalletModel::TransactionCreationFailed:
573  msgParams.first = tr("Transaction creation failed!");
574  msgParams.second = CClientUIInterface::MSG_ERROR;
575  break;
576  case WalletModel::TransactionCommitFailed:
577  msgParams.first = tr("The transaction was rejected with the following reason: %1").arg(sendCoinsReturn.reasonCommitFailed);
578  msgParams.second = CClientUIInterface::MSG_ERROR;
579  break;
580  case WalletModel::AbsurdFee:
581  msgParams.first = tr("A fee higher than %1 is considered an absurdly high fee.").arg(BitcoinUnits::formatWithUnit(model->getOptionsModel()->getDisplayUnit(), model->node().getMaxTxFee()));
582  break;
583  case WalletModel::PaymentRequestExpired:
584  msgParams.first = tr("Payment request expired.");
585  msgParams.second = CClientUIInterface::MSG_ERROR;
586  break;
587  // included to prevent a compiler warning.
588  case WalletModel::OK:
589  default:
590  return;
591  }
592 
593  Q_EMIT message(tr("Send Coins"), msgParams.first, msgParams.second);
594 }
