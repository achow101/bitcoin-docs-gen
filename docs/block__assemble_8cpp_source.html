54 static void AssembleBlock(benchmark::State& state)
55 {

57  CScriptWitness witness;
58  witness.stack.push_back(op_true);
59

62

64 
65  // Switch to regtest so we can mine faster
66  // Also segwit is active, so we can include witness transactions
67  SelectParams(CBaseChainParams::REGTEST);
68 
69  InitScriptExecutionCache();
70 
71  boost::thread_group thread_group;
72  CScheduler scheduler;
73  {
74  ::pblocktree.reset(new CBlockTreeDB(1 << 20, true));
75  ::pcoinsdbview.reset(new CCoinsViewDB(1 << 23, true));
76  ::pcoinsTip.reset(new CCoinsViewCache(pcoinsdbview.get()));
77 
78  const CChainParams& chainparams = Params();
79  thread_group.create_thread(boost::bind(&CScheduler::serviceQueue, &scheduler));
80  GetMainSignals().RegisterBackgroundSignalScheduler(scheduler);
81  LoadGenesisBlock(chainparams);
82  CValidationState state;
83  ActivateBestChain(state, chainparams);
84  assert(::chainActive.Tip() != nullptr);
85  const bool witness_enabled{IsWitnessEnabled(::chainActive.Tip(), chainparams.GetConsensus())};
86  assert(witness_enabled);
87  }
88 
89  // Collect some loose transactions that spend the coinbases of our mined blocks
90  constexpr size_t NUM_BLOCKS{200};
91  std::array<CTransactionRef, NUM_BLOCKS - COINBASE_MATURITY + 1> txs;
92  for (size_t b{0}; b < NUM_BLOCKS; ++b) {
93  CMutableTransaction tx;

95  tx.vin.back().scriptWitness = witness;
96  tx.vout.emplace_back(1337, SCRIPT_PUB);
97  if (NUM_BLOCKS - b >= COINBASE_MATURITY)

99  }

103  assert(ret);
104  }
105

108  }
109
