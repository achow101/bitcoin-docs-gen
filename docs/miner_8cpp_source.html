99 std::unique_ptr<CBlockTemplate> BlockAssembler::CreateNewBlock(const CScript& scriptPubKeyIn, bool fMineWitnessTx)
100 {
101  int64_t nTimeStart = GetTimeMicros();
102 
103  resetBlock();
104 
105  pblocktemplate.reset(new CBlockTemplate());
106 
107  if(!pblocktemplate.get())
108  return nullptr;
109  pblock = &pblocktemplate->block; // pointer for convenience
110 
111  // Add dummy coinbase tx as first transaction
112  pblock->vtx.emplace_back();
113  pblocktemplate->vTxFees.push_back(-1); // updated at end
114  pblocktemplate->vTxSigOpsCost.push_back(-1); // updated at end
115 
116  LOCK2(cs_main, mempool.cs);
117  CBlockIndex* pindexPrev = chainActive.Tip();
118  assert(pindexPrev != nullptr);
119  nHeight = pindexPrev->nHeight + 1;
120 
121  pblock->nVersion = ComputeBlockVersion(pindexPrev, chainparams.GetConsensus());
122  // -regtest only: allow overriding block.nVersion with
123  // -blockversion=N to test forking scenarios
124  if (chainparams.MineBlocksOnDemand())
125  pblock->nVersion = gArgs.GetArg("-blockversion", pblock->nVersion);
126 
127  pblock->nTime = GetAdjustedTime();
128  const int64_t nMedianTimePast = pindexPrev->GetMedianTimePast();
129 
130  nLockTimeCutoff = (STANDARD_LOCKTIME_VERIFY_FLAGS & LOCKTIME_MEDIAN_TIME_PAST)
131  ? nMedianTimePast
132  : pblock->GetBlockTime();
133 
134  // Decide whether to include witness transactions
135  // This is only needed in case the witness softfork activation is reverted
136  // (which would require a very deep reorganization).
137  // Note that the mempool would accept transactions with witness data before
138  // IsWitnessEnabled, but we would only ever mine blocks after IsWitnessEnabled
139  // unless there is a massive block reorganization with the witness softfork
140  // not activated.
141  // TODO: replace this with a call to main to assess validity of a mempool
142  // transaction (which in most cases can be a no-op).
143  fIncludeWitness = IsWitnessEnabled(pindexPrev, chainparams.GetConsensus()) && fMineWitnessTx;
144

146  int nDescendantsUpdated = 0;
147  addPackageTxs(nPackagesSelected, nDescendantsUpdated);
148 
149  int64_t nTime1 = GetTimeMicros();
150 
151  nLastBlockTx = nBlockTx;
152  nLastBlockWeight = nBlockWeight;
153 
154  // Create coinbase transaction.
155  CMutableTransaction coinbaseTx;
156  coinbaseTx.vin.resize(1);
157  coinbaseTx.vin[0].prevout.SetNull();
158  coinbaseTx.vout.resize(1);
159  coinbaseTx.vout[0].scriptPubKey = scriptPubKeyIn;
160  coinbaseTx.vout[0].nValue = nFees + GetBlockSubsidy(nHeight, chainparams.GetConsensus());
161  coinbaseTx.vin[0].scriptSig = CScript() << nHeight << OP_0;
162  pblock->vtx[0] = MakeTransactionRef(std::move(coinbaseTx));
163  pblocktemplate->vchCoinbaseCommitment = GenerateCoinbaseCommitment(*pblock, pindexPrev, chainparams.GetConsensus());
164  pblocktemplate->vTxFees[0] = -nFees;
165 
166  LogPrintf("CreateNewBlock(): block weight: %u txs: %u fees: %ld sigops %d\n", GetBlockWeight(*pblock), nBlockTx, nFees, nBlockSigOpsCost);
167 
168  // Fill in header
169  pblock->hashPrevBlock = pindexPrev->GetBlockHash();
170  UpdateTime(pblock, chainparams.GetConsensus(), pindexPrev);
171  pblock->nBits = GetNextWorkRequired(pindexPrev, pblock, chainparams.GetConsensus());
172  pblock->nNonce = 0;
173  pblocktemplate->vTxSigOpsCost[0] = WITNESS_SCALE_FACTOR * GetLegacySigOpCount(*pblock->vtx[0]);
174 
175  CValidationState state;
176  if (!TestBlockValidity(state, chainparams, *pblock, pindexPrev, false, false)) {
177  throw std::runtime_error(strprintf("%s: TestBlockValidity failed: %s", __func__, FormatStateMessage(state)));
178  }
179  int64_t nTime2 = GetTimeMicros();
180 
181  LogPrint(BCLog::BENCH, "CreateNewBlock() packages: %.2fms (%d packages, %d updated descendants), validity: %.2fms (total %.2fms)\n", 0.001 * (nTime1 - nTimeStart), nPackagesSelected, nDescendantsUpdated, 0.001 * (nTime2 - nTime1), 0.001 * (nTime2 - nTimeStart));

183  return std::move(pblocktemplate);

243 int BlockAssembler::UpdatePackagesForAdded(const CTxMemPool::setEntries& alreadyAdded,
244  indexed_modified_transaction_set &mapModifiedTx)
245 {
246  int nDescendantsUpdated = 0;
247  for (CTxMemPool::txiter it : alreadyAdded) {
248  CTxMemPool::setEntries descendants;
249  mempool.CalculateDescendants(it, descendants);
250  // Insert all descendants (not yet in block) into the modified set
251  for (CTxMemPool::txiter desc : descendants) {
252  if (alreadyAdded.count(desc))
253  continue;
254  ++nDescendantsUpdated;
255  modtxiter mit = mapModifiedTx.find(desc);
256  if (mit == mapModifiedTx.end()) {
257  CTxMemPoolModifiedEntry modEntry(desc);
258  modEntry.nSizeWithAncestors -= it->GetTxSize();
259  modEntry.nModFeesWithAncestors -= it->GetModifiedFee();
260  modEntry.nSigOpCostWithAncestors -= it->GetSigOpCost();
261  mapModifiedTx.insert(modEntry);
262  } else {
263  mapModifiedTx.modify(mit, update_for_parent_inclusion(it));
264  }
265  }
266  }
267  return nDescendantsUpdated;
268 }

270 // Skip entries in mapTx that are already in a block or are present
271 // in mapModifiedTx (which implies that the mapTx ancestor state is
272 // stale due to ancestor inclusion in the block)
273 // Also skip transactions that we've already failed to add. This can happen if
274 // we consider a transaction in mapModifiedTx and it fails: we can then
275 // potentially consider it again while walking mapTx. It's currently
276 // guaranteed to fail again, but as a belt-and-suspenders check we put it in
277 // failedTx and avoid re-evaluation, since the re-evaluation would be using
278 // cached size/sigops/fee values that are not actually correct.

285 void BlockAssembler::SortForBlock(const CTxMemPool::setEntries& package, std::vector<CTxMemPool::txiter>& sortedEntries)
286 {
287  // Sort package by ancestor count
288  // If a transaction A depends on transaction B, then A's ancestor count
289  // must be greater than B's. So this is sufficient to validly order the
290  // transactions for block inclusion.
291  sortedEntries.clear();
292  sortedEntries.insert(sortedEntries.begin(), package.begin(), package.end());
293  std::sort(sortedEntries.begin(), sortedEntries.end(), CompareTxIterByAncestorCount());
294 }

296 // This transaction selection algorithm orders the mempool based
297 // on feerate of a transaction including all unconfirmed ancestors.
298 // Since we don't remove transactions from the mempool as we select them
299 // for block inclusion, we need an alternate method of updating the feerate
300 // of a transaction with its not-yet-selected ancestors as we go.
301 // This is accomplished by walking the in-mempool descendants of selected
302 // transactions and storing a temporary modified state in mapModifiedTxs.
303 // Each time through the loop, we compare the best transaction in
304 // mapModifiedTxs with the next transaction in the mempool to decide what
305 // transaction package to work on next.
306 void BlockAssembler::addPackageTxs(int &nPackagesSelected, int &nDescendantsUpdated)
307 {
308  // mapModifiedTx will store sorted packages after they are modified
309  // because some of their txs are already in the block
310  indexed_modified_transaction_set mapModifiedTx;
311  // Keep track of entries that failed inclusion, to avoid duplicate work
312  CTxMemPool::setEntries failedTx;
313 
314  // Start by adding all descendants of previously added txs to mapModifiedTx
315  // and modifying them for their already included ancestors
316  UpdatePackagesForAdded(inBlock, mapModifiedTx);
317 
318  CTxMemPool::indexed_transaction_set::index<ancestor_score>::type::iterator mi = mempool.mapTx.get<ancestor_score>().begin();
319  CTxMemPool::txiter iter;
320 
321  // Limit the number of attempts to add transactions to the block when it is
322  // close to full; this is just a simple heuristic to finish quickly if the
323  // mempool has a lot of entries.
324  const int64_t MAX_CONSECUTIVE_FAILURES = 1000;
325  int64_t nConsecutiveFailed = 0;
326 
327  while (mi != mempool.mapTx.get<ancestor_score>().end() || !mapModifiedTx.empty())
328  {
329  // First try to find a new transaction in mapTx to evaluate.
330  if (mi != mempool.mapTx.get<ancestor_score>().end() &&
331  SkipMapTxEntry(mempool.mapTx.project<0>(mi), mapModifiedTx, failedTx)) {
332  ++mi;
333  continue;
334  }
335 
336  // Now that mi is not stale, determine which transaction to evaluate:
337  // the next entry from mapTx, or the best from mapModifiedTx?
338  bool fUsingModified = false;
339 
340  modtxscoreiter modit = mapModifiedTx.get<ancestor_score>().begin();
341  if (mi == mempool.mapTx.get<ancestor_score>().end()) {
342  // We're out of entries in mapTx; use the entry from mapModifiedTx
343  iter = modit->iter;
344  fUsingModified = true;
345  } else {
346  // Try to compare the mapTx entry to the mapModifiedTx entry
347  iter = mempool.mapTx.project<0>(mi);
348  if (modit != mapModifiedTx.get<ancestor_score>().end() &&
349  CompareTxMemPoolEntryByAncestorFee()(*modit, CTxMemPoolModifiedEntry(iter))) {
350  // The best entry in mapModifiedTx has higher score
351  // than the one from mapTx.
352  // Switch which transaction (package) to consider
353  iter = modit->iter;
354  fUsingModified = true;
355  } else {
356  // Either no entry in mapModifiedTx, or it's worse than mapTx.
357  // Increment mi for the next loop iteration.
358  ++mi;
359  }
360  }
361 
362  // We skip mapTx entries that are inBlock, and mapModifiedTx shouldn't
363  // contain anything that is inBlock.
364  assert(!inBlock.count(iter));
365 
366  uint64_t packageSize = iter->GetSizeWithAncestors();
367  CAmount packageFees = iter->GetModFeesWithAncestors();
368  int64_t packageSigOpsCost = iter->GetSigOpCostWithAncestors();
369  if (fUsingModified) {
370  packageSize = modit->nSizeWithAncestors;
371  packageFees = modit->nModFeesWithAncestors;
372  packageSigOpsCost = modit->nSigOpCostWithAncestors;
373  }
374 
375  if (packageFees < blockMinFeeRate.GetFee(packageSize)) {
376  // Everything else we might consider has a lower fee rate
377  return;
378  }
379 
380  if (!TestPackage(packageSize, packageSigOpsCost)) {
381  if (fUsingModified) {
382  // Since we always look at the best entry in mapModifiedTx,
383  // we must erase failed entries so that we can consider the
384  // next best entry on the next loop iteration
385  mapModifiedTx.get<ancestor_score>().erase(modit);
386  failedTx.insert(iter);
387  }
388 
389  ++nConsecutiveFailed;
390 
391  if (nConsecutiveFailed > MAX_CONSECUTIVE_FAILURES && nBlockWeight >
392  nBlockMaxWeight - 4000) {
393  // Give up if we're close to full and haven't succeeded in a while
394  break;
395  }
396  continue;
397  }
398 
399  CTxMemPool::setEntries ancestors;
400  uint64_t nNoLimit = std::numeric_limits<uint64_t>::max();
401  std::string dummy;
402  mempool.CalculateMemPoolAncestors(*iter, ancestors, nNoLimit, nNoLimit, nNoLimit, nNoLimit, dummy, false);
403 
404  onlyUnconfirmed(ancestors);
405  ancestors.insert(iter);
406 
407  // Test if all tx's are Final
408  if (!TestPackageTransactions(ancestors)) {
409  if (fUsingModified) {
410  mapModifiedTx.get<ancestor_score>().erase(modit);
411  failedTx.insert(iter);
412  }
413  continue;
414  }
415 
416  // This transaction will make it in; reset the failed counter.
417  nConsecutiveFailed = 0;
418 
419  // Package can be added. Sort the entries in a valid order.
420  std::vector<CTxMemPool::txiter> sortedEntries;
421  SortForBlock(ancestors, sortedEntries);
422 
423  for (size_t i=0; i<sortedEntries.size(); ++i) {
424  AddToBlock(sortedEntries[i]);
425  // Erase from the modified set, if present
426  mapModifiedTx.erase(sortedEntries[i]);
427  }
428 
429  ++nPackagesSelected;
430 
431  // Update transactions that depend on each of these
432  nDescendantsUpdated += UpdatePackagesForAdded(ancestors, mapModifiedTx);
433  }
434 }
