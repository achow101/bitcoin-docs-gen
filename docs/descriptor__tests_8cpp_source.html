47bool EqualDescriptor(std::string a, std::string b)
48{
49 bool a_check = (a.size() > 9 && a[a.size() - 9] == '#');
50 bool b_check = (b.size() > 9 && b[b.size() - 9] == '#');
51 if (a_check != b_check) {
52 if (a_check) a = a.substr(0, a.size() - 9);
53 if (b_check) b = b.substr(0, b.size() - 9);
54 }
55 return a == b;
56}

58std::string UseHInsteadOfApostrophe(const std::string& desc)
59{
60 std::string ret = desc;
61 while (true) {
62 auto it = ret.find('\'');
63 if (it == std::string::npos) break;
64 ret[it] = 'h';
65 }
66
67 // GetDescriptorChecksum returns "" if the checksum exists but is bad.
68 // Switching apostrophes with 'h' breaks the checksum if it exists - recalculate it and replace the broken one.
69 if (GetDescriptorChecksum(ret) == "") {
70 ret = ret.substr(0, desc.size() - 9);
71 ret += std::string("#") + GetDescriptorChecksum(ret);
72 }
73 return ret;
74}

78void DoCheck(const std::string& prv, const std::string& pub, const std::string& norm_prv, const std::string& norm_pub, int flags, const std::vector<std::vector<std::string>>& scripts, const std::optional<OutputType>& type, const std::set<std::vector<uint32_t>>& paths = ONLY_EMPTY,
79 bool replace_apostrophe_with_h_in_prv=false, bool replace_apostrophe_with_h_in_pub=false)
80{
81 FlatSigningProvider keys_priv, keys_pub;
82 std::set<std::vector<uint32_t>> left_paths = paths;
83 std::string error;
84
85 std::unique_ptr<Descriptor> parse_priv;
86 std::unique_ptr<Descriptor> parse_pub;
87 // Check that parsing succeeds.
88 if (replace_apostrophe_with_h_in_prv) {
89 parse_priv = Parse(UseHInsteadOfApostrophe(prv), keys_priv, error);
90 } else {
91 parse_priv = Parse(prv, keys_priv, error);
92 }
93 if (replace_apostrophe_with_h_in_pub) {
94 parse_pub = Parse(UseHInsteadOfApostrophe(pub), keys_pub, error);
95 } else {
96 parse_pub = Parse(pub, keys_pub, error);
97 }
98
99 BOOST_CHECK(parse_priv);
100 BOOST_CHECK(parse_pub);
101
102 // Check that the correct OutputType is inferred
103 BOOST_CHECK(parse_priv->GetOutputType() == type);
104 BOOST_CHECK(parse_pub->GetOutputType() == type);
105
106 // Check private keys are extracted from the private version but not the public one.
107 BOOST_CHECK(keys_priv.keys.size());
108 BOOST_CHECK(!keys_pub.keys.size());
109
110 // Check that both versions serialize back to the public version.
111 std::string pub1 = parse_priv->ToString();
112 std::string pub2 = parse_pub->ToString();
113 BOOST_CHECK(EqualDescriptor(pub, pub1));
114 BOOST_CHECK(EqualDescriptor(pub, pub2));
115
116 // Check that both can be serialized with private key back to the private version, but not without private key.
117 std::string prv1;
118 BOOST_CHECK(parse_priv->ToPrivateString(keys_priv, prv1));
119 BOOST_CHECK(EqualDescriptor(prv, prv1));
120 BOOST_CHECK(!parse_priv->ToPrivateString(keys_pub, prv1));
121 BOOST_CHECK(parse_pub->ToPrivateString(keys_priv, prv1));
122 BOOST_CHECK(EqualDescriptor(prv, prv1));
123 BOOST_CHECK(!parse_pub->ToPrivateString(keys_pub, prv1));
124
125 // Check that private can produce the normalized descriptors
126 std::string norm1;
127 BOOST_CHECK(parse_priv->ToNormalizedString(keys_priv, norm1));
128 BOOST_CHECK(EqualDescriptor(norm1, norm_pub));
129 BOOST_CHECK(parse_pub->ToNormalizedString(keys_priv, norm1));
130 BOOST_CHECK(EqualDescriptor(norm1, norm_pub));
131
132 // Check whether IsRange on both returns the expected result
133 BOOST_CHECK_EQUAL(parse_pub->IsRange(), (flags & RANGE) != 0);
134 BOOST_CHECK_EQUAL(parse_priv->IsRange(), (flags & RANGE) != 0);
135
136 // * For ranged descriptors, the `scripts` parameter is a list of expected result outputs, for subsequent
137 // positions to evaluate the descriptors on (so the first element of `scripts` is for evaluating the
138 // descriptor at 0; the second at 1; and so on). To verify this, we evaluate the descriptors once for
139 // each element in `scripts`.
140 // * For non-ranged descriptors, we evaluate the descriptors at positions 0, 1, and 2, but expect the
141 // same result in each case, namely the first element of `scripts`. Because of that, the size of
142 // `scripts` must be one in that case.
143 if (!(flags & RANGE)) assert(scripts.size() == 1);
144 size_t max = (flags & RANGE) ? scripts.size() : 3;
145
146 // Iterate over the position we'll evaluate the descriptors in.
147 for (size_t i = 0; i < max; ++i) {
148 // Call the expected result scripts `ref`.
149 const auto& ref = scripts[(flags & RANGE) ? i : 0];
150 // When t=0, evaluate the `prv` descriptor; when t=1, evaluate the `pub` descriptor.
151 for (int t = 0; t < 2; ++t) {
152 // When the descriptor is hardened, evaluate with access to the private keys inside.
153 const FlatSigningProvider& key_provider = (flags & HARDENED) ? keys_priv : keys_pub;
154
155 // Evaluate the descriptor selected by `t` in position `i`.
156 FlatSigningProvider script_provider, script_provider_cached;
157 std::vector<CScript> spks, spks_cached;
158 DescriptorCache desc_cache;
159 BOOST_CHECK((t ? parse_priv : parse_pub)->Expand(i, key_provider, spks, script_provider, &desc_cache));
160
161 // Compare the output with the expected result.
162 BOOST_CHECK_EQUAL(spks.size(), ref.size());
163
164 // Try to expand again using cached data, and compare.
165 BOOST_CHECK(parse_pub->ExpandFromCache(i, desc_cache, spks_cached, script_provider_cached));
166 BOOST_CHECK(spks == spks_cached);
167 BOOST_CHECK(script_provider.pubkeys == script_provider_cached.pubkeys);
168 BOOST_CHECK(script_provider.scripts == script_provider_cached.scripts);
169 BOOST_CHECK(script_provider.origins == script_provider_cached.origins);
170
171 // Check whether keys are in the cache
172 const auto& der_xpub_cache = desc_cache.GetCachedDerivedExtPubKeys();
173 const auto& parent_xpub_cache = desc_cache.GetCachedParentExtPubKeys();
174 if ((flags & RANGE) && !(flags & DERIVE_HARDENED)) {
175 // For ranged, unhardened derivation, None of the keys in origins should appear in the cache but the cache should have parent keys
176 // But we can derive one level from each of those parent keys and find them all
177 BOOST_CHECK(der_xpub_cache.empty());
178 BOOST_CHECK(parent_xpub_cache.size() > 0);
179 std::set<CPubKey> pubkeys;
180 for (const auto& xpub_pair : parent_xpub_cache) {
181 const CExtPubKey& xpub = xpub_pair.second;
182 CExtPubKey der;
183 xpub.Derive(der, i);
184 pubkeys.insert(der.pubkey);
185 }
186 for (const auto& origin_pair : script_provider_cached.origins) {
187 const CPubKey& pk = origin_pair.second.first;
188 BOOST_CHECK(pubkeys.count(pk) > 0);
189 }
190 } else if (pub1.find("xpub") != std::string::npos) {
191 // For ranged, hardened derivation, or not ranged, but has an xpub, all of the keys should appear in the cache
192 BOOST_CHECK(der_xpub_cache.size() + parent_xpub_cache.size() == script_provider_cached.origins.size());
193 // Get all of the derived pubkeys
194 std::set<CPubKey> pubkeys;
195 for (const auto& xpub_map_pair : der_xpub_cache) {
196 for (const auto& xpub_pair : xpub_map_pair.second) {
197 const CExtPubKey& xpub = xpub_pair.second;
198 pubkeys.insert(xpub.pubkey);
199 }
200 }
201 // Derive one level from all of the parents
202 for (const auto& xpub_pair : parent_xpub_cache) {
203 const CExtPubKey& xpub = xpub_pair.second;
204 pubkeys.insert(xpub.pubkey);
205 CExtPubKey der;
206 xpub.Derive(der, i);
207 pubkeys.insert(der.pubkey);
208 }
209 for (const auto& origin_pair : script_provider_cached.origins) {
210 const CPubKey& pk = origin_pair.second.first;
211 BOOST_CHECK(pubkeys.count(pk) > 0);
212 }
213 } else {
214 // No xpub, nothing should be cached
215 BOOST_CHECK(der_xpub_cache.empty());
216 BOOST_CHECK(parent_xpub_cache.empty());
217 }
218
219 // Make sure we can expand using cached xpubs for unhardened derivation
220 if (!(flags & DERIVE_HARDENED)) {
221 // Evaluate the descriptor at i + 1
222 FlatSigningProvider script_provider1, script_provider_cached1;
223 std::vector<CScript> spks1, spk1_from_cache;
224 BOOST_CHECK((t ? parse_priv : parse_pub)->Expand(i + 1, key_provider, spks1, script_provider1, nullptr));
225
226 // Try again but use the cache from expanding i. That cache won't have the pubkeys for i + 1, but will have the parent xpub for derivation.
227 BOOST_CHECK(parse_pub->ExpandFromCache(i + 1, desc_cache, spk1_from_cache, script_provider_cached1));
228 BOOST_CHECK(spks1 == spk1_from_cache);
229 BOOST_CHECK(script_provider1.pubkeys == script_provider_cached1.pubkeys);
230 BOOST_CHECK(script_provider1.scripts == script_provider_cached1.scripts);
231 BOOST_CHECK(script_provider1.origins == script_provider_cached1.origins);
232 }
233
234 // For each of the produced scripts, verify solvability, and when possible, try to sign a transaction spending it.
235 for (size_t n = 0; n < spks.size(); ++n) {
236 BOOST_CHECK_EQUAL(ref[n], HexStr(spks[n]));
237 BOOST_CHECK_EQUAL(IsSolvable(Merge(key_provider, script_provider), spks[n]), (flags & UNSOLVABLE) == 0);
238
239 if (flags & SIGNABLE) {
240 CMutableTransaction spend;
241 spend.vin.resize(1);
242 spend.vout.resize(1);
243 BOOST_CHECK_MESSAGE(SignSignature(Merge(keys_priv, script_provider), spks[n], spend, 0, 1, SIGHASH_ALL), prv);
244 }
245
246 /* Infer a descriptor from the generated script, and verify its solvability and that it roundtrips. */
247 auto inferred = InferDescriptor(spks[n], script_provider);
248 BOOST_CHECK_EQUAL(inferred->IsSolvable(), !(flags & UNSOLVABLE));
249 std::vector<CScript> spks_inferred;
250 FlatSigningProvider provider_inferred;
251 BOOST_CHECK(inferred->Expand(0, provider_inferred, spks_inferred, provider_inferred));
252 BOOST_CHECK_EQUAL(spks_inferred.size(), 1U);
253 BOOST_CHECK(spks_inferred[0] == spks[n]);
254 BOOST_CHECK_EQUAL(IsSolvable(provider_inferred, spks_inferred[0]), !(flags & UNSOLVABLE));
255 BOOST_CHECK(provider_inferred.origins == script_provider.origins);
256 }
257
258 // Test whether the observed key path is present in the 'paths' variable (which contains expected, unobserved paths),
259 // and then remove it from that set.
260 for (const auto& origin : script_provider.origins) {
261 BOOST_CHECK_MESSAGE(paths.count(origin.second.second.path), "Unexpected key path: " + prv);
262 left_paths.erase(origin.second.second.path);
263 }
264 }
265 }
266
267 // Verify no expected paths remain that were not observed.
268 BOOST_CHECK_MESSAGE(left_paths.empty(), "Not all expected key paths found: " + prv);
269}

271void Check(const std::string& prv, const std::string& pub, const std::string& norm_prv, const std::string& norm_pub, int flags, const std::vector<std::vector<std::string>>& scripts, const std::optional<OutputType>& type, const std::set<std::vector<uint32_t>>& paths = ONLY_EMPTY)
272{
273 bool found_apostrophes_in_prv = false;
274 bool found_apostrophes_in_pub = false;
275
276 // Do not replace apostrophes with 'h' in prv and pub
277 DoCheck(prv, pub, norm_prv, norm_pub, flags, scripts, type, paths);
278
279 // Replace apostrophes with 'h' in prv but not in pub, if apostrophes are found in prv
280 if (prv.find('\'') != std::string::npos) {
281 found_apostrophes_in_prv = true;
282 DoCheck(prv, pub, norm_prv, norm_pub, flags, scripts, type, paths, /* replace_apostrophe_with_h_in_prv = */true, /*replace_apostrophe_with_h_in_pub = */false);
283 }
284
285 // Replace apostrophes with 'h' in pub but not in prv, if apostrophes are found in pub
286 if (pub.find('\'') != std::string::npos) {
287 found_apostrophes_in_pub = true;
288 DoCheck(prv, pub, norm_prv, norm_pub, flags, scripts, type, paths, /* replace_apostrophe_with_h_in_prv = */false, /*replace_apostrophe_with_h_in_pub = */true);
289 }
290
291 // Replace apostrophes with 'h' both in prv and in pub, if apostrophes are found in both
292 if (found_apostrophes_in_prv && found_apostrophes_in_pub) {
293 DoCheck(prv, pub, norm_prv, norm_pub, flags, scripts, type, paths, /* replace_apostrophe_with_h_in_prv = */true, /*replace_apostrophe_with_h_in_pub = */true);
294 }
295}
