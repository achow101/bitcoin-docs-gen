88 void BaseIndex::ThreadSync()
89 {
90  const CBlockIndex* pindex = m_best_block_index.load();
91  if (!m_synced) {
92  auto& consensus_params = Params().GetConsensus();
93 
94  int64_t last_log_time = 0;
95  int64_t last_locator_write_time = 0;
96  while (true) {
97  if (m_interrupt) {
98  WriteBestBlock(pindex);
99  return;
100  }
101 
102  {
103  LOCK(cs_main);
104  const CBlockIndex* pindex_next = NextSyncBlock(pindex);
105  if (!pindex_next) {
106  WriteBestBlock(pindex);
107  m_best_block_index = pindex;
108  m_synced = true;
109  break;
110  }
111  pindex = pindex_next;
112  }
113 
114  int64_t current_time = GetTime();
115  if (last_log_time + SYNC_LOG_INTERVAL < current_time) {
116  LogPrintf("Syncing %s with block chain from height %d\n",
117  GetName(), pindex->nHeight);
118  last_log_time = current_time;
119  }
120 
121  if (last_locator_write_time + SYNC_LOCATOR_WRITE_INTERVAL < current_time) {
122  WriteBestBlock(pindex);
123  last_locator_write_time = current_time;
124  }
125 
126  CBlock block;
127  if (!ReadBlockFromDisk(block, pindex, consensus_params)) {
128  FatalError("%s: Failed to read block %s from disk",
129  __func__, pindex->GetBlockHash().ToString());
130  return;
131  }
132  if (!WriteBlock(block, pindex)) {
133  FatalError("%s: Failed to write block %s to index database",
134  __func__, pindex->GetBlockHash().ToString());
135  return;
136  }
137  }
138  }
139 
140  if (pindex) {
141  LogPrintf("%s is enabled at height %d\n", GetName(), pindex->nHeight);
142  } else {
143  LogPrintf("%s is enabled\n", GetName());
144  }
145 }

156 void BaseIndex::BlockConnected(const std::shared_ptr<const CBlock>& block, const CBlockIndex* pindex,
157  const std::vector<CTransactionRef>& txn_conflicted)
158 {
159  if (!m_synced) {
160  return;
161  }
162 
163  const CBlockIndex* best_block_index = m_best_block_index.load();
164  if (!best_block_index) {
165  if (pindex->nHeight != 0) {
166  FatalError("%s: First block connected is not the genesis block (height=%d)",
167  __func__, pindex->nHeight);
168  return;
169  }
170  } else {
171  // Ensure block connects to an ancestor of the current best block. This should be the case
172  // most of the time, but may not be immediately after the sync thread catches up and sets
173  // m_synced. Consider the case where there is a reorg and the blocks on the stale branch are
174  // in the ValidationInterface queue backlog even after the sync thread has caught up to the
175  // new chain tip. In this unlikely event, log a warning and let the queue clear.
176  if (best_block_index->GetAncestor(pindex->nHeight - 1) != pindex->pprev) {
177  LogPrintf("%s: WARNING: Block %s does not connect to an ancestor of " /* Continued */
178  "known best chain (tip=%s); not updating index\n",
179  __func__, pindex->GetBlockHash().ToString(),
180  best_block_index->GetBlockHash().ToString());
181  return;
182  }
183  }
184 
185  if (WriteBlock(*block, pindex)) {
186  m_best_block_index = pindex;
187  } else {
188  FatalError("%s: Failed to write block %s to index",
189  __func__, pindex->GetBlockHash().ToString());
190  return;
191  }
192 }

194 void BaseIndex::ChainStateFlushed(const CBlockLocator& locator)
195 {
196  if (!m_synced) {
197  return;
198  }
199 
200  const uint256& locator_tip_hash = locator.vHave.front();
201  const CBlockIndex* locator_tip_index;
202  {
203  LOCK(cs_main);
204  locator_tip_index = LookupBlockIndex(locator_tip_hash);
205  }
206 
207  if (!locator_tip_index) {
208  FatalError("%s: First block (hash=%s) in locator was not found",
209  __func__, locator_tip_hash.ToString());
210  return;
211  }
212 
213  // This checks that ChainStateFlushed callbacks are received after BlockConnected. The check may fail
214  // immediately after the sync thread catches up and sets m_synced. Consider the case where
215  // there is a reorg and the blocks on the stale branch are in the ValidationInterface queue
216  // backlog even after the sync thread has caught up to the new chain tip. In this unlikely
217  // event, log a warning and let the queue clear.
218  const CBlockIndex* best_block_index = m_best_block_index.load();
219  if (best_block_index->GetAncestor(locator_tip_index->nHeight) != locator_tip_index) {
220  LogPrintf("%s: WARNING: Locator contains block (hash=%s) not on known best " /* Continued */
221  "chain (tip=%s); not writing index locator\n",
222  __func__, locator_tip_hash.ToString(),
223  best_block_index->GetBlockHash().ToString());
224  return;
225  }
226 
227  if (!GetDB().WriteBestBlock(locator)) {
228  error("%s: Failed to write locator to disk", __func__);
229  }

232 bool BaseIndex::BlockUntilSyncedToCurrentChain()
233 {
234  AssertLockNotHeld(cs_main);
235 
236  if (!m_synced) {
237  return false;
238  }
239 
240  {
241  // Skip the queue-draining stuff if we know we're caught up with
242  // chainActive.Tip().
243  LOCK(cs_main);
244  const CBlockIndex* chain_tip = chainActive.Tip();
245  const CBlockIndex* best_block_index = m_best_block_index.load();
246  if (best_block_index->GetAncestor(chain_tip->nHeight) == chain_tip) {
247  return true;
248  }
249  }
250 
251  LogPrintf("%s: %s is catching up on block notifications\n", __func__, GetName());
252  SyncWithValidationInterfaceQueue();
253  return true;
