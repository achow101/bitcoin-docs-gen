20 // Assumption: We assume the floating-point types to fulfill the requirements of
21 // IEC 559 (IEEE 754) standard.
22 // Example(s): Floating-point division by zero in ConnectBlock, CreateTransaction
23 // and EstimateMedianVal.
24 static_assert(std::numeric_limits<float>::is_iec559, "IEEE 754 float assumed");

32 // Assumption: We assume floating-point widths.
